  73.   73  * Provides Drupal stream wrapper registry.
  74.   74  *
  75.   75  * A stream wrapper is an abstraction of a file system that allows Drupal to
  76.   76  * use the same set of methods to access both local files and remote resources.
  77.   77  *

  79.   79  * in PHP. PHP's internal stream_get_wrappers() doesn't return the class
  80.   80  * registered to handle a stream, which we need to be able to find the handler
  81.   81  * for class instantiation.
  82.   82  *

  84.   84  * existing scheme will be unregistered and replaced with the specified class.
  85.   85  *
  86.   86  * A stream is referenced as "scheme://target".
  87.   87  *

  89.   89  * wrappers that are appropriate for particular usage. For example, this returns
  90.   90  * only stream wrappers that use local file storage:
  91.   91  * @code

  93.   93  * @endcode
  94.   94  *
  95.   95  * The $filter parameter can only filter to types containing a particular flag.
  96.   96  * In some cases, you may want to filter to types that do not contain a
  97.   97  * particular flag. For example, you may want to retrieve all stream wrappers
  98.   98  * that are not writable, or all stream wrappers that are not local. PHP's
  99.   99  * array_diff_key() function can be used to help with this. For example, this
  100.  100  * returns only stream wrappers that do not use local file storage:
  101.  101  * @code
  102.  102  *   $remote_stream_wrappers = array_diff_key(file_get_stream_wrappers(STREAM_WRAPPERS_ALL), file_get_stream_wrappers(STREAM_WRAPPERS_LOCAL));
  103.  103  * @endcode
  104.  104  *
  105.  105  * @param $filter
  106.  106  *   (Optional) Filters out all types except those with an on bit for each on
  107.  107  *   bit in $filter. For example, if $filter is STREAM_WRAPPERS_WRITE_VISIBLE,
  108.  108  *   which is equal to (STREAM_WRAPPERS_READ | STREAM_WRAPPERS_WRITE |
  109.  109  *   STREAM_WRAPPERS_VISIBLE), then only stream wrappers with all three of these
  110.  110  *   bits set are returned. Defaults to STREAM_WRAPPERS_ALL, which returns all
  111.  111  *   registered stream wrappers.
  112.  112  *
  113.  113  * @return
  114.  114  *   An array keyed by scheme, with values containing an array of information
  115.  115  *   about the stream wrapper, as returned by hook_stream_wrappers(). If $filter
  116.  116  *   is omitted or set to STREAM_WRAPPERS_ALL, the entire Drupal stream wrapper
  117.  117  *   registry is returned. Otherwise only the stream wrappers whose 'type'
  118.  118  *   bitmask has an on bit for each bit specified in $filter are returned.
  119.  119  *
  120.  120  * @see hook_stream_wrappers()
  121.  121  * @see hook_stream_wrappers_alter()
  122.  122  */
  123.  123 function file_get_stream_wrappers($filter = STREAM_WRAPPERS_ALL) {
  124.  124   $wrappers_storage = &drupal_static(__FUNCTION__);
  125.  125 
  126.  126   if (!isset($wrappers_storage)) {
  127.  127     $wrappers = module_invoke_all('stream_wrappers');
  128.  128     foreach ($wrappers as $scheme => $info) {
  129.  129       // Add defaults.
  130.  130       $wrappers[$scheme] += array('type' => STREAM_WRAPPERS_NORMAL);
  131.  131     }
  132.  132     drupal_alter('stream_wrappers', $wrappers);
  133.  133     $existing = stream_get_wrappers();
  134.  134     foreach ($wrappers as $scheme => $info) {
  135.  135       // We only register classes that implement our interface.
  136.  136       if (in_array('DrupalStreamWrapperInterface', class_implements($info['class']), TRUE)) {
  137.  137         // Record whether we are overriding an existing scheme.
  138.  138         if (in_array($scheme, $existing, TRUE)) {
  139.  139           $wrappers[$scheme]['override'] = TRUE;
  140.  140           stream_wrapper_unregister($scheme);
  141.  141         }
  142.  142         else {
  143.  143           $wrappers[$scheme]['override'] = FALSE;
  144.  144         }
  145.  145         if (($info['type'] & STREAM_WRAPPERS_LOCAL) == STREAM_WRAPPERS_LOCAL) {
  146.  146           stream_wrapper_register($scheme, $info['class']);
  147.  147         }
  148.  148         else {
  149.  149           stream_wrapper_register($scheme, $info['class'], STREAM_IS_URL);
  150.  150         }
  151.  151       }
  152.  152       // Pre-populate the static cache with the filters most typically used.
  153.  153       $wrappers_storage[STREAM_WRAPPERS_ALL][$scheme] = $wrappers[$scheme];
  154.  154       if (($info['type'] & STREAM_WRAPPERS_WRITE_VISIBLE) == STREAM_WRAPPERS_WRITE_VISIBLE) {
  155.  155         $wrappers_storage[STREAM_WRAPPERS_WRITE_VISIBLE][$scheme] = $wrappers[$scheme];
  156.  156       }
  157.  157     }
  158.  158   }
  159.  159 
  160.  160   if (!isset($wrappers_storage[$filter])) {
  161.  161     $wrappers_storage[$filter] = array();
  162.  162     foreach ($wrappers_storage[STREAM_WRAPPERS_ALL] as $scheme => $info) {
  163.  163       // Bit-wise filter.
  164.  164       if (($info['type'] & $filter) == $filter) {
  165.  165         $wrappers_storage[$filter][$scheme] = $info;
  166.  166       }
  167.  167     }
  168.  168   }
  169.  169 
  170.  170   return $wrappers_storage[$filter];
  171.  171 }

  367.  367  *   '/', nothing is done and the same string is returned. If a stream wrapper
  368.  368  *   could not be found to generate an external URL, then FALSE is returned.
  369.  369  *
  370.  370  * @see http://drupal.org/node/515192
  371.  371  */
  372.  372 function file_create_url($uri) {
  373.  373   // Allow the URI to be altered, e.g. to serve a file from a CDN or static
  374.  374   // file server.
  375.  375   drupal_alter('file_url', $uri);
  376.  376 
  377.  377   $scheme = file_uri_scheme($uri);
  378.  378 
  379.  379   if (!$scheme) {
  380.  380     // Allow for:
  381.  381     // - root-relative URIs (e.g. /foo.jpg in http://example.com/foo.jpg)
  382.  382     // - protocol-relative URIs (e.g. //bar.jpg, which is expanded to
  383.  383     //   http://example.com/bar.jpg by the browser when viewing a page over
  384.  384     //   HTTP and to https://example.com/bar.jpg when viewing a HTTPS page)
  385.  385     // Both types of relative URIs are characterized by a leading slash, hence
  386.  386     // we can use a single check.
  387.  387     if (drupal_substr($uri, 0, 1) == '/') {
  388.  388       return $uri;
  389.  389     }
  390.  390     else {
  391.  391       // If this is not a properly formatted stream, then it is a shipped file.
  392.  392       // Therefore, return the urlencoded URI with the base URL prepended.
  393.  393       return $GLOBALS['base_url'] . '/' . drupal_encode_path($uri);
  394.  394     }
  395.  395   }
  396.  396   elseif ($scheme == 'http' || $scheme == 'https') {
  397.  397     // Check for HTTP so that we don't have to implement getExternalUrl() for
  398.  398     // the HTTP wrapper.
  399.  399     return $uri;
  400.  400   }
  401.  401   else {
  402.  402     // Attempt to return an external URL using the appropriate wrapper.
  403.  403     if ($wrapper = file_stream_wrapper_get_instance_by_uri($uri)) {
  404.  404       return $wrapper->getExternalUrl();
  405.  405     }
  406.  406     else {
  407.  407       return FALSE;
  408.  408     }
  409.  409   }
  410.  410 }

  425.  425  *
  426.  426  * @return
  427.  427  *   TRUE if the directory exists (or was created) and is writable. FALSE

  430.  430 function file_prepare_directory(&$directory, $options = FILE_MODIFY_PERMISSIONS) {
  431.  431   if (!file_stream_wrapper_valid_scheme(file_uri_scheme($directory))) {
  432.  432     // Only trim if we're not dealing with a stream.
  433.  433     $directory = rtrim($directory, '/\\');

  437.  437   if (!is_dir($directory)) {

  439.  439     // permissions.

  442.  442     }
  443.  443     return FALSE;
  444.  444   }

  446.  446   $writable = is_writable($directory);
  447.  447   if (!$writable && ($options & FILE_MODIFY_PERMISSIONS)) {
  448.  448     return drupal_chmod($directory);
  449.  449   }
  450.  450

  702.  702  * The file_delete() function is typically called after removing a file usage
  703.  703  * to remove the record from the file_managed table and delete the file itself.

  711.  711  *   be omitted if all module references to a file are being deleted.
  712.  712  * @param $id
  713.  713  *   (optional) The unique, numeric ID of the object containing the referenced
  714.  714  *   file. May be omitted if all module references to a file are being deleted.
  715.  715  * @param $count
  716.  716  *   (optional) The number of references to delete from the object. Defaults to
  717.  717  *   1. 0 may be specified to delete all references to the file within a
  718.  718  *   specific object.
  719.  719  *
  720.  720  * @see file_usage_add()
  721.  721  * @see file_usage_list()
  722.  722  * @see file_delete()
  723.  723  */
  724.  724 function file_usage_delete(stdClass $file, $module, $type = NULL, $id = NULL, $count = 1) {
  725.  725   // Delete rows that have a exact or less value to prevent empty rows.
  726.  726   $query = db_delete('file_usage')
  727.  727     ->condition('module', $module)
  728.  728     ->condition('fid', $file->fid);
  729.  729   if ($type && $id) {
  730.  730     $query
  731.  731       ->condition('type', $type)
  732.  732       ->condition('id', $id);
  733.  733   }
  734.  734   if ($count) {
  735.  735     $query->condition('count', $count, '<=');
  736.  736   }
  737.  737   $result = $query->execute();
  738.  738 
  739.  739   // If the row has more than the specified count decrement it by that number.
