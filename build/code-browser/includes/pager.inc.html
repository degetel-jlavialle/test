  190. 190   if (!isset($page_array[$element])) {

  196. 196 /**
  197. 197  * Initializes a pager for theme('pager').
  198. 198  *
  199. 199  * This function sets up the necessary global variables so that future calls
  200. 200  * to theme('pager') will render a pager that correctly corresponds to the
  201. 201  * items being displayed.
  202. 202  *
  203. 203  * If the items being displayed result from a database query performed using
  204. 204  * Drupal's database API, and if you have control over the construction of the
  205. 205  * database query, you do not need to call this function directly; instead, you
  206. 206  * can simply extend the query object with the 'PagerDefault' extender before
  207. 207  * executing it. For example:

  210. 210  * @endcode

  212. 212  * However, if you are using a different method for generating the items to be
  213. 213  * paged through, then you should call this function in preparation.
  214. 214  *
  215. 215  * The following example shows how this function can be used in a page callback
  216. 216  * that invokes an external datastore with an SQL-like syntax:
  217. 217  * @code
  218. 218  *   // First find the total number of items and initialize the pager.
  219. 219  *   $where = "status = 1";
  220. 220  *   $total = mymodule_select("SELECT COUNT(*) FROM data " . $where)->result();
  221. 221  *   $num_per_page = variable_get('mymodule_num_per_page', 10);
  222. 222  *   $page = pager_default_initialize($total, $num_per_page);
  223. 223  *
  224. 224  *   // Next, retrieve and display the items for the current page.
  225. 225  *   $offset = $num_per_page * $page;
  226. 226  *   $result = mymodule_select("SELECT * FROM data " . $where . " LIMIT %d, %d", $offset, $num_per_page)->fetchAll();
  227. 227  *   $output = theme('mymodule_results', array('result' => $result));
  228. 228  *
  229. 229  *   // Finally, display the pager controls, and return.
  230. 230  *   $output .= theme('pager');
  231. 231  *   return $output;
  232. 232  * @endcode
  233. 233  *
  234. 234  * A second example involves a page callback that invokes an external search
  235. 235  * service where the total number of matching results is provided as part of
  236. 236  * the returned set (so that we do not need a separate query in order to obtain
  237. 237  * this information). Here, we call pager_find_page() to calculate the desired
  238. 238  * offset before the search is invoked:
  239. 239  * @code
  240. 240  *   // Perform the query, using the requested offset from pager_find_page().
  241. 241  *   // This comes from a URL parameter, so here we are assuming that the URL
  242. 242  *   // parameter corresponds to an actual page of results that will exist
  243. 243  *   // within the set.
  244. 244  *   $page = pager_find_page();
  245. 245  *   $num_per_page = variable_get('mymodule_num_per_page', 10);
  246. 246  *   $offset = $num_per_page * $page;
  247. 247  *   $result = mymodule_remote_search($keywords, $offset, $num_per_page);
  248. 248  *
  249. 249  *   // Now that we have the total number of results, initialize the pager.
  250. 250  *   pager_default_initialize($result->total, $num_per_page);
  251. 251  *
  252. 252  *   // Display the search results.
  253. 253  *   $output = theme('search_results', array('results' => $result->data, 'type' => 'remote'));
  254. 254  *
  255. 255  *   // Finally, display the pager controls, and return.
  256. 256  *   $output .= theme('pager');
  257. 257  *   return $output;
  258. 258  * @endcode
  259. 259  *
  260. 260  * @param $total
  261. 261  *  The total number of items to be paged.
  262. 262  * @param $limit
  263. 263  *  The number of items the calling code will display per page.
  264. 264  * @param $element
  265. 265  *  An optional integer to distinguish between multiple pagers on one page.
  266. 266  *
  267. 267  * @return
  268. 268  *   The number of the current page, within the pager represented by $element.
  269. 269  *   This is determined from the URL query parameter $_GET['page'], or 0 by
  270. 270  *   default. However, if a page that does not correspond to the actual range
  271. 271  *   of the result set was requested, this function will return the closest
  272. 272  *   page actually within the result set.
  273. 273  */
  274. 274 function pager_default_initialize($total, $limit, $element = 0) {
  275. 275   global $pager_page_array, $pager_total, $pager_total_items, $pager_limits;
  276. 276 
  277. 277   $page = pager_find_page($element);
  278. 278 
  279. 279   // We calculate the total of pages as ceil(items / limit).
  280. 280   $pager_total_items[$element] = $total;
  281. 281   $pager_total[$element] = ceil($pager_total_items[$element] / $limit);
  282. 282   $pager_page_array[$element] = max(0, min($page, ((int) $pager_total[$element]) - 1));
  283. 283   $pager_limits[$element] = $limit;
  284. 284   return $pager_page_array[$element];
  285. 285 }

  312. 312  *   - element: An optional integer to distinguish between multiple pagers on
  313. 313  *     one page.

  315. 315  *     the pager links.

  317. 317  *

  320. 320 function theme_pager($variables) {
  321. 321   $tags = $variables['tags'];
  322. 322   $element = $variables['element'];
  323. 323   $parameters = $variables['parameters'];
  324. 324   $quantity = $variables['quantity'];
  325. 325   global $pager_page_array, $pager_total;
  326. 326 
  327. 327   // Calculate various markers within this pager piece:
  328. 328   // Middle is used to "center" pages around the current page.
  329. 329   $pager_middle = ceil($quantity / 2);
  330. 330   // current is the page we are currently paged to
  331. 331   $pager_current = $pager_page_array[$element] + 1;
  332. 332   // first is the first page listed by this pager piece (re quantity)
  333. 333   $pager_first = $pager_current - $pager_middle + 1;
  334. 334   // last is the last page listed by this pager piece (re quantity)
  335. 335   $pager_last = $pager_current + $quantity - $pager_middle;
  336. 336   // max is the maximum page number
  337. 337   $pager_max = $pager_total[$element];
  338. 338   // End of marker calculations.
  339. 339 
  340. 340   // Prepare for generation loop.
  341. 341   $i = $pager_first;
  342. 342   if ($pager_last > $pager_max) {
  343. 343     // Adjust "center" if at end of query.
  344. 344     $i = $i + ($pager_max - $pager_last);
  345. 345     $pager_last = $pager_max;
  346. 346   }
  347. 347   if ($i <= 0) {
  348. 348     // Adjust "center" if at start of query.
  349. 349     $pager_last = $pager_last + (1 - $i);
  350. 350     $i = 1;
  351. 351   }
  352. 352   // End of generation loop preparation.
  353. 353 
  354. 354   $li_first = theme('pager_first', array('text' => (isset($tags[0]) ? $tags[0] : t('« first')), 'element' => $element, 'parameters' => $parameters));
  355. 355   $li_previous = theme('pager_previous', array('text' => (isset($tags[1]) ? $tags[1] : t('‹ previous')), 'element' => $element, 'interval' => 1, 'parameters' => $parameters));
  356. 356   $li_next = theme('pager_next', array('text' => (isset($tags[3]) ? $tags[3] : t('next ›')), 'element' => $element, 'interval' => 1, 'parameters' => $parameters));
  357. 357   $li_last = theme('pager_last', array('text' => (isset($tags[4]) ? $tags[4] : t('last »')), 'element' => $element, 'parameters' => $parameters));
  358. 358 
  359. 359   if ($pager_total[$element] > 1) {
  360. 360     if ($li_first) {
  361. 361       $items[] = array(
  362. 362         'class' => array('pager-first'),
  363. 363         'data' => $li_first,
  364. 364       );
  365. 365     }
  366. 366     if ($li_previous) {
  367. 367       $items[] = array(
  368. 368         'class' => array('pager-previous'),
  369. 369         'data' => $li_previous,
  370. 370       );
  371. 371     }
  372. 372 
  373. 373     // When there is more than one page, create the pager list.
  374. 374     if ($i != $pager_max) {
  375. 375       if ($i > 1) {
  376. 376         $items[] = array(
  377. 377           'class' => array('pager-ellipsis'),
  378. 378           'data' => '…',
