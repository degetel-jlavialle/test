  101.  101     $updates = array();
  102.  102 
  103.  103     foreach (module_list() as $loaded_module) {
  104.  104       $updates[$loaded_module] = array();
  105.  105     }
  106.  106 
  107.  107     // Prepare regular expression to match all possible defined hook_update_N().
  108.  108     $regexp = '/^(?P<module>.+)_update_(?P<version>\d+)$/';
  109.  109     $functions = get_defined_functions();
  110.  110     // Narrow this down to functions ending with an integer, since all
  111.  111     // hook_update_N() functions end this way, and there are other
  112.  112     // possible functions which match '_update_'. We use preg_grep() here
  113.  113     // instead of foreaching through all defined functions, since the loop
  114.  114     // through all PHP functions can take significant page execution time
  115.  115     // and this function is called on every administrative page via
  116.  116     // system_requirements().
  117.  117     foreach (preg_grep('/_\d+$/', $functions['user']) as $function) {
  118.  118       // If this function is a module update function, add it to the list of
  119.  119       // module updates.
  120.  120       if (preg_match($regexp, $function, $matches)) {
  121.  121         $updates[$matches['module']][] = $matches['version'];
  122.  122       }
  123.  123     }
  124.  124     // Ensure that updates are applied in numerical order.
  125.  125     foreach ($updates as &$module_updates) {
  126.  126       sort($module_updates, SORT_NUMERIC);
  127.  127     }
  128.  128   }

  130.  130 }

  584.  584 /**

  591.  591   $default_settings = 'sites/default/default.settings.php';
  592.  592   drupal_static_reset('conf_path');
  593.  593   $settings_file = conf_path(FALSE) . '/' . $prefix . 'settings.php';
  594.  594 
  595.  595   // Build list of setting names and insert the values into the global namespace.
  596.  596   $keys = array();
  597.  597   foreach ($settings as $setting => $data) {
  598.  598     $GLOBALS[$setting] = $data['value'];
  599.  599     $keys[] = $setting;
  600.  600   }
  601.  601 
  602.  602   $buffer = NULL;
  603.  603   $first = TRUE;
  604.  604   if ($fp = fopen(DRUPAL_ROOT . '/' . $default_settings, 'r')) {
  605.  605     // Step line by line through settings.php.
  606.  606     while (!feof($fp)) {
  607.  607       $line = fgets($fp);
  608.  608       if ($first && substr($line, 0, 5) != '<?php') {
  609.  609         $buffer = "<?php\n\n";
  610.  610       }
  611.  611       $first = FALSE;
  612.  612       // Check for constants.
  613.  613       if (substr($line, 0, 7) == 'define(') {
  614.  614         preg_match('/define\(\s*[\'"]([A-Z_-]+)[\'"]\s*,(.*?)\);/', $line, $variable);
  615.  615         if (in_array($variable[1], $keys)) {
  616.  616           $setting = $settings[$variable[1]];
  617.  617           $buffer .= str_replace($variable[2], " '" . $setting['value'] . "'", $line);
  618.  618           unset($settings[$variable[1]]);
  619.  619           unset($settings[$variable[2]]);
  620.  620         }
  621.  621         else {
  622.  622           $buffer .= $line;
  623.  623         }
  624.  624       }
  625.  625       // Check for variables.
  626.  626       elseif (substr($line, 0, 1) == '$') {
  627.  627         preg_match('/\$([^ ]*) /', $line, $variable);
  628.  628         if (in_array($variable[1], $keys)) {
  629.  629           // Write new value to settings.php in the following format:
  630.  630           //    $'setting' = 'value'; // 'comment'
  631.  631           $setting = $settings[$variable[1]];
  632.  632           $buffer .= '$' . $variable[1] . " = " . var_export($setting['value'], TRUE) . ";" . (!empty($setting['comment']) ? ' // ' . $setting['comment'] . "\n" : "\n");
  633.  633           unset($settings[$variable[1]]);
  634.  634         }
  635.  635         else {
  636.  636           $buffer .= $line;
  637.  637         }
  638.  638       }
  639.  639       else {
  640.  640         $buffer .= $line;

  642.  642     }

  644.  644 
  645.  645     // Add required settings that were missing from settings.php.
  646.  646     foreach ($settings as $setting => $data) {
  647.  647       if ($data['required']) {
  648.  648         $buffer .= "\$$setting = " . var_export($data['value'], TRUE) . ";\n";
  649.  649       }
  650.  650     }
  651.  651 
  652.  652     $fp = fopen(DRUPAL_ROOT . '/' . $settings_file, 'w');
  653.  653     if ($fp && fwrite($fp, $buffer) === FALSE) {
  654.  654       throw new Exception(st('Failed to modify %settings. Verify the file permissions.', array('%settings' => $settings_file)));
  655.  655     }
  656.  656   }
  657.  657   else {
  658.  658     throw new Exception(st('Failed to open %settings. Verify the file permissions.', array('%settings' => $default_settings)));
  659.  659   }
  660.  660 }
