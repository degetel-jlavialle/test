  124.  124  *
  125.  125  * @param $languages
  126.  126  *   An array of language objects for enabled languages ordered by weight.
  127.  127  *
  128.  128  * @return
  129.  129  *   A valid language code on success, FALSE otherwise.
  130.  130  */
  131.  131 function locale_language_from_browser($languages) {
  132.  132   if (empty($_SERVER['HTTP_ACCEPT_LANGUAGE'])) {
  133.  133     return FALSE;
  134.  134   }
  135.  135 
  136.  136   // The Accept-Language header contains information about the language
  137.  137   // preferences configured in the user's browser / operating system.
  138.  138   // RFC 2616 (section 14.4) defines the Accept-Language header as follows:
  139.  139   //   Accept-Language = "Accept-Language" ":"
  140.  140   //                  1#( language-range [ ";" "q" "=" qvalue ] )
  141.  141   //   language-range  = ( ( 1*8ALPHA *( "-" 1*8ALPHA ) ) | "*" )
  142.  142   // Samples: "hu, en-us;q=0.66, en;q=0.33", "hu,en-us;q=0.5"
  143.  143   $browser_langcodes = array();
  144.  144   if (preg_match_all('@(?<=[, ]|^)([a-zA-Z-]+|\*)(?:;q=([0-9.]+))?(?:$|\s*,\s*)@', trim($_SERVER['HTTP_ACCEPT_LANGUAGE']), $matches, PREG_SET_ORDER)) {
  145.  145     foreach ($matches as $match) {
  146.  146       // We can safely use strtolower() here, tags are ASCII.
  147.  147       // RFC2616 mandates that the decimal part is no more than three digits,
  148.  148       // so we multiply the qvalue by 1000 to avoid floating point comparisons.
  149.  149       $langcode = strtolower($match[1]);
  150.  150       $qvalue = isset($match[2]) ? (float) $match[2] : 1;
  151.  151       $browser_langcodes[$langcode] = (int) ($qvalue * 1000);
  152.  152     }
  153.  153   }
  154.  154 
  155.  155   // We should take pristine values from the HTTP headers, but Internet Explorer
  156.  156   // from version 7 sends only specific language tags (eg. fr-CA) without the
  157.  157   // corresponding generic tag (fr) unless explicitly configured. In that case,
  158.  158   // we assume that the lowest value of the specific tags is the value of the
  159.  159   // generic language to be as close to the HTTP 1.1 spec as possible.
  160.  160   // See http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.4 and
  161.  161   // http://blogs.msdn.com/b/ie/archive/2006/10/17/accept-language-header-for-internet-explorer-7.aspx
  162.  162   asort($browser_langcodes);
  163.  163   foreach ($browser_langcodes as $langcode => $qvalue) {
  164.  164     $generic_tag = strtok($langcode, '-');
  165.  165     if (!isset($browser_langcodes[$generic_tag])) {
  166.  166       $browser_langcodes[$generic_tag] = $qvalue;
  167.  167     }
  168.  168   }
  169.  169 
  170.  170   // Find the enabled language with the greatest qvalue, following the rules
  171.  171   // of RFC 2616 (section 14.4). If several languages have the same qvalue,
  172.  172   // prefer the one with the greatest weight.
  173.  173   $best_match_langcode = FALSE;
  174.  174   $max_qvalue = 0;
  175.  175   foreach ($languages as $langcode => $language) {
  176.  176     // Language tags are case insensitive (RFC2616, sec 3.10).
  177.  177     $langcode = strtolower($langcode);
  178.  178 
  179.  179     // If nothing matches below, the default qvalue is the one of the wildcard
  180.  180     // language, if set, or is 0 (which will never match).
  181.  181     $qvalue = isset($browser_langcodes['*']) ? $browser_langcodes['*'] : 0;
  182.  182 
  183.  183     // Find the longest possible prefix of the browser-supplied language
  184.  184     // ('the language-range') that matches this site language ('the language tag').
  185.  185     $prefix = $langcode;
  186.  186     do {
  187.  187       if (isset($browser_langcodes[$prefix])) {
  188.  188         $qvalue = $browser_langcodes[$prefix];
  189.  189         break;
  190.  190       }
  191.  191     }
  192.  192     while ($prefix = substr($prefix, 0, strrpos($prefix, '-')));
  193.  193 
  194.  194     // Find the best match.
  195.  195     if ($qvalue > $max_qvalue) {
  196.  196       $best_match_langcode = $language->language;
  197.  197       $max_qvalue = $qvalue;
  198.  198     }
  199.  199   }
  200.  200 
  201.  201   return $best_match_langcode;
  202.  202 }

  264.  264 function locale_language_from_url($languages) {
  265.  265   $language_url = FALSE;
  266.  266 
  267.  267   if (!language_negotiation_get_any(LOCALE_LANGUAGE_NEGOTIATION_URL)) {
  268.  268     return $language_url;
  269.  269   }
  270.  270 
  271.  271   switch (variable_get('locale_language_negotiation_url_part', LOCALE_LANGUAGE_NEGOTIATION_URL_PREFIX)) {
  272.  272     case LOCALE_LANGUAGE_NEGOTIATION_URL_PREFIX:
  273.  273       // $_GET['q'] might not be available at this time, because
  274.  274       // path initialization runs after the language bootstrap phase.
  275.  275       list($language, $_GET['q']) = language_url_split_prefix(isset($_GET['q']) ? $_GET['q'] : NULL, $languages);
  276.  276       if ($language !== FALSE) {
  277.  277         $language_url = $language->language;
  278.  278       }
  279.  279       break;
  280.  280 
  281.  281     case LOCALE_LANGUAGE_NEGOTIATION_URL_DOMAIN:
  282.  282       // Get only the host, not the port.
  283.  283       $http_host= $_SERVER['HTTP_HOST'];
  284.  284       if (strpos($http_host, ':') !== FALSE) {
  285.  285         $http_host_tmp = explode(':', $http_host);
  286.  286         $http_host = current($http_host_tmp);
  287.  287       }
  288.  288       foreach ($languages as $language) {
  289.  289         // Skip check if the language doesn't have a domain.
  290.  290         if ($language->domain) {
  291.  291           // Only compare the domains not the protocols or ports.
  292.  292           // Remove protocol and add http:// so parse_url works
  293.  293           $host = 'http://' . str_replace(array('http://', 'https://'), '', $language->domain);
  294.  294           $host = parse_url($host, PHP_URL_HOST);
  295.  295           if ($http_host == $host) {
  296.  296             $language_url = $language->language;
  297.  297             break;
  298.  298           }
  299.  299         }
  300.  300       }
  301.  301       break;
  302.  302   }
  303.  303 
  304.  304   return $language_url;
  305.  305 }

  307.  307 /**
  308.  308  * Determines the language to be assigned to URLs when none is detected.
  309.  309  *
  310.  310  * The language negotiation process has a fallback chain that ends with the
  311.  311  * default language provider. Each built-in language type has a separate
  312.  312  * initialization:
  313.  313  * - Interface language, which is the only configurable one, always gets a valid
  314.  314  *   value. If no request-specific language is detected, the default language
  315.  315  *   will be used.
  316.  316  * - Content language merely inherits the interface language by default.
  317.  317  * - URL language is detected from the requested URL and will be used to rewrite
  318.  318  *   URLs appearing in the page being rendered. If no language can be detected,
  319.  319  *   there are two possibilities:
  320.  320  *   - If the default language has no configured path prefix or domain, then the
  321.  321  *     default language is used. This guarantees that (missing) URL prefixes are
  322.  322  *     preserved when navigating through the site.
  323.  323  *   - If the default language has a configured path prefix or domain, a
  324.  324  *     requested URL having an empty prefix or domain is an anomaly that must be
  325.  325  *     fixed. This is done by introducing a prefix or domain in the rendered
  326.  326  *     page matching the detected interface language.
  327.  327  *
  328.  328  * @param $languages
  329.  329  *   (optional) An array of valid language objects. This is passed by
  330.  330  *   language_provider_invoke() to every language provider callback, but it is
  331.  331  *   not actually needed here. Defaults to NULL.
  332.  332  * @param $language_type
  333.  333  *   (optional) The language type to fall back to. Defaults to the interface
  334.  334  *   language.
  335.  335  *
  336.  336  * @return
  337.  337  *   A valid language code.
  338.  338  */
  339.  339 function locale_language_url_fallback($language = NULL, $language_type = LANGUAGE_TYPE_INTERFACE) {
  340.  340   $default = language_default();
  341.  341   $prefix = (variable_get('locale_language_negotiation_url_part', LOCALE_LANGUAGE_NEGOTIATION_URL_PREFIX) == LOCALE_LANGUAGE_NEGOTIATION_URL_PREFIX);
  342.  342 
  343.  343   // If the default language is not configured to convey language information,
  344.  344   // a missing URL language information indicates that URL language should be
  345.  345   // the default one, otherwise we fall back to an already detected language.
  346.  346   if (($prefix && empty($default->prefix)) || (!$prefix && empty($default->domain))) {
  347.  347     return $default->language;
  348.  348   }
  349.  349   else {
  350.  350     return $GLOBALS[$language_type]->language;
  351.  351   }
  352.  352 }

  674.  674 /**

  676.  676  *
  677.  677  * @param $op
  678.  678  *   Storage operation type: db-store or mem-store.
  679.  679  * @param $file
  680.  680  *   Drupal file object corresponding to the PO file to import.
  681.  681  * @param $mode
  682.  682  *   Should existing translations be replaced LOCALE_IMPORT_KEEP or
  683.  683  *   LOCALE_IMPORT_OVERWRITE.
  684.  684  * @param $lang
  685.  685  *   Language code.
  686.  686  * @param $group
  687.  687  *   Text group to import PO file into (eg. 'default' for interface
  688.  688  *   translations).
  689.  689  */
  690.  690 function _locale_import_read_po($op, $file, $mode = NULL, $lang = NULL, $group = 'default') {
  691.  691 
  692.  692   // The file will get closed by PHP on returning from this function.
  693.  693   $fd = fopen($file->uri, 'rb');
  694.  694   if (!$fd) {
  695.  695     _locale_import_message('The translation import failed, because the file %filename could not be read.', $file);
  696.  696     return FALSE;
  697.  697   }
  698.  698 
  699.  699   /*
  700.  700    * The parser context. Can be:
  701.  701    *  - 'COMMENT' (#)
  702.  702    *  - 'MSGID' (msgid)
  703.  703    *  - 'MSGID_PLURAL' (msgid_plural)
  704.  704    *  - 'MSGCTXT' (msgctxt)
  705.  705    *  - 'MSGSTR' (msgstr or msgstr[])
  706.  706    *  - 'MSGSTR_ARR' (msgstr_arg)
  707.  707    */
  708.  708   $context = 'COMMENT';
  709.  709 
  710.  710   // Current entry being read.
  711.  711   $current = array();
  712.  712 
  713.  713   // Current plurality for 'msgstr[]'.
  714.  714   $plural = 0;
  715.  715 
  716.  716   // Current line.
  717.  717   $lineno = 0;
  718.  718 
  719.  719   while (!feof($fd)) {
  720.  720     // A line should not be longer than 10 * 1024.
  721.  721     $line = fgets($fd, 10 * 1024);
  722.  722 
  723.  723     if ($lineno == 0) {
  724.  724       // The first line might come with a UTF-8 BOM, which should be removed.
  725.  725       $line = str_replace("\xEF\xBB\xBF", '', $line);
  726.  726     }
  727.  727
