  299. 299   public function rollback($savepoint_name = 'drupal_transaction') {
  300. 300     if ($this->savepointSupport) {
  301. 301       return parent::rollBack($savepoint_name);
  302. 302     }
  303. 303 
  304. 304     if (!$this->inTransaction()) {
  305. 305       throw new DatabaseTransactionNoActiveException();
  306. 306     }
  307. 307     // A previous rollback to an earlier savepoint may mean that the savepoint
  308. 308     // in question has already been rolled back.
  309. 309     if (!in_array($savepoint_name, $this->transactionLayers)) {
  310. 310       return;
  311. 311     }
  312. 312 
  313. 313     // We need to find the point we're rolling back to, all other savepoints
  314. 314     // before are no longer needed.
  315. 315     while ($savepoint = array_pop($this->transactionLayers)) {
  316. 316       if ($savepoint == $savepoint_name) {
  317. 317         // Mark whole stack of transactions as needed roll back.
  318. 318         $this->willRollback = TRUE;
  319. 319         // If it is the last the transaction in the stack, then it is not a
  320. 320         // savepoint, it is the transaction itself so we will need to roll back
  321. 321         // the transaction rather than a savepoint.
  322. 322         if (empty($this->transactionLayers)) {
  323. 323           break;
  324. 324         }
  325. 325         return;
  326. 326       }
  327. 327     }
  328. 328     if ($this->supportsTransactions()) {
  329. 329       PDO::rollBack();
  330. 330     }
  331. 331   }

  393. 393   /**
  394. 394    * SQLite specific implementation of getStatement().
  395. 395    *
  396. 396    * The PDO SQLite layer doesn't replace numeric placeholders in queries
  397. 397    * correctly, and this makes numeric expressions (such as COUNT(*) >= :count)
  398. 398    * fail. We replace numeric placeholders in the query ourselves to work
  399. 399    * around this bug.
  400. 400    *
  401. 401    * See http://bugs.php.net/bug.php?id=45259 for more details.
  402. 402    */
  403. 403   protected function getStatement($query, &$args = array()) {
  404. 404     if (count($args)) {
  405. 405       // Check if $args is a simple numeric array.
  406. 406       if (range(0, count($args) - 1) === array_keys($args)) {
  407. 407         // In that case, we have unnamed placeholders.
  408. 408         $count = 0;
  409. 409         $new_args = array();
  410. 410         foreach ($args as $value) {
  411. 411           if (is_float($value) || is_int($value)) {
  412. 412             if (is_float($value)) {
  413. 413               // Force the conversion to float so as not to loose precision
  414. 414               // in the automatic cast.
