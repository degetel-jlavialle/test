  10.   10  * @{
  11.   11  * Functions for Drupal's Ajax framework.
  12.   12  *
  13.   13  * Drupal's Ajax framework is used to dynamically update parts of a page's HTML
  14.   14  * based on data from the server. Upon a specified event, such as a button
  15.   15  * click, a callback function is triggered which performs server-side logic and
  16.   16  * may return updated markup, which is then replaced on-the-fly with no page
  17.   17  * refresh necessary.
  18.   18  *
  19.   19  * This framework creates a PHP macro language that allows the server to
  20.   20  * instruct JavaScript to perform actions on the client browser. When using
  21.   21  * forms, it can be used with the #ajax property.
  22.   22  * The #ajax property can be used to bind events to the Ajax framework. By
  23.   23  * default, #ajax uses 'system/ajax' as its path for submission and thus calls
  24.   24  * ajax_form_callback() and a defined #ajax['callback'] function.
  25.   25  * However, you may optionally specify a different path to request or a
  26.   26  * different callback function to invoke, which can return updated HTML or can
  27.   27  * also return a richer set of
  28.   28  * @link ajax_commands Ajax framework commands @endlink.
  29.   29  *
  30.   30  * Standard form handling is as follows:
  31.   31  *   - A form element has a #ajax property that includes #ajax['callback'] and
  32.   32  *     omits #ajax['path']. See below about using #ajax['path'] to implement
  33.   33  *     advanced use-cases that require something other than standard form
  34.   34  *     handling.
  35.   35  *   - On the specified element, Ajax processing is triggered by a change to
  36.   36  *     that element.
  37.   37  *   - The browser submits an HTTP POST request to the 'system/ajax' Drupal
  38.   38  *     path.
  39.   39  *   - The menu page callback for 'system/ajax', ajax_form_callback(), calls
  40.   40  *     drupal_process_form() to process the form submission and rebuild the
  41.   41  *     form if necessary. The form is processed in much the same way as if it
  42.   42  *     were submitted without Ajax, with the same #process functions and
  43.   43  *     validation and submission handlers called in either case, making it easy
  44.   44  *     to create Ajax-enabled forms that degrade gracefully when JavaScript is
  45.   45  *     disabled.
  46.   46  *   - After form processing is complete, ajax_form_callback() calls the
  47.   47  *     function named by #ajax['callback'], which returns the form element that
  48.   48  *     has been updated and needs to be returned to the browser, or
  49.   49  *     alternatively, an array of custom Ajax commands.
  50.   50  *   - The page delivery callback for 'system/ajax', ajax_deliver(), renders the
  51.   51  *     element returned by #ajax['callback'], and returns the JSON string
  52.   52  *     created by ajax_render() to the browser.
  53.   53  *   - The browser unserializes the returned JSON string into an array of
  54.   54  *     command objects and executes each command, resulting in the old page
  55.   55  *     content within and including the HTML element specified by
  56.   56  *     #ajax['wrapper'] being replaced by the new content returned by
  57.   57  *     #ajax['callback'], using a JavaScript animation effect specified by
  58.   58  *     #ajax['effect'].
  59.   59  *
  60.   60  * A simple example of basic Ajax use from the
  61.   61  * @link http://drupal.org/project/examples Examples module @endlink follows:
  62.   62  * @code
  63.   63  * function main_page() {
  64.   64  *   return drupal_get_form('ajax_example_simplest');
  65.   65  * }
  66.   66  *
  67.   67  * function ajax_example_simplest($form, &$form_state) {
  68.   68  *   $form = array();
  69.   69  *   $form['changethis'] = array(
  70.   70  *     '#type' => 'select',
  71.   71  *     '#options' => array(
  72.   72  *       'one' => 'one',
  73.   73  *       'two' => 'two',
  74.   74  *       'three' => 'three',
  75.   75  *     ),
  76.   76  *     '#ajax' => array(
  77.   77  *       'callback' => 'ajax_example_simplest_callback',
  78.   78  *       'wrapper' => 'replace_textfield_div',
  79.   79  *      ),
  80.   80  *   );
  81.   81 
  82.   82  *   // This entire form element will be replaced with an updated value.
  83.   83  *   $form['replace_textfield'] = array(
  84.   84  *     '#type' => 'textfield',
  85.   85  *     '#title' => t("The default value will be changed"),
  86.   86  *     '#description' => t("Say something about why you chose") . "'" .
  87.   87  *       (!empty($form_state['values']['changethis'])
  88.   88  *       ? $form_state['values']['changethis'] : t("Not changed yet")) . "'",
  89.   89  *     '#prefix' => '<div id="replace_textfield_div">',
  90.   90  *     '#suffix' => '</div>',
  91.   91  *   );
  92.   92  *   return $form;
  93.   93  * }
  94.   94  *
  95.   95  * function ajax_example_simplest_callback($form, $form_state) {
  96.   96  *   // The form has already been submitted and updated. We can return the replaced
  97.   97  *   // item as it is.
  98.   98  *   return $form['replace_textfield'];
  99.   99  * }
  100.  100  * @endcode
  101.  101  *
  102.  102  * In the above example, the 'changethis' element is Ajax-enabled. The default
  103.  103  * #ajax['event'] is 'change', so when the 'changethis' element changes,
  104.  104  * an Ajax call is made. The form is submitted and reprocessed, and then the
  105.  105  * callback is called. In this case, the form has been automatically
  106.  106  * built changing $form['replace_textfield']['#description'], so the callback
  107.  107  * just returns that part of the form.
  108.  108  *
  109.  109  * To implement Ajax handling in a form, add '#ajax' to the form
  110.  110  * definition of a field. That field will trigger an Ajax event when it is
  111.  111  * clicked (or changed, depending on the kind of field). #ajax supports
  112.  112  * the following parameters (either 'path' or 'callback' is required at least):
  113.  113  * - #ajax['callback']: The callback to invoke to handle the server side of the
  114.  114  *   Ajax event, which will receive a $form and $form_state as arguments, and
  115.  115  *   returns a renderable array (most often a form or form fragment), an HTML
  116.  116  *   string, or an array of Ajax commands. If returning a renderable array or
  117.  117  *   a string, the value will replace the original element named in
  118.  118  *   #ajax['wrapper'], and
  119.  119  *   theme_status_messages()
  120.  120  *   will be prepended to that
  121.  121  *   element. (If the status messages are not wanted, return an array
  122.  122  *   of Ajax commands instead.)
  123.  123  *   #ajax['wrapper']. If an array of Ajax commands is returned, it will be
  124.  124  *   executed by the calling code.
  125.  125  * - #ajax['path']: The menu path to use for the request. This is often omitted
  126.  126  *   and the default is used. This path should map
  127.  127  *   to a menu page callback that returns data using ajax_render(). Defaults to
  128.  128  *   'system/ajax', which invokes ajax_form_callback(), eventually calling
  129.  129  *   the function named in #ajax['callback']. If you use a custom
  130.  130  *   path, you must set up the menu entry and handle the entire callback in your
  131.  131  *   own code.
  132.  132  * - #ajax['wrapper']: The CSS ID of the area to be replaced by the content
  133.  133  *   returned by the #ajax['callback'] function. The content returned from
  134.  134  *   the callback will replace the entire element named by #ajax['wrapper'].
  135.  135  *   The wrapper is usually created using #prefix and #suffix properties in the
  136.  136  *   form. Note that this is the wrapper ID, not a CSS selector. So to replace
  137.  137  *   the element referred to by the CSS selector #some-selector on the page,
  138.  138  *   use #ajax['wrapper'] = 'some-selector', not '#some-selector'.
  139.  139  * - #ajax['effect']: The jQuery effect to use when placing the new HTML.
  140.  140  *   Defaults to no effect. Valid options are 'none', 'slide', or 'fade'.
  141.  141  * - #ajax['speed']: The effect speed to use. Defaults to 'slow'. May be
  142.  142  *   'slow', 'fast' or a number in milliseconds which represents the length
  143.  143  *   of time the effect should run.
  144.  144  * - #ajax['event']: The JavaScript event to respond to. This is normally
  145.  145  *   selected automatically for the type of form widget being used, and
  146.  146  *   is only needed if you need to override the default behavior.
  147.  147  * - #ajax['prevent']: A JavaScript event to prevent when 'event' is triggered.
  148.  148  *   Defaults to 'click' for #ajax on #type 'submit', 'button', and
  149.  149  *   'image_button'. Multiple events may be specified separated by spaces.
  150.  150  *   For example, when binding #ajax behaviors to form buttons, pressing the
  151.  151  *   ENTER key within a textfield triggers the 'click' event of the form's first
  152.  152  *   submit button. Triggering Ajax in this situation leads to problems, like
  153.  153  *   breaking autocomplete textfields. Because of that, Ajax behaviors are bound
  154.  154  *   to the 'mousedown' event on form buttons by default. However, binding to
  155.  155  *   'mousedown' rather than 'click' means that it is possible to trigger a
  156.  156  *   click by pressing the mouse, holding the mouse button down until the Ajax
  157.  157  *   request is complete and the button is re-enabled, and then releasing the
  158.  158  *   mouse button. For this case, 'prevent' can be set to 'click', so an
  159.  159  *   additional event handler is bound to prevent such a click from triggering a
  160.  160  *   non-Ajax form submission. This also prevents a textfield's ENTER press
  161.  161  *   triggering a button's non-Ajax form submission behavior.
  162.  162  * - #ajax['method']: The jQuery method to use to place the new HTML.
  163.  163  *   Defaults to 'replaceWith'. May be: 'replaceWith', 'append', 'prepend',
  164.  164  *   'before', 'after', or 'html'. See the
  165.  165  *   @link http://api.jquery.com/category/manipulation/ jQuery manipulators documentation @endlink
  166.  166  *   for more information on these methods.
  167.  167  * - #ajax['progress']: Choose either a throbber or progress bar that is
  168.  168  *   displayed while awaiting a response from the callback, and add an optional
  169.  169  *   message. Possible keys: 'type', 'message', 'url', 'interval'.
  170.  170  *   More information is available in the
  171.  171  *   @link forms_api_reference.html Form API Reference @endlink
  172.  172  *
  173.  173  * In addition to using Form API for doing in-form modification, Ajax may be
  174.  174  * enabled by adding classes to buttons and links. By adding the 'use-ajax'
  175.  175  * class to a link, the link will be loaded via an Ajax call. When using this
  176.  176  * method, the href of the link can contain '/nojs/' as part of the path. When
  177.  177  * the Ajax framework makes the request, it will convert this to '/ajax/'.
  178.  178  * The server is then able to easily tell if this request was made through an
  179.  179  * actual Ajax request or in a degraded state, and respond appropriately.
  180.  180  *
  181.  181  * Similarly, submit buttons can be given the class 'use-ajax-submit'. The
  182.  182  * form will then be submitted via Ajax to the path specified in the #action.
  183.  183  * Like the ajax-submit class above, this path will have '/nojs/' replaced with
  184.  184  * '/ajax/' so that the submit handler can tell if the form was submitted
  185.  185  * in a degraded state or not.
  186.  186  *
  187.  187  * When responding to Ajax requests, the server should do what it needs to do
  188.  188  * for that request, then create a commands array. This commands array will
  189.  189  * be converted to a JSON object and returned to the client, which will then
  190.  190  * iterate over the array and process it like a macro language.
  191.  191  *
  192.  192  * Each command item is an associative array which will be converted to a
  193.  193  * command object on the JavaScript side. $command_item['command'] is the type
  194.  194  * of command, e.g. 'alert' or 'replace', and will correspond to a method in the
  195.  195  * Drupal.ajax[command] space. The command array may contain any other data that
  196.  196  * the command needs to process, e.g. 'method', 'selector', 'settings', etc.
  197.  197  *
  198.  198  * Commands are usually created with a couple of helper functions, so they
  199.  199  * look like this:
  200.  200  * @code
  201.  201  *   $commands = array();
  202.  202  *   // Replace the content of '#object-1' on the page with 'some html here'.
  203.  203  *   $commands[] = ajax_command_replace('#object-1', 'some html here');
  204.  204  *   // Add a visual "changed" marker to the '#object-1' element.
  205.  205  *   $commands[] = ajax_command_changed('#object-1');
  206.  206  *   // Menu 'page callback' and #ajax['callback'] functions are supposed to
  207.  207  *   // return render arrays. If returning an Ajax commands array, it must be
  208.  208  *   // encapsulated in a render array structure.
  209.  209  *   return array('#type' => 'ajax', '#commands' => $commands);
  210.  210  * @endcode
  211.  211  *
  212.  212  * When returning an Ajax command array, it is often useful to have
  213.  213  * status messages rendered along with other tasks in the command array.
  214.  214  * In that case the the Ajax commands array may be constructed like this:
  215.  215  * @code
  216.  216  *   $commands = array();
  217.  217  *   $commands[] = ajax_command_replace(NULL, $output);
  218.  218  *   $commands[] = ajax_command_prepend(NULL, theme('status_messages'));
  219.  219  *   return array('#type' => 'ajax', '#commands' => $commands);
  220.  220  * @endcode
  221.  221  *
  222.  222  * See @link ajax_commands Ajax framework commands @endlink
  223.  223  */
  224.  224 
  225.  225 /**
  226.  226  * Renders a commands array into JSON.
  227.  227  *
  228.  228  * @param $commands
  229.  229  *   A list of macro commands generated by the use of ajax_command_*()
  230.  230  *   functions.
  231.  231  */
  232.  232 function ajax_render($commands = array()) {
  233.  233   // Ajax responses aren't rendered with html.tpl.php, so we have to call
  234.  234   // drupal_get_css() and drupal_get_js() here, in order to have new files added
  235.  235   // during this request to be loaded by the page. We only want to send back
  236.  236   // files that the page hasn't already loaded, so we implement simple diffing
  237.  237   // logic using array_diff_key().
  238.  238   foreach (array('css', 'js') as $type) {
  239.  239     // It is highly suspicious if $_POST['ajax_page_state'][$type] is empty,
  240.  240     // since the base page ought to have at least one JS file and one CSS file
  241.  241     // loaded. It probably indicates an error, and rather than making the page
  242.  242     // reload all of the files, instead we return no new files.
  243.  243     if (empty($_POST['ajax_page_state'][$type])) {
  244.  244       $items[$type] = array();
  245.  245     }
  246.  246     else {
  247.  247       $function = 'drupal_add_' . $type;
  248.  248       $items[$type] = $function();
  249.  249       drupal_alter($type, $items[$type]);
  250.  250       // @todo Inline CSS and JS items are indexed numerically. These can't be
  251.  251       //   reliably diffed with array_diff_key(), since the number can change
  252.  252       //   due to factors unrelated to the inline content, so for now, we strip
  253.  253       //   the inline items from Ajax responses, and can add support for them
  254.  254       //   when drupal_add_css() and drupal_add_js() are changed to use a hash
  255.  255       //   of the inline content as the array key.
  256.  256       foreach ($items[$type] as $key => $item) {
  257.  257         if (is_numeric($key)) {
  258.  258           unset($items[$type][$key]);
  259.  259         }
  260.  260       }
  261.  261       // Ensure that the page doesn't reload what it already has.
  262.  262       $items[$type] = array_diff_key($items[$type], $_POST['ajax_page_state'][$type]);
  263.  263     }
  264.  264   }
  265.  265 
  266.  266   // Render the HTML to load these files, and add AJAX commands to insert this
  267.  267   // HTML in the page. We pass TRUE as the $skip_alter argument to prevent the
  268.  268   // data from being altered again, as we already altered it above. Settings are
  269.  269   // handled separately, afterwards.
  270.  270   if (isset($items['js']['settings'])) {
  271.  271     unset($items['js']['settings']);
  272.  272   }
  273.  273   $styles = drupal_get_css($items['css'], TRUE);
  274.  274   $scripts_footer = drupal_get_js('footer', $items['js'], TRUE);
  275.  275   $scripts_header = drupal_get_js('header', $items['js'], TRUE);
  276.  276 
  277.  277   $extra_commands = array();
  278.  278   if (!empty($styles)) {
  279.  279     $extra_commands[] = ajax_command_prepend('head', $styles);
  280.  280   }
  281.  281   if (!empty($scripts_header)) {
  282.  282     $extra_commands[] = ajax_command_prepend('head', $scripts_header);
  283.  283   }
  284.  284   if (!empty($scripts_footer)) {
  285.  285     $extra_commands[] = ajax_command_append('body', $scripts_footer);
  286.  286   }
  287.  287   if (!empty($extra_commands)) {
  288.  288     $commands = array_merge($extra_commands, $commands);
  289.  289   }
  290.  290 
  291.  291   // Now add a command to merge changes and additions to Drupal.settings.
  292.  292   $scripts = drupal_add_js();
  293.  293   if (!empty($scripts['settings'])) {
  294.  294     $settings = $scripts['settings'];
  295.  295     array_unshift($commands, ajax_command_settings(call_user_func_array('array_merge_recursive', $settings['data']), TRUE));
  296.  296   }
  297.  297 
  298.  298   // Allow modules to alter any Ajax response.
  299.  299   drupal_alter('ajax_render', $commands);
  300.  300 
  301.  301   return drupal_json_encode($commands);
  302.  302 }

  304.  304 /**
  305.  305  * Gets a form submitted via #ajax during an Ajax callback.
  306.  306  *
  307.  307  * This will load a form from the form cache used during Ajax operations. It
  308.  308  * pulls the form info from $_POST.
  309.  309  *
  310.  310  * @return
  311.  311  *   An array containing the $form, $form_state, $form_id, $form_build_id and an
  312.  312  *   initial list of Ajax $commands. Use the list() function to break these
  313.  313  *   apart:
  314.  314  *   @code
  315.  315  *     list($form, $form_state, $form_id, $form_build_id, $commands) = ajax_get_form();
  316.  316  *   @endcode
  317.  317  */
  318.  318 function ajax_get_form() {
  319.  319   $form_state = form_state_defaults();
  320.  320 
  321.  321   $form_build_id = $_POST['form_build_id'];
  322.  322 
  323.  323   // Get the form from the cache.
  324.  324   $form = form_get_cache($form_build_id, $form_state);
  325.  325   if (!$form) {
  326.  326     // If $form cannot be loaded from the cache, the form_build_id in $_POST
  327.  327     // must be invalid, which means that someone performed a POST request onto
  328.  328     // system/ajax without actually viewing the concerned form in the browser.
  329.  329     // This is likely a hacking attempt as it never happens under normal
  330.  330     // circumstances, so we just do nothing.
  331.  331     watchdog('ajax', 'Invalid form POST data.', array(), WATCHDOG_WARNING);
  332.  332     drupal_exit();
  333.  333   }
  334.  334 
  335.  335   // When a page level cache is enabled, the form-build id might have been
  336.  336   // replaced from within form_get_cache. If this is the case, it is also
  337.  337   // necessary to update it in the browser by issuing an appropriate Ajax
  338.  338   // command.
  339.  339   $commands = array();
  340.  340   if (isset($form['#build_id_old']) && $form['#build_id_old'] != $form['#build_id']) {
  341.  341     // If the form build ID has changed, issue an Ajax command to update it.
  342.  342     $commands[] = ajax_command_update_build_id($form);
  343.  343     $form_build_id = $form['#build_id'];
  344.  344   }
  345.  345 
  346.  346   // Since some of the submit handlers are run, redirects need to be disabled.
  347.  347   $form_state['no_redirect'] = TRUE;
  348.  348 
  349.  349   // When a form is rebuilt after Ajax processing, its #build_id and #action
  350.  350   // should not change.
  351.  351   // @see drupal_rebuild_form()
  352.  352   $form_state['rebuild_info']['copy']['#build_id'] = TRUE;
  353.  353   $form_state['rebuild_info']['copy']['#action'] = TRUE;
  354.  354 
  355.  355   // The form needs to be processed; prepare for that by setting a few internal
  356.  356   // variables.
  357.  357   $form_state['input'] = $_POST;
  358.  358   $form_id = $form['#form_id'];
  359.  359 
  360.  360   return array($form, $form_state, $form_id, $form_build_id, $commands);
  361.  361 }

  363.  363 /**
  364.  364  * Menu callback; handles Ajax requests for the #ajax Form API property.
  365.  365  *
  366.  366  * This rebuilds the form from cache and invokes the defined #ajax['callback']
  367.  367  * to return an Ajax command structure for JavaScript. In case no 'callback' has
  368.  368  * been defined, nothing will happen.
  369.  369  *
  370.  370  * The Form API #ajax property can be set both for buttons and other input
  371.  371  * elements.
  372.  372  *
  373.  373  * This function is also the canonical example of how to implement
  374.  374  * #ajax['path']. If processing is required that cannot be accomplished with
  375.  375  * a callback, re-implement this function and set #ajax['path'] to the
  376.  376  * enhanced function.
  377.  377  *
  378.  378  * @see system_menu()
  379.  379  */
  380.  380 function ajax_form_callback() {
  381.  381   list($form, $form_state, $form_id, $form_build_id, $commands) = ajax_get_form();
  382.  382   drupal_process_form($form['#form_id'], $form, $form_state);
  383.  383 
  384.  384   // We need to return the part of the form (or some other content) that needs
  385.  385   // to be re-rendered so the browser can update the page with changed content.
  386.  386   // Since this is the generic menu callback used by many Ajax elements, it is
  387.  387   // up to the #ajax['callback'] function of the element (may or may not be a
  388.  388   // button) that triggered the Ajax request to determine what needs to be
  389.  389   // rendered.
  390.  390   if (!empty($form_state['triggering_element'])) {
  391.  391     $callback = $form_state['triggering_element']['#ajax']['callback'];
  392.  392   }
  393.  393   if (!empty($callback) && function_exists($callback)) {
  394.  394     $result = $callback($form, $form_state);
  395.  395 
  396.  396     if (!(is_array($result) && isset($result['#type']) && $result['#type'] == 'ajax')) {
  397.  397       // Turn the response into a #type=ajax array if it isn't one already.
  398.  398       $result = array(
  399.  399         '#type' => 'ajax',
  400.  400         '#commands' => ajax_prepare_response($result),
  401.  401       );
  402.  402     }
  403.  403 
  404.  404     $result['#commands'] = array_merge($commands, $result['#commands']);
  405.  405 
  406.  406     return $result;
  407.  407   }
  408.  408 }

  410.  410 /**
  411.  411  * Theme callback for Ajax requests.
  412.  412  *
  413.  413  * Many different pages can invoke an Ajax request to system/ajax or another
  414.  414  * generic Ajax path. It is almost always desired for an Ajax response to be
  415.  415  * rendered using the same theme as the base page, because most themes are built
  416.  416  * with the assumption that they control the entire page, so if the CSS for two
  417.  417  * themes are both loaded for a given page, they may conflict with each other.
  418.  418  * For example, Bartik is Drupal's default theme, and Seven is Drupal's default
  419.  419  * administration theme. Depending on whether the "Use the administration theme
  420.  420  * when editing or creating content" checkbox is checked, the node edit form may
  421.  421  * be displayed in either theme, but the Ajax response to the Field module's
  422.  422  * "Add another item" button should be rendered using the same theme as the rest
  423.  423  * of the page. Therefore, system_menu() sets the 'theme callback' for
  424.  424  * 'system/ajax' to this function, and it is recommended that modules
  425.  425  * implementing other generic Ajax paths do the same.
  426.  426  *
  427.  427  * @see system_menu()
  428.  428  * @see file_menu()
  429.  429  */
  430.  430 function ajax_base_page_theme() {
  431.  431   if (!empty($_POST['ajax_page_state']['theme']) && !empty($_POST['ajax_page_state']['theme_token'])) {
  432.  432     $theme = $_POST['ajax_page_state']['theme'];
  433.  433     $token = $_POST['ajax_page_state']['theme_token'];
  434.  434 
  435.  435     // Prevent a request forgery from giving a person access to a theme they
  436.  436     // shouldn't be otherwise allowed to see. However, since everyone is allowed
  437.  437     // to see the default theme, token validation isn't required for that, and
  438.  438     // bypassing it allows most use-cases to work even when accessed from the
  439.  439     // page cache.
  440.  440     if ($theme === variable_get('theme_default', 'bartik') || drupal_valid_token($token, $theme)) {
  441.  441       return $theme;
  442.  442     }
  443.  443   }
  444.  444 }

  446.  446 /**
  447.  447  * Packages and sends the result of a page callback as an Ajax response.
  448.  448  *
  449.  449  * This function is the equivalent of drupal_deliver_html_page(), but for Ajax
  450.  450  * requests. Like that function, it:
  451.  451  * - Adds needed HTTP headers.
  452.  452  * - Prints rendered output.
  453.  453  * - Performs end-of-request tasks.
  454.  454  *
  455.  455  * @param $page_callback_result
  456.  456  *   The result of a page callback. Can be one of:
  457.  457  *   - NULL: to indicate no content.
  458.  458  *   - An integer menu status constant: to indicate an error condition.
  459.  459  *   - A string of HTML content.
  460.  460  *   - A renderable array of content.
  461.  461  *
  462.  462  * @see drupal_deliver_html_page()
  463.  463  */
  464.  464 function ajax_deliver($page_callback_result) {
  465.  465   // Browsers do not allow JavaScript to read the contents of a user's local
  466.  466   // files. To work around that, the jQuery Form plugin submits forms containing
  467.  467   // a file input element to an IFRAME, instead of using XHR. Browsers do not
  468.  468   // normally expect JSON strings as content within an IFRAME, so the response
  469.  469   // must be customized accordingly.
  470.  470   // @see http://malsup.com/jquery/form/#file-upload
  471.  471   // @see Drupal.ajax.prototype.beforeSend()
  472.  472   $iframe_upload = !empty($_POST['ajax_iframe_upload']);
  473.  473 
  474.  474   // Emit a Content-Type HTTP header if none has been added by the page callback
  475.  475   // or by a wrapping delivery callback.
  476.  476   if (is_null(drupal_get_http_header('Content-Type'))) {
  477.  477     if (!$iframe_upload) {
  478.  478       // Standard JSON can be returned to a browser's XHR object, and to
  479.  479       // non-browser user agents.
  480.  480       // @see http://www.ietf.org/rfc/rfc4627.txt?number=4627
  481.  481       drupal_add_http_header('Content-Type', 'application/json; charset=utf-8');
  482.  482     }
  483.  483     else {
  484.  484       // Browser IFRAMEs expect HTML. With most other content types, Internet
  485.  485       // Explorer presents the user with a download prompt.
  486.  486       drupal_add_http_header('Content-Type', 'text/html; charset=utf-8');
  487.  487     }
  488.  488   }
  489.  489 
  490.  490   // Print the response.
  491.  491   $commands = ajax_prepare_response($page_callback_result);
  492.  492   $json = ajax_render($commands);
  493.  493   if (!$iframe_upload) {
  494.  494     // Standard JSON can be returned to a browser's XHR object, and to
  495.  495     // non-browser user agents.
  496.  496     print $json;
  497.  497   }
  498.  498   else {
  499.  499     // Browser IFRAMEs expect HTML. Browser extensions, such as Linkification
  500.  500     // and Skype's Browser Highlighter, convert URLs, phone numbers, etc. into
  501.  501     // links. This corrupts the JSON response. Protect the integrity of the
  502.  502     // JSON data by making it the value of a textarea.
  503.  503     // @see http://malsup.com/jquery/form/#file-upload
  504.  504     // @see http://drupal.org/node/1009382
  505.  505     print '<textarea>' . $json . '</textarea>';
  506.  506   }
  507.  507 
  508.  508   // Perform end-of-request tasks.
  509.  509   ajax_footer();
  510.  510 }

  512.  512 /**
  513.  513  * Converts the return value of a page callback into an Ajax commands array.
  514.  514  *
  515.  515  * @param $page_callback_result
  516.  516  *   The result of a page callback. Can be one of:
  517.  517  *   - NULL: to indicate no content.
  518.  518  *   - An integer menu status constant: to indicate an error condition.
  519.  519  *   - A string of HTML content.
  520.  520  *   - A renderable array of content.
  521.  521  *
  522.  522  * @return
  523.  523  *   An Ajax commands array that can be passed to ajax_render().
  524.  524  */
  525.  525 function ajax_prepare_response($page_callback_result) {
  526.  526   $commands = array();
  527.  527   if (!isset($page_callback_result)) {
  528.  528     // Simply delivering an empty commands array is sufficient. This results
  529.  529     // in the Ajax request being completed, but nothing being done to the page.
  530.  530   }
  531.  531   elseif (is_int($page_callback_result)) {
  532.  532     switch ($page_callback_result) {
  533.  533       case MENU_NOT_FOUND:
  534.  534         $commands[] = ajax_command_alert(t('The requested page could not be found.'));
  535.  535         break;
  536.  536 
  537.  537       case MENU_ACCESS_DENIED:
  538.  538         $commands[] = ajax_command_alert(t('You are not authorized to access this page.'));
  539.  539         break;
  540.  540 
  541.  541       case MENU_SITE_OFFLINE:
  542.  542         $commands[] = ajax_command_alert(filter_xss_admin(variable_get('maintenance_mode_message',
  543.  543           t('@site is currently under maintenance. We should be back shortly. Thank you for your patience.', array('@site' => variable_get('site_name', 'Drupal'))))));
  544.  544         break;
  545.  545     }
  546.  546   }
  547.  547   elseif (is_array($page_callback_result) && isset($page_callback_result['#type']) && ($page_callback_result['#type'] == 'ajax')) {
  548.  548     // Complex Ajax callbacks can return a result that contains an error message
  549.  549     // or a specific set of commands to send to the browser.
  550.  550     $page_callback_result += element_info('ajax');
  551.  551     $error = $page_callback_result['#error'];
  552.  552     if (isset($error) && $error !== FALSE) {
  553.  553       if ((empty($error) || $error === TRUE)) {
  554.  554         $error = t('An error occurred while handling the request: The server received invalid input.');
  555.  555       }
  556.  556       $commands[] = ajax_command_alert($error);
  557.  557     }
  558.  558     else {
  559.  559       $commands = $page_callback_result['#commands'];
  560.  560     }
  561.  561   }
  562.  562   else {
  563.  563     // Like normal page callbacks, simple Ajax callbacks can return HTML
  564.  564     // content, as a string or render array. This HTML is inserted in some
  565.  565     // relationship to #ajax['wrapper'], as determined by which jQuery DOM
  566.  566     // manipulation method is used. The method used is specified by
  567.  567     // #ajax['method']. The default method is 'replaceWith', which completely
  568.  568     // replaces the old wrapper element and its content with the new HTML.
  569.  569     $html = is_string($page_callback_result) ? $page_callback_result : drupal_render($page_callback_result);
  570.  570     $commands[] = ajax_command_insert(NULL, $html);
  571.  571     // Add the status messages inside the new content's wrapper element, so that
  572.  572     // on subsequent Ajax requests, it is treated as old content.
  573.  573     $commands[] = ajax_command_prepend(NULL, theme('status_messages'));
  574.  574   }
  575.  575 
  576.  576   return $commands;
  577.  577 }

  579.  579 /**
  580.  580  * Performs end-of-Ajax-request tasks.
  581.  581  *
  582.  582  * This function is the equivalent of drupal_page_footer(), but for Ajax
  583.  583  * requests.
  584.  584  *
  585.  585  * @see drupal_page_footer()
  586.  586  */
  587.  587 function ajax_footer() {
  588.  588   // Even for Ajax requests, invoke hook_exit() implementations. There may be
  589.  589   // modules that need very fast Ajax responses, and therefore, run Ajax
  590.  590   // requests with an early bootstrap.
  591.  591   if (drupal_get_bootstrap_phase() == DRUPAL_BOOTSTRAP_FULL && (!defined('MAINTENANCE_MODE') || MAINTENANCE_MODE != 'update')) {
  592.  592     module_invoke_all('exit');
  593.  593   }
  594.  594 
  595.  595   // Commit the user session. See above comment about the possibility of this
  596.  596   // function running without session.inc loaded.
  597.  597   if (function_exists('drupal_session_commit')) {
  598.  598     drupal_session_commit();

  621.  621 /**
  622.  622  * Adds Ajax information about an element to communicate with JavaScript.
  623.  623  *
  624.  624  * If #ajax['path'] is set on an element, this additional JavaScript is added
  625.  625  * to the page header to attach the Ajax behaviors. See ajax.js for more
  626.  626  * information.
  627.  627  *
  628.  628  * @param $element
  629.  629  *   An associative array containing the properties of the element.
  630.  630  *   Properties used:
  631.  631  *   - #ajax['event']
  632.  632  *   - #ajax['prevent']
  633.  633  *   - #ajax['path']
  634.  634  *   - #ajax['options']
  635.  635  *   - #ajax['wrapper']
  636.  636  *   - #ajax['parameters']
  637.  637  *   - #ajax['effect']
  638.  638  *
  639.  639  * @return
  640.  640  *   The processed element with the necessary JavaScript attached to it.
  641.  641  */
  642.  642 function ajax_pre_render_element($element) {
  643.  643   // Skip already processed elements.
  644.  644   if (isset($element['#ajax_processed'])) {
  645.  645     return $element;
  646.  646   }
  647.  647   // Initialize #ajax_processed, so we do not process this element again.
  648.  648   $element['#ajax_processed'] = FALSE;
  649.  649 
  650.  650   // Nothing to do if there is neither a callback nor a path.
  651.  651   if (!(isset($element['#ajax']['callback']) || isset($element['#ajax']['path']))) {
  652.  652     return $element;
  653.  653   }
  654.  654 
  655.  655   // Add a reasonable default event handler if none was specified.
  656.  656   if (isset($element['#ajax']) && !isset($element['#ajax']['event'])) {
  657.  657     switch ($element['#type']) {
  658.  658       case 'submit':
  659.  659       case 'button':
  660.  660       case 'image_button':
  661.  661         // Pressing the ENTER key within a textfield triggers the click event of
  662.  662         // the form's first submit button. Triggering Ajax in this situation
  663.  663         // leads to problems, like breaking autocomplete textfields, so we bind
  664.  664         // to mousedown instead of click.
  665.  665         // @see http://drupal.org/node/216059
  666.  666         $element['#ajax']['event'] = 'mousedown';
  667.  667         // Retain keyboard accessibility by setting 'keypress'. This causes
  668.  668         // ajax.js to trigger 'event' when SPACE or ENTER are pressed while the
  669.  669         // button has focus.
  670.  670         $element['#ajax']['keypress'] = TRUE;
  671.  671         // Binding to mousedown rather than click means that it is possible to
  672.  672         // trigger a click by pressing the mouse, holding the mouse button down
  673.  673         // until the Ajax request is complete and the button is re-enabled, and
  674.  674         // then releasing the mouse button. Set 'prevent' so that ajax.js binds
  675.  675         // an additional handler to prevent such a click from triggering a
  676.  676         // non-Ajax form submission. This also prevents a textfield's ENTER
  677.  677         // press triggering this button's non-Ajax form submission behavior.
  678.  678         if (!isset($element['#ajax']['prevent'])) {
  679.  679           $element['#ajax']['prevent'] = 'click';
  680.  680         }
  681.  681         break;
  682.  682 
  683.  683       case 'password':
  684.  684       case 'textfield':
  685.  685       case 'textarea':
  686.  686         $element['#ajax']['event'] = 'blur';
  687.  687         break;
  688.  688 
  689.  689       case 'radio':
  690.  690       case 'checkbox':
  691.  691       case 'select':
  692.  692         $element['#ajax']['event'] = 'change';
  693.  693         break;
  694.  694 
  695.  695       case 'link':
  696.  696         $element['#ajax']['event'] = 'click';
  697.  697         break;
  698.  698 
  699.  699       default:
  700.  700         return $element;
  701.  701     }
  702.  702   }
  703.  703 
  704.  704   // Attach JavaScript settings to the element.
  705.  705   if (isset($element['#ajax']['event'])) {
  706.  706     $element['#attached']['library'][] = array('system', 'jquery.form');
  707.  707     $element['#attached']['library'][] = array('system', 'drupal.ajax');
  708.  708 
  709.  709     $settings = $element['#ajax'];
  710.  710 
  711.  711     // Assign default settings.
  712.  712     $settings += array(
  713.  713       'path' => 'system/ajax',
  714.  714       'options' => array(),
  715.  715     );
  716.  716 
  717.  717     // @todo Legacy support. Remove in Drupal 8.
  718.  718     if (isset($settings['method']) && $settings['method'] == 'replace') {
  719.  719       $settings['method'] = 'replaceWith';
  720.  720     }
  721.  721 
  722.  722     // Change path to URL.
  723.  723     $settings['url'] = url($settings['path'], $settings['options']);
  724.  724     unset($settings['path'], $settings['options']);
  725.  725 
  726.  726     // Add special data to $settings['submit'] so that when this element
  727.  727     // triggers an Ajax submission, Drupal's form processing can determine which
  728.  728     // element triggered it.
  729.  729     // @see _form_element_triggered_scripted_submission()
  730.  730     if (isset($settings['trigger_as'])) {
  731.  731       // An element can add a 'trigger_as' key within #ajax to make the element
  732.  732       // submit as though another one (for example, a non-button can use this
  733.  733       // to submit the form as though a button were clicked). When using this,
  734.  734       // the 'name' key is always required to identify the element to trigger
  735.  735       // as. The 'value' key is optional, and only needed when multiple elements
  736.  736       // share the same name, which is commonly the case for buttons.
  737.  737       $settings['submit']['_triggering_element_name'] = $settings['trigger_as']['name'];
  738.  738       if (isset($settings['trigger_as']['value'])) {
  739.  739         $settings['submit']['_triggering_element_value'] = $settings['trigger_as']['value'];
  740.  740       }
  741.  741       unset($settings['trigger_as']);
  742.  742     }
  743.  743     elseif (isset($element['#name'])) {
  744.  744       // Most of the time, elements can submit as themselves, in which case the
  745.  745       // 'trigger_as' key isn't needed, and the element's name is used.
  746.  746       $settings['submit']['_triggering_element_name'] = $element['#name'];
  747.  747       // If the element is a (non-image) button, its name may not identify it
  748.  748       // uniquely, in which case a match on value is also needed.
  749.  749       // @see _form_button_was_clicked()
  750.  750       if (isset($element['#button_type']) && empty($element['#has_garbage_value'])) {
  751.  751         $settings['submit']['_triggering_element_value'] = $element['#value'];
  752.  752       }
  753.  753     }
  754.  754 
  755.  755     // Convert a simple #ajax['progress'] string into an array.
  756.  756     if (isset($settings['progress']) && is_string($settings['progress'])) {
  757.  757       $settings['progress'] = array('type' => $settings['progress']);
  758.  758     }
  759.  759     // Change progress path to a full URL.
  760.  760     if (isset($settings['progress']['path'])) {
  761.  761       $settings['progress']['url'] = url($settings['progress']['path']);
  762.  762       unset($settings['progress']['path']);
  763.  763     }
  764.  764 
  765.  765     $element['#attached']['js'][] = array(
  766.  766       'type' => 'setting',
  767.  767       'data' => array('ajax' => array($element['#id'] => $settings)),
  768.  768     );
  769.  769 
  770.  770     // Indicate that Ajax processing was successful.
  771.  771     $element['#ajax_processed'] = TRUE;
  772.  772   }
  773.  773   return $element;
  774.  774 }
