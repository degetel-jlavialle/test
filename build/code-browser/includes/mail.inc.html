  123. 123   $default_from = variable_get('site_mail', ini_get('sendmail_from'));
  124. 124 
  125. 125   // Bundle up the variables into a structured array for altering.
  126. 126   $message = array(
  127. 127     'id'       => $module . '_' . $key,
  128. 128     'module'   => $module,
  129. 129     'key'      => $key,
  130. 130     'to'       => $to,
  131. 131     'from'     => isset($from) ? $from : $default_from,
  132. 132     'language' => $language,
  133. 133     'params'   => $params,
  134. 134     'send'     => TRUE,
  135. 135     'subject'  => '',
  136. 136     'body'     => array()
  137. 137   );
  138. 138 
  139. 139   // Build the default headers
  140. 140   $headers = array(
  141. 141     'MIME-Version'              => '1.0',
  142. 142     'Content-Type'              => 'text/plain; charset=UTF-8; format=flowed; delsp=yes',
  143. 143     'Content-Transfer-Encoding' => '8Bit',
  144. 144     'X-Mailer'                  => 'Drupal'
  145. 145   );
  146. 146   if ($default_from) {
  147. 147     // To prevent e-mail from looking like spam, the addresses in the Sender and
  148. 148     // Return-Path headers should have a domain authorized to use the originating
  149. 149     // SMTP server.
  150. 150     $headers['From'] = $headers['Sender'] = $headers['Return-Path'] = $default_from;
  151. 151   }
  152. 152   if ($from) {
  153. 153     $headers['From'] = $from;
  154. 154   }
  155. 155   $message['headers'] = $headers;
  156. 156 
  157. 157   // Build the e-mail (get subject and body, allow additional headers) by
  158. 158   // invoking hook_mail() on this module. We cannot use module_invoke() as
  159. 159   // we need to have $message by reference in hook_mail().
  160. 160   if (function_exists($function = $module . '_mail')) {
  161. 161     $function($key, $message, $params);
  162. 162   }
  163. 163 
  164. 164   // Invoke hook_mail_alter() to allow all modules to alter the resulting e-mail.
  165. 165   drupal_alter('mail', $message);
  166. 166 
  167. 167   // Retrieve the responsible implementation for this message.
  168. 168   $system = drupal_mail_system($module, $key);
  169. 169 
  170. 170   // Format the message body.
  171. 171   $message = $system->format($message);
  172. 172 
  173. 173   // Optionally send e-mail.
  174. 174   if ($send) {
  175. 175     // The original caller requested sending. Sending was canceled by one or
  176. 176     // more hook_mail_alter() implementations. We set 'result' to NULL, because
  177. 177     // FALSE indicates an error in sending.
  178. 178     if (empty($message['send'])) {
  179. 179       $message['result'] = NULL;
  180. 180     }
  181. 181     // Sending was originally requested and was not canceled.
  182. 182     else {
  183. 183       $message['result'] = $system->mail($message);
  184. 184       // Log errors.
  185. 185       if (!$message['result']) {
  186. 186         watchdog('mail', 'Error sending e-mail (from %from to %to).', array('%from' => $message['from'], '%to' => $message['to']), WATCHDOG_ERROR);
  187. 187         drupal_set_message(t('Unable to send e-mail. Contact the site administrator if the problem persists.'), 'error');
  188. 188       }
  189. 189     }
  190. 190   }
  191. 191 
  192. 192   return $message;
  193. 193 }

  195. 195 /**
  196. 196  * Returns an object that implements the MailSystemInterface interface.
  197. 197  *
  198. 198  * Allows for one or more custom mail backends to format and send mail messages
  199. 199  * composed using drupal_mail().
  200. 200  *
  201. 201  * An implementation needs to implement the following methods:
  202. 202  * - format: Allows to preprocess, format, and postprocess a mail
  203. 203  *   message before it is passed to the sending system. By default, all messages
  204. 204  *   may contain HTML and are converted to plain-text by the DefaultMailSystem
  205. 205  *   implementation. For example, an alternative implementation could override
  206. 206  *   the default implementation and additionally sanitize the HTML for usage in
  207. 207  *   a MIME-encoded e-mail, but still invoking the DefaultMailSystem
  208. 208  *   implementation to generate an alternate plain-text version for sending.
  209. 209  * - mail: Sends a message through a custom mail sending engine.
  210. 210  *   By default, all messages are sent via PHP's mail() function by the
  211. 211  *   DefaultMailSystem implementation.
  212. 212  *
  213. 213  * The selection of a particular implementation is controlled via the variable
  214. 214  * 'mail_system', which is a keyed array.  The default implementation
  215. 215  * is the class whose name is the value of 'default-system' key. A more specific
  216. 216  * match first to key and then to module will be used in preference to the
  217. 217  * default. To specify a different class for all mail sent by one module, set
  218. 218  * the class name as the value for the key corresponding to the module name. To
  219. 219  * specify a class for a particular message sent by one module, set the class
  220. 220  * name as the value for the array key that is the message id, which is
  221. 221  * "${module}_${key}".
  222. 222  *
  223. 223  * For example to debug all mail sent by the user module by logging it to a
  224. 224  * file, you might set the variable as something like:
  225. 225  *
  226. 226  * @code
  227. 227  * array(
  228. 228  *   'default-system' => 'DefaultMailSystem',
  229. 229  *   'user' => 'DevelMailLog',
  230. 230  * );
  231. 231  * @endcode
  232. 232  *
  233. 233  * Finally, a different system can be specified for a specific e-mail ID (see
  234. 234  * the $key param), such as one of the keys used by the contact module:
  235. 235  *
  236. 236  * @code
  237. 237  * array(
  238. 238  *   'default-system' => 'DefaultMailSystem',
  239. 239  *   'user' => 'DevelMailLog',
  240. 240  *   'contact_page_autoreply' => 'DrupalDevNullMailSend',
  241. 241  * );
  242. 242  * @endcode
  243. 243  *
  244. 244  * Other possible uses for system include a mail-sending class that actually
  245. 245  * sends (or duplicates) each message to SMS, Twitter, instant message, etc, or
  246. 246  * a class that queues up a large number of messages for more efficient bulk
  247. 247  * sending or for sending via a remote gateway so as to reduce the load
  248. 248  * on the local server.
  249. 249  *
  250. 250  * @param $module
  251. 251  *   The module name which was used by drupal_mail() to invoke hook_mail().
  252. 252  * @param $key
  253. 253  *   A key to identify the e-mail sent. The final e-mail ID for the e-mail
  254. 254  *   alter hook in drupal_mail() would have been {$module}_{$key}.
  255. 255  *
  256. 256  * @return MailSystemInterface
  257. 257  */
  258. 258 function drupal_mail_system($module, $key) {
  259. 259   $instances = &drupal_static(__FUNCTION__, array());
  260. 260 
  261. 261   $id = $module . '_' . $key;
  262. 262 
  263. 263   $configuration = variable_get('mail_system', array('default-system' => 'DefaultMailSystem'));
  264. 264 
  265. 265   // Look for overrides for the default class, starting from the most specific
  266. 266   // id, and falling back to the module name.
  267. 267   if (isset($configuration[$id])) {
  268. 268     $class = $configuration[$id];
  269. 269   }
  270. 270   elseif (isset($configuration[$module])) {
  271. 271     $class = $configuration[$module];
  272. 272   }
  273. 273   else {
  274. 274     $class = $configuration['default-system'];
  275. 275   }
  276. 276 
  277. 277   if (empty($instances[$class])) {
  278. 278     $interfaces = class_implements($class);
  279. 279     if (isset($interfaces['MailSystemInterface'])) {
  280. 280       $instances[$class] = new $class();
  281. 281     }
  282. 282     else {
  283. 283       throw new Exception(t('Class %class does not implement interface %interface', array('%class' => $class, '%interface' => 'MailSystemInterface')));
  284. 284     }
  285. 285   }
  286. 286   return $instances[$class];
  287. 287 }

  346. 346  *   subsequent wrapped lines. Others are replaced by spaces.
