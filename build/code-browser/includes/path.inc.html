  24.  24 /**
  25.  25  * Given an alias, return its Drupal system URL if one exists. Given a Drupal
  26.  26  * system URL return one of its aliases if such a one exists. Otherwise,
  27.  27  * return FALSE.
  28.  28  *
  29.  29  * @param $action
  30.  30  *   One of the following values:
  31.  31  *   - wipe: delete the alias cache.
  32.  32  *   - alias: return an alias for a given Drupal system path (if one exists).
  33.  33  *   - source: return the Drupal system URL for a path alias (if one exists).
  34.  34  * @param $path
  35.  35  *   The path to investigate for corresponding aliases or system URLs.
  36.  36  * @param $path_language
  37.  37  *   Optional language code to search the path with. Defaults to the page language.
  38.  38  *   If there's no path defined for that language it will search paths without
  39.  39  *   language.
  40.  40  *
  41.  41  * @return
  42.  42  *   Either a Drupal system path, an aliased path, or FALSE if no path was
  43.  43  *   found.
  44.  44  */
  45.  45 function drupal_lookup_path($action, $path = '', $path_language = NULL) {
  46.  46   global $language_url;
  47.  47   // Use the advanced drupal_static() pattern, since this is called very often.
  48.  48   static $drupal_static_fast;
  49.  49   if (!isset($drupal_static_fast)) {
  50.  50     $drupal_static_fast['cache'] = &drupal_static(__FUNCTION__);
  51.  51   }
  52.  52   $cache = &$drupal_static_fast['cache'];
  53.  53 
  54.  54   if (!isset($cache)) {
  55.  55     $cache = array(
  56.  56       'map' => array(),
  57.  57       'no_source' => array(),
  58.  58       'whitelist' => NULL,
  59.  59       'system_paths' => array(),
  60.  60       'no_aliases' => array(),
  61.  61       'first_call' => TRUE,
  62.  62     );
  63.  63   }
  64.  64 
  65.  65   // Retrieve the path alias whitelist.
  66.  66   if (!isset($cache['whitelist'])) {
  67.  67     $cache['whitelist'] = variable_get('path_alias_whitelist', NULL);
  68.  68     if (!isset($cache['whitelist'])) {
  69.  69       $cache['whitelist'] = drupal_path_alias_whitelist_rebuild();
  70.  70     }
  71.  71   }
  72.  72 
  73.  73   // If no language is explicitly specified we default to the current URL
  74.  74   // language. If we used a language different from the one conveyed by the
  75.  75   // requested URL, we might end up being unable to check if there is a path
  76.  76   // alias matching the URL path.
  77.  77   $path_language = $path_language ? $path_language : $language_url->language;
  78.  78 
  79.  79   if ($action == 'wipe') {
  80.  80     $cache = array();
  81.  81     $cache['whitelist'] = drupal_path_alias_whitelist_rebuild();
  82.  82   }
  83.  83   elseif ($cache['whitelist'] && $path != '') {
  84.  84     if ($action == 'alias') {
  85.  85       // During the first call to drupal_lookup_path() per language, load the
  86.  86       // expected system paths for the page from cache.
  87.  87       if (!empty($cache['first_call'])) {
  88.  88         $cache['first_call'] = FALSE;
  89.  89 
  90.  90         $cache['map'][$path_language] = array();
  91.  91         // Load system paths from cache.
  92.  92         $cid = current_path();
  93.  93         if ($cached = cache_get($cid, 'cache_path')) {
  94.  94           $cache['system_paths'] = $cached->data;
  95.  95           // Now fetch the aliases corresponding to these system paths.
  96.  96           $args = array(
  97.  97             ':system' => $cache['system_paths'],
  98.  98             ':language' => $path_language,
  99.  99             ':language_none' => LANGUAGE_NONE,
  100. 100           );
  101. 101           // Always get the language-specific alias before the language-neutral
  102. 102           // one. For example 'de' is less than 'und' so the order needs to be
  103. 103           // ASC, while 'xx-lolspeak' is more than 'und' so the order needs to
  104. 104           // be DESC. We also order by pid ASC so that fetchAllKeyed() returns
  105. 105           // the most recently created alias for each source. Subsequent queries
  106. 106           // using fetchField() must use pid DESC to have the same effect.
  107. 107           // For performance reasons, the query builder is not used here.
  108. 108           if ($path_language == LANGUAGE_NONE) {
  109. 109             // Prevent PDO from complaining about a token the query doesn't use.
  110. 110             unset($args[':language']);
  111. 111             $result = db_query('SELECT source, alias FROM {url_alias} WHERE source IN (:system) AND language = :language_none ORDER BY pid ASC', $args);
  112. 112           }
  113. 113           elseif ($path_language < LANGUAGE_NONE) {
  114. 114             $result = db_query('SELECT source, alias FROM {url_alias} WHERE source IN (:system) AND language IN (:language, :language_none) ORDER BY language ASC, pid ASC', $args);
  115. 115           }
  116. 116           else {
  117. 117             $result = db_query('SELECT source, alias FROM {url_alias} WHERE source IN (:system) AND language IN (:language, :language_none) ORDER BY language DESC, pid ASC', $args);
  118. 118           }
  119. 119           $cache['map'][$path_language] = $result->fetchAllKeyed();
  120. 120           // Keep a record of paths with no alias to avoid querying twice.
  121. 121           $cache['no_aliases'][$path_language] = array_flip(array_diff_key($cache['system_paths'], array_keys($cache['map'][$path_language])));
  122. 122         }
  123. 123       }
  124. 124       // If the alias has already been loaded, return it.
  125. 125       if (isset($cache['map'][$path_language][$path])) {
  126. 126         return $cache['map'][$path_language][$path];
  127. 127       }
  128. 128       // Check the path whitelist, if the top_level part before the first /
  129. 129       // is not in the list, then there is no need to do anything further,
  130. 130       // it is not in the database.
  131. 131       elseif (!isset($cache['whitelist'][strtok($path, '/')])) {
  132. 132         return FALSE;
  133. 133       }
  134. 134       // For system paths which were not cached, query aliases individually.
  135. 135       elseif (!isset($cache['no_aliases'][$path_language][$path])) {
  136. 136         $args = array(
  137. 137           ':source' => $path,
  138. 138           ':language' => $path_language,
  139. 139           ':language_none' => LANGUAGE_NONE,
  140. 140         );
  141. 141         // See the queries above.
  142. 142         if ($path_language == LANGUAGE_NONE) {
  143. 143           unset($args[':language']);
  144. 144           $alias = db_query("SELECT alias FROM {url_alias} WHERE source = :source AND language = :language_none ORDER BY pid DESC", $args)->fetchField();
  145. 145         }
  146. 146         elseif ($path_language > LANGUAGE_NONE) {
  147. 147           $alias = db_query("SELECT alias FROM {url_alias} WHERE source = :source AND language IN (:language, :language_none) ORDER BY language DESC, pid DESC", $args)->fetchField();
  148. 148         }
  149. 149         else {
  150. 150           $alias = db_query("SELECT alias FROM {url_alias} WHERE source = :source AND language IN (:language, :language_none) ORDER BY language ASC, pid DESC", $args)->fetchField();
  151. 151         }
  152. 152         $cache['map'][$path_language][$path] = $alias;
  153. 153         return $alias;
  154. 154       }
  155. 155     }
  156. 156     // Check $no_source for this $path in case we've already determined that there
  157. 157     // isn't a path that has this alias
  158. 158     elseif ($action == 'source' && !isset($cache['no_source'][$path_language][$path])) {
  159. 159       // Look for the value $path within the cached $map
  160. 160       $source = FALSE;
  161. 161       if (!isset($cache['map'][$path_language]) || !($source = array_search($path, $cache['map'][$path_language]))) {
  162. 162         $args = array(
  163. 163           ':alias' => $path,
  164. 164           ':language' => $path_language,
  165. 165           ':language_none' => LANGUAGE_NONE,
  166. 166         );
  167. 167         // See the queries above.
  168. 168         if ($path_language == LANGUAGE_NONE) {
  169. 169           unset($args[':language']);
  170. 170           $result = db_query("SELECT source FROM {url_alias} WHERE alias = :alias AND language = :language_none ORDER BY pid DESC", $args);
  171. 171         }
  172. 172         elseif ($path_language > LANGUAGE_NONE) {
  173. 173           $result = db_query("SELECT source FROM {url_alias} WHERE alias = :alias AND language IN (:language, :language_none) ORDER BY language DESC, pid DESC", $args);
  174. 174         }
  175. 175         else {
  176. 176           $result = db_query("SELECT source FROM {url_alias} WHERE alias = :alias AND language IN (:language, :language_none) ORDER BY language ASC, pid DESC", $args);
  177. 177         }
  178. 178         if ($source = $result->fetchField()) {
  179. 179           $cache['map'][$path_language][$source] = $path;
  180. 180         }
  181. 181         else {
  182. 182           // We can't record anything into $map because we do not have a valid
  183. 183           // index and there is no need because we have not learned anything
  184. 184           // about any Drupal path. Thus cache to $no_source.
  185. 185           $cache['no_source'][$path_language][$path] = TRUE;
  186. 186         }
  187. 187       }
  188. 188       return $source;
  189. 189     }
  190. 190   }
  191. 191 
  192. 192   return FALSE;
  193. 193 }

  195. 195 /**
  196. 196  * Cache system paths for a page.
  197. 197  *
  198. 198  * Cache an array of the system paths available on each page. We assume
  199. 199  * that aliases will be needed for the majority of these paths during
  200. 200  * subsequent requests, and load them in a single query during
  201. 201  * drupal_lookup_path().
  202. 202  */
  203. 203 function drupal_cache_system_paths() {
  204. 204   // Check if the system paths for this page were loaded from cache in this
  205. 205   // request to avoid writing to cache on every request.
  206. 206   $cache = &drupal_static('drupal_lookup_path', array());
  207. 207   if (empty($cache['system_paths']) && !empty($cache['map'])) {
  208. 208     // Generate a cache ID (cid) specifically for this page.
  209. 209     $cid = current_path();
  210. 210     // The static $map array used by drupal_lookup_path() includes all
  211. 211     // system paths for the page request.
  212. 212     if ($paths = current($cache['map'])) {
  213. 213       $data = array_keys($paths);
  214. 214       $expire = REQUEST_TIME + (60 * 60 * 24);
  215. 215       cache_set($cid, $data, 'cache_path', $expire);
  216. 216     }
  217. 217   }

  259. 259 function drupal_get_normal_path($path, $path_language = NULL) {

  261. 261 
  262. 262   // Lookup the path alias first.
  263. 263   if ($source = drupal_lookup_path('source', $path, $path_language)) {
  264. 264     $path = $source;

  272. 272     $function($path, $original_path, $path_language);
  273. 273   }
  274. 274

  301. 301 /**
  302. 302  * Check if a path matches any pattern in a set of patterns.
  303. 303  *
  304. 304  * @param $path
  305. 305  *   The path to match.
  306. 306  * @param $patterns
  307. 307  *   String containing a set of patterns separated by \n, \r or \r\n.
  308. 308  *
  309. 309  * @return
  310. 310  *   Boolean value: TRUE if the path matches a pattern, FALSE otherwise.
  311. 311  */
  312. 312 function drupal_match_path($path, $patterns) {
  313. 313   $regexps = &drupal_static(__FUNCTION__);
  314. 314 
  315. 315   if (!isset($regexps[$patterns])) {
  316. 316     // Convert path settings to a regular expression.
  317. 317     // Therefore replace newlines with a logical or, /* with asterisks and the <front> with the frontpage.
  318. 318     $to_replace = array(
  319. 319       '/(\r\n?|\n)/', // newlines
  320. 320       '/\\\\\*/',     // asterisks
  321. 321       '/(^|\|)\\\\<front\\\\>($|\|)/' // <front>
  322. 322     );
  323. 323     $replacements = array(
  324. 324       '|',
  325. 325       '.*',
  326. 326       '\1' . preg_quote(variable_get('site_frontpage', 'node'), '/') . '\2'
  327. 327     );
  328. 328     $patterns_quoted = preg_quote($patterns, '/');
  329. 329     $regexps[$patterns] = '/^(' . preg_replace($to_replace, $replacements, $patterns_quoted) . ')$/';
  330. 330   }
  331. 331   return (bool)preg_match($regexps[$patterns], $path);
  332. 332 }
