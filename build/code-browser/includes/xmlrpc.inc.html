  14.  14 /**
  15.  15  * Turns a data structure into objects with 'data' and 'type' attributes.
  16.  16  *
  17.  17  * @param $data
  18.  18  *   The data structure.
  19.  19  * @param $type
  20.  20  *   Optional type to assign to $data.
  21.  21  *
  22.  22  * @return object
  23.  23  *   An XML-RPC data object containing the input $data.
  24.  24  */
  25.  25 function xmlrpc_value($data, $type = FALSE) {
  26.  26   $xmlrpc_value = new stdClass();
  27.  27   $xmlrpc_value->data = $data;
  28.  28   if (!$type) {
  29.  29     $type = xmlrpc_value_calculate_type($xmlrpc_value);
  30.  30   }
  31.  31   $xmlrpc_value->type = $type;
  32.  32   if ($type == 'struct') {
  33.  33     // Turn all the values in the array into new xmlrpc_values
  34.  34     foreach ($xmlrpc_value->data as $key => $value) {
  35.  35       $xmlrpc_value->data[$key] = xmlrpc_value($value);
  36.  36     }
  37.  37   }
  38.  38   if ($type == 'array') {
  39.  39     for ($i = 0, $j = count($xmlrpc_value->data); $i < $j; $i++) {
  40.  40       $xmlrpc_value->data[$i] = xmlrpc_value($xmlrpc_value->data[$i]);
  41.  41     }
  42.  42   }
  43.  43   return $xmlrpc_value;
  44.  44 }

  46.  46 /**
  47.  47  * Maps a PHP type to an XML-RPC type.
  48.  48  *
  49.  49  * @param $xmlrpc_value
  50.  50  *   Variable whose type should be mapped.
  51.  51  *
  52.  52  * @return string
  53.  53  *   The corresponding XML-RPC type.
  54.  54  *
  55.  55  * @see http://www.xmlrpc.com/spec#scalars
  56.  56  */
  57.  57 function xmlrpc_value_calculate_type($xmlrpc_value) {
  58.  58   // http://www.php.net/gettype: Never use gettype() to test for a certain type
  59.  59   // [...] Instead, use the is_* functions.
  60.  60   if (is_bool($xmlrpc_value->data)) {
  61.  61     return 'boolean';
  62.  62   }
  63.  63   if (is_double($xmlrpc_value->data)) {
  64.  64     return 'double';
  65.  65   }
  66.  66   if (is_int($xmlrpc_value->data)) {
  67.  67     return 'int';
  68.  68   }
  69.  69   if (is_array($xmlrpc_value->data)) {
  70.  70     // empty or integer-indexed arrays are 'array', string-indexed arrays 'struct'
  71.  71     return empty($xmlrpc_value->data) || range(0, count($xmlrpc_value->data) - 1) === array_keys($xmlrpc_value->data) ? 'array' : 'struct';
  72.  72   }
  73.  73   if (is_object($xmlrpc_value->data)) {
  74.  74     if (isset($xmlrpc_value->data->is_date)) {
  75.  75       return 'date';
  76.  76     }
  77.  77     if (isset($xmlrpc_value->data->is_base64)) {
  78.  78       return 'base64';
  79.  79     }
  80.  80     $xmlrpc_value->data = get_object_vars($xmlrpc_value->data);
  81.  81     return 'struct';
  82.  82   }
  83.  83   // default
  84.  84   return 'string';
  85.  85 }

  87.  87 /**
  88.  88  * Generates XML representing the given value.
  89.  89  *
  90.  90  * @param $xmlrpc_value
  91.  91  *   A value to be represented in XML.
  92.  92  *
  93.  93  * @return
  94.  94  *   XML representation of $xmlrpc_value.
  95.  95  */
  96.  96 function xmlrpc_value_get_xml($xmlrpc_value) {
  97.  97   switch ($xmlrpc_value->type) {
  98.  98     case 'boolean':
  99.  99       return '<boolean>' . (($xmlrpc_value->data) ? '1' : '0') . '</boolean>';
  100. 100 
  101. 101     case 'int':
  102. 102       return '<int>' . $xmlrpc_value->data . '</int>';
  103. 103 
  104. 104     case 'double':
  105. 105       return '<double>' . $xmlrpc_value->data . '</double>';
  106. 106 
  107. 107     case 'string':
  108. 108       // Note: we don't escape apostrophes because of the many blogging clients
  109. 109       // that don't support numerical entities (and XML in general) properly.
  110. 110       return '<string>' . htmlspecialchars($xmlrpc_value->data) . '</string>';
  111. 111 
  112. 112     case 'array':
  113. 113       $return = '<array><data>' . "\n";
  114. 114       foreach ($xmlrpc_value->data as $item) {
  115. 115         $return .= '  <value>' . xmlrpc_value_get_xml($item) . "</value>\n";
  116. 116       }
  117. 117       $return .= '</data></array>';
  118. 118       return $return;
  119. 119 
  120. 120     case 'struct':
  121. 121       $return = '<struct>' . "\n";
  122. 122       foreach ($xmlrpc_value->data as $name => $value) {
  123. 123         $return .= "  <member><name>" . check_plain($name) . "</name><value>";
  124. 124         $return .= xmlrpc_value_get_xml($value) . "</value></member>\n";
  125. 125       }
  126. 126       $return .= '</struct>';
  127. 127       return $return;
  128. 128 
  129. 129     case 'date':
  130. 130       return xmlrpc_date_get_xml($xmlrpc_value->data);
  131. 131 
  132. 132     case 'base64':
  133. 133       return xmlrpc_base64_get_xml($xmlrpc_value->data);
  134. 134   }
  135. 135   return FALSE;
  136. 136 }

  161. 161 /**
  162. 162  * Parses an XML-RPC message.
  163. 163  *
  164. 164  * If parsing fails, the faultCode and faultString will be added to the message
  165. 165  * object.
  166. 166  *
  167. 167  * @param $xmlrpc_message
  168. 168  *   An object generated by xmlrpc_message().
  169. 169  *
  170. 170  * @return
  171. 171  *   TRUE if parsing succeeded; FALSE otherwise.
  172. 172  */
  173. 173 function xmlrpc_message_parse($xmlrpc_message) {
  174. 174   $xmlrpc_message->_parser = xml_parser_create();
  175. 175   // Set XML parser to take the case of tags into account.
  176. 176   xml_parser_set_option($xmlrpc_message->_parser, XML_OPTION_CASE_FOLDING, FALSE);
  177. 177   // Set XML parser callback functions
  178. 178   xml_set_element_handler($xmlrpc_message->_parser, 'xmlrpc_message_tag_open', 'xmlrpc_message_tag_close');
  179. 179   xml_set_character_data_handler($xmlrpc_message->_parser, 'xmlrpc_message_cdata');
  180. 180   xmlrpc_message_set($xmlrpc_message);
  181. 181 
  182. 182   // Strip XML declaration.
  183. 183   $header = preg_replace('/<\?xml.*?\?'.'>/s', '', substr($xmlrpc_message->message, 0, 100), 1);
  184. 184   $xml = trim(substr_replace($xmlrpc_message->message, $header, 0, 100));
  185. 185   if ($xml == '') {
  186. 186     return FALSE;
  187. 187   }
  188. 188   // Strip DTD.
  189. 189   $header = preg_replace('/^<!DOCTYPE[^>]*+>/i', '', substr($xml, 0, 200), 1);
  190. 190   $xml = trim(substr_replace($xml, $header, 0, 200));
  191. 191   if ($xml == '') {
  192. 192     return FALSE;
  193. 193   }
  194. 194   // Confirm the XML now starts with a valid root tag. A root tag can end in [> \t\r\n]
  195. 195   $root_tag = substr($xml, 0, strcspn(substr($xml, 0, 20), "> \t\r\n"));
  196. 196   // Reject a second DTD.
  197. 197   if (strtoupper($root_tag) == '<!DOCTYPE') {
  198. 198     return FALSE;
  199. 199   }
  200. 200   if (!in_array($root_tag, array('<methodCall', '<methodResponse', '<fault'))) {
  201. 201     return FALSE;
  202. 202   }
  203. 203   // Skip parsing if there is an unreasonably large number of tags.
  204. 204   try {
  205. 205     $dom = new DOMDocument();
  206. 206     @$dom->loadXML($xml);
  207. 207     if ($dom->getElementsByTagName('*')->length > variable_get('xmlrpc_message_maximum_tag_count', 30000)) {
  208. 208       return FALSE;
  209. 209     }
  210. 210   }
  211. 211   catch (Exception $e) {
  212. 212     return FALSE;
  213. 213   }
  214. 214 
  215. 215   if (!xml_parse($xmlrpc_message->_parser, $xml)) {
  216. 216     return FALSE;
  217. 217   }
  218. 218   xml_parser_free($xmlrpc_message->_parser);
  219. 219 
  220. 220   // Grab the error messages, if any.
  221. 221   $xmlrpc_message = xmlrpc_message_get();
  222. 222   if (!isset($xmlrpc_message->messagetype)) {
  223. 223     return FALSE;
  224. 224   }
  225. 225   elseif ($xmlrpc_message->messagetype == 'fault') {
  226. 226     $xmlrpc_message->fault_code = $xmlrpc_message->params[0]['faultCode'];
  227. 227     $xmlrpc_message->fault_string = $xmlrpc_message->params[0]['faultString'];
  228. 228   }
  229. 229   return TRUE;
  230. 230 }

  269. 269   $xmlrpc_message->last_open = $tag;

  272. 272     case 'methodResponse':
  273. 273     case 'fault':

  275. 275       break;
  276. 276

  285. 285       $xmlrpc_message->array_structs[] = array();

  288. 288   xmlrpc_message_set($xmlrpc_message);
  289. 289 }

  300. 300 /**
  301. 301  * Handles closing tags for XML parsing in xmlrpc_message_parse().
  302. 302  */
  303. 303 function xmlrpc_message_tag_close($parser, $tag) {
  304. 304   $xmlrpc_message = xmlrpc_message_get();
  305. 305   $value_flag = FALSE;
  306. 306   switch ($tag) {
  307. 307     case 'int':
  308. 308     case 'i4':
  309. 309       $value = (int)trim($xmlrpc_message->current_tag_contents);
  310. 310       $value_flag = TRUE;
  311. 311       break;
  312. 312 
  313. 313     case 'double':
  314. 314       $value = (double)trim($xmlrpc_message->current_tag_contents);
  315. 315       $value_flag = TRUE;
  316. 316       break;
  317. 317 
  318. 318     case 'string':
  319. 319       $value = $xmlrpc_message->current_tag_contents;
  320. 320       $value_flag = TRUE;
  321. 321       break;
  322. 322 
  323. 323     case 'dateTime.iso8601':
  324. 324       $value = xmlrpc_date(trim($xmlrpc_message->current_tag_contents));
  325. 325       // $value = $iso->getTimestamp();
  326. 326       $value_flag = TRUE;
  327. 327       break;
  328. 328 
  329. 329     case 'value':
  330. 330       // If no type is indicated, the type is string
  331. 331       // We take special care for empty values
  332. 332       if (trim($xmlrpc_message->current_tag_contents) != '' || (isset($xmlrpc_message->last_open) && ($xmlrpc_message->last_open == 'value'))) {
  333. 333         $value = (string) $xmlrpc_message->current_tag_contents;
  334. 334         $value_flag = TRUE;
  335. 335       }
  336. 336       unset($xmlrpc_message->last_open);
  337. 337       break;
  338. 338 
  339. 339     case 'boolean':
  340. 340       $value = (boolean)trim($xmlrpc_message->current_tag_contents);
  341. 341       $value_flag = TRUE;
  342. 342       break;
  343. 343 
  344. 344     case 'base64':
  345. 345       $value = base64_decode(trim($xmlrpc_message->current_tag_contents));
  346. 346       $value_flag = TRUE;
  347. 347       break;
  348. 348 
  349. 349     // Deal with stacks of arrays and structs
  350. 350     case 'data':
  351. 351     case 'struct':
  352. 352       $value = array_pop($xmlrpc_message->array_structs);
  353. 353       array_pop($xmlrpc_message->array_structs_types);
  354. 354       $value_flag = TRUE;
  355. 355       break;
  356. 356 
  357. 357     case 'member':
  358. 358       array_pop($xmlrpc_message->current_struct_name);
  359. 359       break;
  360. 360 
  361. 361     case 'name':
  362. 362       $xmlrpc_message->current_struct_name[] = trim($xmlrpc_message->current_tag_contents);
  363. 363       break;
  364. 364 
  365. 365     case 'methodName':
  366. 366       $xmlrpc_message->methodname = trim($xmlrpc_message->current_tag_contents);
  367. 367       break;
  368. 368   }
  369. 369   if ($value_flag) {
  370. 370     if (count($xmlrpc_message->array_structs) > 0) {
  371. 371       // Add value to struct or array
  372. 372       if ($xmlrpc_message->array_structs_types[count($xmlrpc_message->array_structs_types) - 1] == 'struct') {
  373. 373         // Add to struct
  374. 374         $xmlrpc_message->array_structs[count($xmlrpc_message->array_structs) - 1][$xmlrpc_message->current_struct_name[count($xmlrpc_message->current_struct_name) - 1]] = $value;
  375. 375       }
  376. 376       else {
  377. 377         // Add to array
  378. 378         $xmlrpc_message->array_structs[count($xmlrpc_message->array_structs) - 1][] = $value;
  379. 379       }
  380. 380     }
  381. 381     else {
  382. 382       // Just add as a parameter
  383. 383       $xmlrpc_message->params[] = $value;
  384. 384     }
  385. 385   }
  386. 386   if (!in_array($tag, array("data", "struct", "member"))) {
  387. 387     $xmlrpc_message->current_tag_contents = '';
  388. 388   }
  389. 389   xmlrpc_message_set($xmlrpc_message);
  390. 390 }

  392. 392 /**
  393. 393  * Constructs an object representing an XML-RPC request.
  394. 394  *
  395. 395  * @param $method
  396. 396  *   The name of the method to be called.
  397. 397  * @param $args
  398. 398  *   An array of parameters to send with the method.
  399. 399  *

  403. 403 function xmlrpc_request($method, $args) {

  407. 407   $xmlrpc_request->xml = <<<EOD
  408. 408 <?xml version="1.0"?>

  410. 410 <methodName>{$xmlrpc_request->method}</methodName>
  411. 411 <params>
  412. 412

  414. 414   foreach ($xmlrpc_request->args as $arg) {

  418. 418     $xmlrpc_request->xml .= "</value></param>\n";
  419. 419   }
  420. 420   $xmlrpc_request->xml .= '</params></methodCall>';
  421. 421   return $xmlrpc_request;
  422. 422 }

  424. 424 /**
  425. 425  * Generates, temporarily saves, and returns an XML-RPC error object.
  426. 426  *
  427. 427  * @param $code
