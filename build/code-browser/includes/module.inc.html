  33.   33 /**
  34.   34  * Returns a list of currently active modules.
  35.   35  *
  36.   36  * Usually, this returns a list of all enabled modules. When called early on in
  37.   37  * the bootstrap, it will return a list of vital modules only (those needed to
  38.   38  * generate cached pages).
  39.   39  *
  40.   40  * All parameters to this function are optional and should generally not be
  41.   41  * changed from their defaults.
  42.   42  *
  43.   43  * @param $refresh
  44.   44  *   (optional) Whether to force the module list to be regenerated (such as
  45.   45  *   after the administrator has changed the system settings). Defaults to
  46.   46  *   FALSE.
  47.   47  * @param $bootstrap_refresh
  48.   48  *   (optional) When $refresh is TRUE, setting $bootstrap_refresh to TRUE forces
  49.   49  *   the module list to be regenerated using the reduced set of modules loaded
  50.   50  *   in "bootstrap mode" for cached pages. Otherwise, setting $refresh to TRUE
  51.   51  *   generates the complete list of enabled modules.
  52.   52  * @param $sort
  53.   53  *   (optional) By default, modules are ordered by weight and module name. Set
  54.   54  *   this option to TRUE to return a module list ordered only by module name.
  55.   55  * @param $fixed_list
  56.   56  *   (optional) If an array of module names is provided, this will override the
  57.   57  *   module list with the given set of modules. This will persist until the next
  58.   58  *   call with $refresh set to TRUE or with a new $fixed_list passed in. This
  59.   59  *   parameter is primarily intended for internal use (e.g., in install.php and
  60.   60  *   update.php).
  61.   61  *
  62.   62  * @return
  63.   63  *   An associative array whose keys and values are the names of the modules in
  64.   64  *   the list.
  65.   65  */
  66.   66 function module_list($refresh = FALSE, $bootstrap_refresh = FALSE, $sort = FALSE, $fixed_list = NULL) {
  67.   67   static $list = array(), $sorted_list;
  68.   68 
  69.   69   if (empty($list) || $refresh || $fixed_list) {
  70.   70     $list = array();
  71.   71     $sorted_list = NULL;
  72.   72     if ($fixed_list) {
  73.   73       foreach ($fixed_list as $name => $module) {
  74.   74         drupal_get_filename('module', $name, $module['filename']);
  75.   75         $list[$name] = $name;
  76.   76       }
  77.   77     }
  78.   78     else {
  79.   79       if ($refresh) {
  80.   80         // For the $refresh case, make sure that system_list() returns fresh
  81.   81         // data.
  82.   82         drupal_static_reset('system_list');
  83.   83       }
  84.   84       if ($bootstrap_refresh) {
  85.   85         $list = system_list('bootstrap');
  86.   86       }
  87.   87       else {
  88.   88         // Not using drupal_map_assoc() here as that requires common.inc.
  89.   89         $list = array_keys(system_list('module_enabled'));
  90.   90         $list = (!empty($list) ? array_combine($list, $list) : array());
  91.   91       }
  92.   92     }
  93.   93   }
  94.   94   if ($sort) {
  95.   95     if (!isset($sorted_list)) {
  96.   96       $sorted_list = $list;
  97.   97       ksort($sorted_list);
  98.   98     }
  99.   99     return $sorted_list;
  100.  100   }
  101.  101   return $list;
  102.  102 }

  104.  104 /**
  105.  105  * Builds a list of bootstrap modules and enabled modules and themes.
  106.  106  *
  107.  107  * @param $type
  108.  108  *   The type of list to return:
  109.  109  *   - module_enabled: All enabled modules.
  110.  110  *   - bootstrap: All enabled modules required for bootstrap.
  111.  111  *   - theme: All themes.
  112.  112  *
  113.  113  * @return
  114.  114  *   An associative array of modules or themes, keyed by name. For $type
  115.  115  *   'bootstrap', the array values equal the keys. For $type 'module_enabled'
  116.  116  *   or 'theme', the array values are objects representing the respective
  117.  117  *   database row, with the 'info' property already unserialized.
  118.  118  *
  119.  119  * @see module_list()
  120.  120  * @see list_themes()
  121.  121  */
  122.  122 function system_list($type) {
  123.  123   $lists = &drupal_static(__FUNCTION__);
  124.  124 
  125.  125   // For bootstrap modules, attempt to fetch the list from cache if possible.
  126.  126   // if not fetch only the required information to fire bootstrap hooks
  127.  127   // in case we are going to serve the page from cache.
  128.  128   if ($type == 'bootstrap') {
  129.  129     if (isset($lists['bootstrap'])) {
  130.  130       return $lists['bootstrap'];
  131.  131     }
  132.  132     if ($cached = cache_get('bootstrap_modules', 'cache_bootstrap')) {
  133.  133       $bootstrap_list = $cached->data;
  134.  134     }
  135.  135     else {
  136.  136       $bootstrap_list = db_query("SELECT name, filename FROM {system} WHERE status = 1 AND bootstrap = 1 AND type = 'module' ORDER BY weight ASC, name ASC")->fetchAllAssoc('name');
  137.  137       cache_set('bootstrap_modules', $bootstrap_list, 'cache_bootstrap');
  138.  138     }
  139.  139     // To avoid a separate database lookup for the filepath, prime the
  140.  140     // drupal_get_filename() static cache for bootstrap modules only.
  141.  141     // The rest is stored separately to keep the bootstrap module cache small.
  142.  142     foreach ($bootstrap_list as $module) {
  143.  143       drupal_get_filename('module', $module->name, $module->filename);
  144.  144     }
  145.  145     // We only return the module names here since module_list() doesn't need
  146.  146     // the filename itself.
  147.  147     $lists['bootstrap'] = array_keys($bootstrap_list);
  148.  148   }
  149.  149   // Otherwise build the list for enabled modules and themes.
  150.  150   elseif (!isset($lists['module_enabled'])) {
  151.  151     if ($cached = cache_get('system_list', 'cache_bootstrap')) {
  152.  152       $lists = $cached->data;
  153.  153     }
  154.  154     else {
  155.  155       $lists = array(
  156.  156         'module_enabled' => array(),
  157.  157         'theme' => array(),
  158.  158         'filepaths' => array(),
  159.  159       );
  160.  160       // The module name (rather than the filename) is used as the fallback
  161.  161       // weighting in order to guarantee consistent behavior across different
  162.  162       // Drupal installations, which might have modules installed in different
  163.  163       // locations in the file system. The ordering here must also be
  164.  164       // consistent with the one used in module_implements().
  165.  165       $result = db_query("SELECT * FROM {system} WHERE type = 'theme' OR (type = 'module' AND status = 1) ORDER BY weight ASC, name ASC");
  166.  166       foreach ($result as $record) {
  167.  167         $record->info = unserialize($record->info);
  168.  168         // Build a list of all enabled modules.
  169.  169         if ($record->type == 'module') {
  170.  170           $lists['module_enabled'][$record->name] = $record;
  171.  171         }
  172.  172         // Build a list of themes.
  173.  173         if ($record->type == 'theme') {
  174.  174           $lists['theme'][$record->name] = $record;
  175.  175         }
  176.  176         // Build a list of filenames so drupal_get_filename can use it.
  177.  177         if ($record->status) {
  178.  178           $lists['filepaths'][] = array('type' => $record->type, 'name' => $record->name, 'filepath' => $record->filename);
  179.  179         }
  180.  180       }
  181.  181       foreach ($lists['theme'] as $key => $theme) {
  182.  182         if (!empty($theme->info['base theme'])) {
  183.  183           // Make a list of the theme's base themes.
  184.  184           require_once DRUPAL_ROOT . '/includes/theme.inc';
  185.  185           $lists['theme'][$key]->base_themes = drupal_find_base_themes($lists['theme'], $key);
  186.  186           // Don't proceed if there was a problem with the root base theme.
  187.  187           if (!current($lists['theme'][$key]->base_themes)) {
  188.  188             continue;
  189.  189           }
  190.  190           // Determine the root base theme.
  191.  191           $base_key = key($lists['theme'][$key]->base_themes);
  192.  192           // Add to the list of sub-themes for each of the theme's base themes.
  193.  193           foreach (array_keys($lists['theme'][$key]->base_themes) as $base_theme) {
  194.  194             $lists['theme'][$base_theme]->sub_themes[$key] = $lists['theme'][$key]->info['name'];
  195.  195           }
  196.  196           // Add the base theme's theme engine info.
  197.  197           $lists['theme'][$key]->info['engine'] = isset($lists['theme'][$base_key]->info['engine']) ? $lists['theme'][$base_key]->info['engine'] : 'theme';
  198.  198         }
  199.  199         else {
  200.  200           // A plain theme is its own engine.
  201.  201           $base_key = $key;
  202.  202           if (!isset($lists['theme'][$key]->info['engine'])) {
  203.  203             $lists['theme'][$key]->info['engine'] = 'theme';
  204.  204           }
  205.  205         }
  206.  206         // Set the theme engine prefix.
  207.  207         $lists['theme'][$key]->prefix = ($lists['theme'][$key]->info['engine'] == 'theme') ? $base_key : $lists['theme'][$key]->info['engine'];
  208.  208       }
  209.  209       cache_set('system_list', $lists, 'cache_bootstrap');
  210.  210     }
  211.  211     // To avoid a separate database lookup for the filepath, prime the
  212.  212     // drupal_get_filename() static cache with all enabled modules and themes.
  213.  213     foreach ($lists['filepaths'] as $item) {
  214.  214       drupal_get_filename($item['type'], $item['name'], $item['filepath']);
  215.  215     }
  216.  216   }
  217.  217 
  218.  218   return $lists[$type];
  219.  219 }

  232.  232 /**
  233.  233  * Determines which modules require and are required by each module.
  234.  234  *
  235.  235  * @param $files
  236.  236  *   The array of filesystem objects used to rebuild the cache.
  237.  237  *
  238.  238  * @return
  239.  239  *   The same array with the new keys for each module:
  240.  240  *   - requires: An array with the keys being the modules that this module
  241.  241  *     requires.
  242.  242  *   - required_by: An array with the keys being the modules that will not work
  243.  243  *     without this module.
  244.  244  */
  245.  245 function _module_build_dependencies($files) {
  246.  246   require_once DRUPAL_ROOT . '/includes/graph.inc';
  247.  247   foreach ($files as $filename => $file) {
  248.  248     $graph[$file->name]['edges'] = array();
  249.  249     if (isset($file->info['dependencies']) && is_array($file->info['dependencies'])) {
  250.  250       foreach ($file->info['dependencies'] as $dependency) {
  251.  251         $dependency_data = drupal_parse_dependency($dependency);
  252.  252         $graph[$file->name]['edges'][$dependency_data['name']] = $dependency_data;
  253.  253       }
  254.  254     }
  255.  255   }
  256.  256   drupal_depth_first_search($graph);
  257.  257   foreach ($graph as $module => $data) {
  258.  258     $files[$module]->required_by = isset($data['reverse_paths']) ? $data['reverse_paths'] : array();
  259.  259     $files[$module]->requires = isset($data['paths']) ? $data['paths'] : array();
  260.  260     $files[$module]->sort = $data['weight'];
  261.  261   }
  262.  262   return $files;
  263.  263 }

  295.  295 /**

  297.  297  *
  298.  298  * Examples:
  299.  299  * @code
  300.  300  *   // Load node.admin.inc from the node module.

  307.  307  * instead. Do not use this function in a global context since it requires
  308.  308  * Drupal to be fully bootstrapped, use require_once DRUPAL_ROOT . '/path/file'
  309.  309  * instead.
  310.  310  *
  311.  311  * @param $type
  312.  312  *   The include file's type (file extension).
  313.  313  * @param $module
  314.  314  *   The module to which the include file belongs.
  315.  315  * @param $name
  316.  316  *   (optional) The base file name (without the $type extension). If omitted,
  317.  317  *   $module is used; i.e., resulting in "$module.$type" by default.
  318.  318  *
  319.  319  * @return
  320.  320  *   The name of the included file, if successful; FALSE otherwise.
  321.  321  */
  322.  322 function module_load_include($type, $module, $name = NULL) {
  323.  323   if (!isset($name)) {
  324.  324     $name = $module;
  325.  325   }
  326.  326 
  327.  327   if (function_exists('drupal_get_path')) {
  328.  328     $file = DRUPAL_ROOT . '/' . drupal_get_path('module', $module) . "/$name.$type";
  329.  329     if (is_file($file)) {
  330.  330       require_once $file;
  331.  331       return $file;
  332.  332     }
  333.  333   }
  334.  334   return FALSE;
  335.  335 }

  347.  347 /**
  348.  348  * Enables or installs a given list of modules.
  349.  349  *
  350.  350  * Definitions:
  351.  351  * - "Enabling" is the process of activating a module for use by Drupal.
  352.  352  * - "Disabling" is the process of deactivating a module.
  353.  353  * - "Installing" is the process of enabling it for the first time or after it
  354.  354  *   has been uninstalled.
  355.  355  * - "Uninstalling" is the process of removing all traces of a module.
  356.  356  *
  357.  357  * Order of events:
  358.  358  * - Gather and add module dependencies to $module_list (if applicable).
  359.  359  * - For each module that is being enabled:
  360.  360  *   - Install module schema and update system registries and caches.
  361.  361  *   - If the module is being enabled for the first time or had been
  362.  362  *     uninstalled, invoke hook_install() and add it to the list of installed
  363.  363  *     modules.
  364.  364  *   - Invoke hook_enable().
  365.  365  * - Invoke hook_modules_installed().
  366.  366  * - Invoke hook_modules_enabled().
  367.  367  *
  368.  368  * @param $module_list
  369.  369  *   An array of module names.
  370.  370  * @param $enable_dependencies
  371.  371  *   If TRUE, dependencies will automatically be added and enabled in the
  372.  372  *   correct order. This incurs a significant performance cost, so use FALSE
  373.  373  *   if you know $module_list is already complete and in the correct order.
  374.  374  *
  375.  375  * @return
  376.  376  *   FALSE if one or more dependencies are missing, TRUE otherwise.
  377.  377  *
  378.  378  * @see hook_install()
  379.  379  * @see hook_enable()
  380.  380  * @see hook_modules_installed()
  381.  381  * @see hook_modules_enabled()
  382.  382  */
  383.  383 function module_enable($module_list, $enable_dependencies = TRUE) {
  384.  384   if ($enable_dependencies) {
  385.  385     // Get all module data so we can find dependencies and sort.
  386.  386     $module_data = system_rebuild_module_data();
  387.  387     // Create an associative array with weights as values.
  388.  388     $module_list = array_flip(array_values($module_list));
  389.  389 
  390.  390     while (list($module) = each($module_list)) {
  391.  391       if (!isset($module_data[$module])) {
  392.  392         // This module is not found in the filesystem, abort.
  393.  393         return FALSE;
  394.  394       }
  395.  395       if ($module_data[$module]->status) {
  396.  396         // Skip already enabled modules.
  397.  397         unset($module_list[$module]);
  398.  398         continue;
  399.  399       }
  400.  400       $module_list[$module] = $module_data[$module]->sort;
  401.  401 
  402.  402       // Add dependencies to the list, with a placeholder weight.
  403.  403       // The new modules will be processed as the while loop continues.
  404.  404       foreach (array_keys($module_data[$module]->requires) as $dependency) {
  405.  405         if (!isset($module_list[$dependency])) {
  406.  406           $module_list[$dependency] = 0;
  407.  407         }
  408.  408       }
  409.  409     }
  410.  410 
  411.  411     if (!$module_list) {
  412.  412       // Nothing to do. All modules already enabled.
  413.  413       return TRUE;
  414.  414     }
  415.  415 
  416.  416     // Sort the module list by pre-calculated weights.
  417.  417     arsort($module_list);
  418.  418     $module_list = array_keys($module_list);
  419.  419   }
  420.  420 
  421.  421   // Required for module installation checks.
  422.  422   include_once DRUPAL_ROOT . '/includes/install.inc';
  423.  423 
  424.  424   $modules_installed = array();
  425.  425   $modules_enabled = array();
  426.  426   foreach ($module_list as $module) {
  427.  427     // Only process modules that are not already enabled.
  428.  428     $existing = db_query("SELECT status FROM {system} WHERE type = :type AND name = :name", array(
  429.  429       ':type' => 'module',
  430.  430       ':name' => $module))
  431.  431       ->fetchObject();
  432.  432     if ($existing->status == 0) {
  433.  433       // Load the module's code.
  434.  434       drupal_load('module', $module);
  435.  435       module_load_install($module);
  436.  436 
  437.  437       // Update the database and module list to reflect the new module. This
  438.  438       // needs to be done first so that the module's hook implementations,
  439.  439       // hook_schema() in particular, can be called while it is being
  440.  440       // installed.
  441.  441       db_update('system')
  442.  442         ->fields(array('status' => 1))
  443.  443         ->condition('type', 'module')
  444.  444         ->condition('name', $module)
  445.  445         ->execute();
  446.  446       // Refresh the module list to include it.
  447.  447       system_list_reset();
  448.  448       module_list(TRUE);
  449.  449       module_implements('', FALSE, TRUE);
  450.  450       _system_update_bootstrap_status();
  451.  451       // Update the registry to include it.
  452.  452       registry_update();
  453.  453       // Refresh the schema to include it.
  454.  454       drupal_get_schema(NULL, TRUE);
  455.  455       // Update the theme registry to include it.
  456.  456       drupal_theme_rebuild();
  457.  457       // Clear entity cache.
  458.  458       entity_info_cache_clear();
  459.  459 
  460.  460       // Now install the module if necessary.
  461.  461       if (drupal_get_installed_schema_version($module, TRUE) == SCHEMA_UNINSTALLED) {
  462.  462         drupal_install_schema($module);
  463.  463 
  464.  464         // Set the schema version to the number of the last update provided
  465.  465         // by the module.
  466.  466         $versions = drupal_get_schema_versions($module);
  467.  467         $version = $versions ? max($versions) : SCHEMA_INSTALLED;
  468.  468 
  469.  469         // If the module has no current updates, but has some that were
  470.  470         // previously removed, set the version to the value of
  471.  471         // hook_update_last_removed().
  472.  472         if ($last_removed = module_invoke($module, 'update_last_removed')) {
  473.  473           $version = max($version, $last_removed);
  474.  474         }
  475.  475         drupal_set_installed_schema_version($module, $version);
  476.  476         // Allow the module to perform install tasks.
  477.  477         module_invoke($module, 'install');
  478.  478         // Record the fact that it was installed.
  479.  479         $modules_installed[] = $module;
  480.  480         watchdog('system', '%module module installed.', array('%module' => $module), WATCHDOG_INFO);
  481.  481       }
  482.  482 
  483.  483       // Enable the module.
  484.  484       module_invoke($module, 'enable');
  485.  485 
  486.  486       // Record the fact that it was enabled.
  487.  487       $modules_enabled[] = $module;
  488.  488       watchdog('system', '%module module enabled.', array('%module' => $module), WATCHDOG_INFO);
  489.  489     }
  490.  490   }
  491.  491 
  492.  492   // If any modules were newly installed, invoke hook_modules_installed().
  493.  493   if (!empty($modules_installed)) {
  494.  494     module_invoke_all('modules_installed', $modules_installed);
  495.  495   }
  496.  496 
  497.  497   // If any modules were newly enabled, invoke hook_modules_enabled().
  498.  498   if (!empty($modules_enabled)) {
  499.  499     module_invoke_all('modules_enabled', $modules_enabled);
  500.  500   }
  501.  501 
  502.  502   return TRUE;
  503.  503 }
  504.  504 
  505.  505 /**
  506.  506  * Disables a given set of modules.
  507.  507  *
  508.  508  * @param $module_list
  509.  509  *   An array of module names.
  510.  510  * @param $disable_dependents
  511.  511  *   If TRUE, dependent modules will automatically be added and disabled in the
  512.  512  *   correct order. This incurs a significant performance cost, so use FALSE
  513.  513  *   if you know $module_list is already complete and in the correct order.
  514.  514  */
  515.  515 function module_disable($module_list, $disable_dependents = TRUE) {
  516.  516   if ($disable_dependents) {
  517.  517     // Get all module data so we can find dependents and sort.
  518.  518     $module_data = system_rebuild_module_data();
  519.  519     // Create an associative array with weights as values.
  520.  520     $module_list = array_flip(array_values($module_list));
  521.  521 
  522.  522     $profile = drupal_get_profile();
  523.  523     while (list($module) = each($module_list)) {
  524.  524       if (!isset($module_data[$module]) || !$module_data[$module]->status) {
  525.  525         // This module doesn't exist or is already disabled, skip it.
  526.  526         unset($module_list[$module]);
  527.  527         continue;
  528.  528       }
  529.  529       $module_list[$module] = $module_data[$module]->sort;
  530.  530 
  531.  531       // Add dependent modules to the list, with a placeholder weight.
  532.  532       // The new modules will be processed as the while loop continues.
  533.  533       foreach ($module_data[$module]->required_by as $dependent => $dependent_data) {
  534.  534         if (!isset($module_list[$dependent]) && $dependent != $profile) {
  535.  535           $module_list[$dependent] = 0;
  536.  536         }
  537.  537       }
  538.  538     }
  539.  539 
  540.  540     // Sort the module list by pre-calculated weights.
  541.  541     asort($module_list);
  542.  542     $module_list = array_keys($module_list);
  543.  543   }
  544.  544 
  545.  545   $invoke_modules = array();
  546.  546 
  547.  547   foreach ($module_list as $module) {
  548.  548     if (module_exists($module)) {
  549.  549       // Check if node_access table needs rebuilding.
  550.  550       if (!node_access_needs_rebuild() && module_hook($module, 'node_grants')) {
  551.  551         node_access_needs_rebuild(TRUE);
  552.  552       }
  553.  553 
  554.  554       module_load_install($module);
  555.  555       module_invoke($module, 'disable');
  556.  556       db_update('system')
  557.  557         ->fields(array('status' => 0))
  558.  558         ->condition('type', 'module')
  559.  559         ->condition('name', $module)
  560.  560         ->execute();
  561.  561       $invoke_modules[] = $module;
  562.  562       watchdog('system', '%module module disabled.', array('%module' => $module), WATCHDOG_INFO);
  563.  563     }
  564.  564   }
  565.  565 
  566.  566   if (!empty($invoke_modules)) {
  567.  567     // Refresh the module list to exclude the disabled modules.
  568.  568     system_list_reset();
  569.  569     module_list(TRUE);
  570.  570     module_implements('', FALSE, TRUE);
  571.  571     entity_info_cache_clear();
  572.  572     // Invoke hook_modules_disabled before disabling modules,
  573.  573     // so we can still call module hooks to get information.
  574.  574     module_invoke_all('modules_disabled', $invoke_modules);
  575.  575     // Update the registry to remove the newly-disabled module.
  576.  576     registry_update();
  577.  577     _system_update_bootstrap_status();
  578.  578     // Update the theme registry to remove the newly-disabled module.
  579.  579     drupal_theme_rebuild();
  580.  580   }
  581.  581 
  582.  582   // If there remains no more node_access module, rebuilding will be
  583.  583   // straightforward, we can do it right now.
  584.  584   if (node_access_needs_rebuild() && count(module_implements('node_grants')) == 0) {
  585.  585     node_access_rebuild();
  586.  586   }
  587.  587 }
  588.  588 
  589.  589 /**
  590.  590  * @defgroup hooks Hooks
  591.  591  * @{
  592.  592  * Allow modules to interact with the Drupal core.
  593.  593  *
  594.  594  * Drupal's module system is based on the concept of "hooks". A hook is a PHP
  595.  595  * function that is named foo_bar(), where "foo" is the name of the module
  596.  596  * (whose filename is thus foo.module) and "bar" is the name of the hook. Each
  597.  597  * hook has a defined set of parameters and a specified result type.
  598.  598  *
  599.  599  * To extend Drupal, a module need simply implement a hook. When Drupal wishes
  600.  600  * to allow intervention from modules, it determines which modules implement a
  601.  601  * hook and calls that hook in all enabled modules that implement it.
  602.  602  *
  603.  603  * The available hooks to implement are explained here in the Hooks section of
  604.  604  * the developer documentation. The string "hook" is used as a placeholder for
  605.  605  * the module name in the hook definitions. For example, if the module file is
  606.  606  * called example.module, then hook_help() as implemented by that module would
  607.  607  * be defined as example_help().
  608.  608  *
  609.  609  * The example functions included are not part of the Drupal core, they are
  610.  610  * just models that you can modify. Only the hooks implemented within modules
  611.  611  * are executed when running Drupal.
  612.  612  *
  613.  613  * @see themeable
  614.  614  * @see callbacks
  615.  615  */
  616.  616 
  617.  617  /**
  618.  618   * @defgroup callbacks Callbacks
  619.  619   * @{
  620.  620   * Callback function signatures.
  621.  621   *
  622.  622   * Drupal's API sometimes uses callback functions to allow you to define how
  623.  623   * some type of processing happens. A callback is a function with a defined
  624.  624   * signature, which you define in a module. Then you pass the function name as
  625.  625   * a parameter to a Drupal API function or return it as part of a hook
  626.  626   * implementation return value, and your function is called at an appropriate
  627.  627   * time. For instance, when setting up batch processing you might need to
  628.  628   * provide a callback function for each processing step and/or a callback for
  629.  629   * when processing is finished; you would do that by defining these functions
  630.  630   * and passing their names into the batch setup function.
  631.  631   *
  632.  632   * Callback function signatures, like hook definitions, are described by
  633.  633   * creating and documenting dummy functions in a *.api.php file; normally, the
  634.  634   * dummy callback function's name should start with "callback_", and you should
  635.  635   * document the parameters and return value and provide a sample function body.
  636.  636   * Then your API documentation can refer to this callback function in its
  637.  637   * documentation. A user of your API can usually name their callback function
  638.  638   * anything they want, although a standard name would be to replace "callback_"
  639.  639   * with the module name.
  640.  640   *
  641.  641   * @see hooks
  642.  642   * @see themeable
  643.  643   *
  644.  644   * @}
  645.  645   */
  646.  646 
  647.  647 /**
  648.  648  * Determines whether a module implements a hook.
  649.  649  *
  650.  650  * @param $module
  651.  651  *   The name of the module (without the .module extension).
  652.  652  * @param $hook
  653.  653  *   The name of the hook (e.g. "help" or "menu").
  654.  654  *
  655.  655  * @return
  656.  656  *   TRUE if the module is both installed and enabled, and the hook is
  657.  657  *   implemented in that module.
  658.  658  */
  659.  659 function module_hook($module, $hook) {
  660.  660   $function = $module . '_' . $hook;
  661.  661   if (function_exists($function)) {
  662.  662     return TRUE;
  663.  663   }
  664.  664   // If the hook implementation does not exist, check whether it may live in an
  665.  665   // optional include file registered via hook_hook_info().
  666.  666   $hook_info = module_hook_info();
  667.  667   if (isset($hook_info[$hook]['group'])) {
  668.  668     module_load_include('inc', $module, $module . '.' . $hook_info[$hook]['group']);
  669.  669     if (function_exists($function)) {
  670.  670       return TRUE;
  671.  671     }
  672.  672   }
  673.  673   return FALSE;
  674.  674 }
  675.  675 
  676.  676 /**
  677.  677  * Determines which modules are implementing a hook.
  678.  678  *
  679.  679  * @param $hook
  680.  680  *   The name of the hook (e.g. "help" or "menu").
  681.  681  * @param $sort
  682.  682  *   By default, modules are ordered by weight and filename, settings this option
  683.  683  *   to TRUE, module list will be ordered by module name.
  684.  684  * @param $reset
  685.  685  *   For internal use only: Whether to force the stored list of hook
  686.  686  *   implementations to be regenerated (such as after enabling a new module,
  687.  687  *   before processing hook_enable).
  688.  688  *
  689.  689  * @return
  690.  690  *   An array with the names of the modules which are implementing this hook.
  691.  691  *
  692.  692  * @see module_implements_write_cache()
  693.  693  */
  694.  694 function module_implements($hook, $sort = FALSE, $reset = FALSE) {
  695.  695   // Use the advanced drupal_static() pattern, since this is called very often.
  696.  696   static $drupal_static_fast;
  697.  697   if (!isset($drupal_static_fast)) {
  698.  698     $drupal_static_fast['implementations'] = &drupal_static(__FUNCTION__);
  699.  699   }
  700.  700   $implementations = &$drupal_static_fast['implementations'];
  701.  701 
  702.  702   // We maintain a persistent cache of hook implementations in addition to the
  703.  703   // static cache to avoid looping through every module and every hook on each
  704.  704   // request. Benchmarks show that the benefit of this caching outweighs the
  705.  705   // additional database hit even when using the default database caching
  706.  706   // backend and only a small number of modules are enabled. The cost of the
  707.  707   // cache_get() is more or less constant and reduced further when non-database
  708.  708   // caching backends are used, so there will be more significant gains when a
  709.  709   // large number of modules are installed or hooks invoked, since this can
  710.  710   // quickly lead to module_hook() being called several thousand times
  711.  711   // per request.
  712.  712   if ($reset) {
  713.  713     $implementations = array();

  789.  789   // case common.inc, subsystems, and modules are not loaded yet, so it does not

  897.  897         $return = array_merge_recursive($return, $result);
  898.  898       }
  899.  899       elseif (isset($result)) {
  900.  900         $return[] = $result;
  901.  901       }
  902.  902     }
  903.  903   }
  904.  904 
  905.  905   return $return;
  906.  906 }

  932.  932 /**
  933.  933  * Passes alterable variables to specific hook_TYPE_alter() implementations.
  934.  934  *
  935.  935  * This dispatch function hands off the passed-in variables to type-specific
  936.  936  * hook_TYPE_alter() implementations in modules. It ensures a consistent
  937.  937  * interface for all altering operations.
  938.  938  *
  939.  939  * A maximum of 2 alterable arguments is supported (a third is supported for
  940.  940  * legacy reasons, but should not be used in new code). In case more arguments
  941.  941  * need to be passed and alterable, modules provide additional variables
  942.  942  * assigned by reference in the last $context argument:
  943.  943  * @code
  944.  944  *   $context = array(
  945.  945  *     'alterable' => &$alterable,
  946.  946  *     'unalterable' => $unalterable,
  947.  947  *     'foo' => 'bar',
  948.  948  *   );
  949.  949  *   drupal_alter('mymodule_data', $alterable1, $alterable2, $context);
  950.  950  * @endcode
  951.  951  *
  952.  952  * Note that objects are always passed by reference in PHP5. If it is absolutely
  953.  953  * required that no implementation alters a passed object in $context, then an
  954.  954  * object needs to be cloned:
  955.  955  * @code
  956.  956  *   $context = array(
  957.  957  *     'unalterable_object' => clone $object,
  958.  958  *   );
  959.  959  *   drupal_alter('mymodule_data', $data, $context);
  960.  960  * @endcode
  961.  961  *
  962.  962  * @param $type
  963.  963  *   A string describing the type of the alterable $data. 'form', 'links',
  964.  964  *   'node_content', and so on are several examples. Alternatively can be an
  965.  965  *   array, in which case hook_TYPE_alter() is invoked for each value in the
  966.  966  *   array, ordered first by module, and then for each module, in the order of
  967.  967  *   values in $type. For example, when Form API is using drupal_alter() to
  968.  968  *   execute both hook_form_alter() and hook_form_FORM_ID_alter()
  969.  969  *   implementations, it passes array('form', 'form_' . $form_id) for $type.
  970.  970  * @param $data
  971.  971  *   The variable that will be passed to hook_TYPE_alter() implementations to be
  972.  972  *   altered. The type of this variable depends on the value of the $type
  973.  973  *   argument. For example, when altering a 'form', $data will be a structured
  974.  974  *   array. When altering a 'profile', $data will be an object.
  975.  975  * @param $context1
  976.  976  *   (optional) An additional variable that is passed by reference.
  977.  977  * @param $context2
  978.  978  *   (optional) An additional variable that is passed by reference. If more
  979.  979  *   context needs to be provided to implementations, then this should be an
  980.  980  *   associative array as described above.
  981.  981  * @param $context3
  982.  982  *   (optional) An additional variable that is passed by reference. This
  983.  983  *   parameter is deprecated and will not exist in Drupal 8; consequently, it
  984.  984  *   should not be used for new Drupal 7 code either. It is here only for
  985.  985  *   backwards compatibility with older code that passed additional arguments
  986.  986  *   to drupal_alter().
  987.  987  */
  988.  988 function drupal_alter($type, &$data, &$context1 = NULL, &$context2 = NULL, &$context3 = NULL) {
  989.  989   // Use the advanced drupal_static() pattern, since this is called very often.
  990.  990   static $drupal_static_fast;
  991.  991   if (!isset($drupal_static_fast)) {
  992.  992     $drupal_static_fast['functions'] = &drupal_static(__FUNCTION__);
  993.  993   }
  994.  994   $functions = &$drupal_static_fast['functions'];
  995.  995 
  996.  996   // Most of the time, $type is passed as a string, so for performance,
  997.  997   // normalize it to that. When passed as an array, usually the first item in
  998.  998   // the array is a generic type, and additional items in the array are more
  999.  999   // specific variants of it, as in the case of array('form', 'form_FORM_ID').
  1000. 1000   if (is_array($type)) {
  1001. 1001     $cid = implode(',', $type);
  1002. 1002     $extra_types = $type;
  1003. 1003     $type = array_shift($extra_types);
  1004. 1004     // Allow if statements in this function to use the faster isset() rather
  1005. 1005     // than !empty() both when $type is passed as a string, or as an array with
  1006. 1006     // one item.
  1007. 1007     if (empty($extra_types)) {
  1008. 1008       unset($extra_types);
  1009. 1009     }
  1010. 1010   }
  1011. 1011   else {
  1012. 1012     $cid = $type;
  1013. 1013   }
  1014. 1014 
  1015. 1015   // Some alter hooks are invoked many times per page request, so statically
  1016. 1016   // cache the list of functions to call, and on subsequent calls, iterate
  1017. 1017   // through them quickly.
  1018. 1018   if (!isset($functions[$cid])) {
  1019. 1019     $functions[$cid] = array();
  1020. 1020     $hook = $type . '_alter';
  1021. 1021     $modules = module_implements($hook);
  1022. 1022     if (!isset($extra_types)) {
  1023. 1023       // For the more common case of a single hook, we do not need to call
  1024. 1024       // function_exists(), since module_implements() returns only modules with
  1025. 1025       // implementations.
  1026. 1026       foreach ($modules as $module) {
  1027. 1027         $functions[$cid][] = $module . '_' . $hook;
  1028. 1028       }
  1029. 1029     }
  1030. 1030     else {
  1031. 1031       // For multiple hooks, we need $modules to contain every module that
  1032. 1032       // implements at least one of them.
  1033. 1033       $extra_modules = array();
  1034. 1034       foreach ($extra_types as $extra_type) {
  1035. 1035         $extra_modules = array_merge($extra_modules, module_implements($extra_type . '_alter'));
  1036. 1036       }
  1037. 1037       // If any modules implement one of the extra hooks that do not implement
  1038. 1038       // the primary hook, we need to add them to the $modules array in their
  1039. 1039       // appropriate order. module_implements() can only return ordered
  1040. 1040       // implementations of a single hook. To get the ordered implementations
  1041. 1041       // of multiple hooks, we mimic the module_implements() logic of first
  1042. 1042       // ordering by module_list(), and then calling
  1043. 1043       // drupal_alter('module_implements').
  1044. 1044       if (array_diff($extra_modules, $modules)) {
  1045. 1045         // Merge the arrays and order by module_list().
  1046. 1046         $modules = array_intersect(module_list(), array_merge($modules, $extra_modules));
  1047. 1047         // Since module_implements() already took care of loading the necessary
  1048. 1048         // include files, we can safely pass FALSE for the array values.
  1049. 1049         $implementations = array_fill_keys($modules, FALSE);
  1050. 1050         // Let modules adjust the order solely based on the primary hook. This
