  63.  63 WebFXCookie.prototype.getCookie = function (key) {
  64.  64  if (this.cookies) {
  65.  65  var start = this.cookies.indexOf(' ' + key + '=');
  66.  66  if (start == -1) { return null; }
  67.  67  var end = this.cookies.indexOf(";", start);
  68.  68  if (end == -1) { end = this.cookies.length; }
  69.  69  end -= start;
  70.  70  var cookie = this.cookies.substr(start,end);
  71.  71  return unescape(cookie.substr(cookie.indexOf('=') + 1, cookie.length - cookie.indexOf('=') + 1));
  72.  72  }
  73.  73  else { return null; }
  74.  74 }

  90.  90 WebFXTreeAbstractNode.prototype.add = function (node) {
  91.  91  node.parentNode = this;
  92.  92  this.childNodes[this.childNodes.length] = node;
  93.  93  var root = this;
  94.  94  if (this.childNodes.length >=2) {
  95.  95  this.childNodes[this.childNodes.length -2]._last = false;
  96.  96  }
  97.  97  while (root.parentNode) { root = root.parentNode; }
  98.  98  if (root.rendered) {
  99.  99  if (this.childNodes.length >= 2) {
  100. 100  document.getElementById(this.childNodes[this.childNodes.length -2].id + '-plus').src = ((this.childNodes[this.childNodes.length -2].folder)?webFXTreeConfig.tMinusIcon:webFXTreeConfig.tIcon);
  101. 101  if (this.childNodes[this.childNodes.length -2].folder) {
  102. 102  this.childNodes[this.childNodes.length -2].plusIcon = webFXTreeConfig.tPlusIcon;
  103. 103  this.childNodes[this.childNodes.length -2].minusIcon = webFXTreeConfig.tMinusIcon;
  104. 104  }
  105. 105  this.childNodes[this.childNodes.length -2]._last = false;
  106. 106  }
  107. 107  this._last = true;
  108. 108  var foo = this;
  109. 109  while (foo.parentNode) {
  110. 110  for (var i = 0; i < foo.parentNode.childNodes.length; i++) {
  111. 111  if (foo.id == foo.parentNode.childNodes[i].id) { break; }
  112. 112  }
  113. 113  if (++i == foo.parentNode.childNodes.length) { foo.parentNode._last = true; }
  114. 114  else { foo.parentNode._last = false; }
  115. 115  foo = foo.parentNode;
  116. 116  }
  117. 117  document.getElementById(this.id + '-cont').insertAdjacentHTML("beforeEnd", node.toString());
  118. 118  if ((!this.folder) && (!this.openIcon)) {
  119. 119  this.icon = webFXTreeConfig.folderIcon;
  120. 120  this.openIcon = webFXTreeConfig.openFolderIcon;
  121. 121  }
  122. 122  this.folder = true;
  123. 123  this.indent();
  124. 124  this.expand();
  125. 125  }
  126. 126  return node;
  127. 127 }

  188. 188 WebFXTreeAbstractNode.prototype.indent = function(lvl, del, last, level) {
  189. 189  /*
  190. 190   * Since we only want to modify items one level below ourself,
  191. 191   * and since the rightmost indentation position is occupied by
  192. 192   * the plus icon we set this to -2
  193. 193   */
  194. 194  if (lvl == null) { lvl = -2; }
  195. 195  var state = 0;
  196. 196  for (var i = this.childNodes.length - 1; i >= 0 ; i--) {
  197. 197  state = this.childNodes[i].indent(lvl + 1, del, last, level);
  198. 198  if (state) { return; }
  199. 199  }
  200. 200  if (del) {
  201. 201  if (level >= this._level) {
  202. 202  if (this.folder) {
  203. 203  document.getElementById(this.id + '-plus').src = (this.open)?webFXTreeConfig.lMinusIcon:webFXTreeConfig.lPlusIcon;
  204. 204  this.plusIcon = webFXTreeConfig.lPlusIcon;
  205. 205  this.minusIcon = webFXTreeConfig.lMinusIcon;
  206. 206  }
  207. 207  else { document.getElementById(this.id + '-plus').src = webFXTreeConfig.lIcon; }
  208. 208  return 1;
  209. 209  }
  210. 210  }
  211. 211  var foo = document.getElementById(this.id + '-indent-' + lvl);
  212. 212  if (foo) {
  213. 213  if ((del) && (last)) { foo._last = true; }
  214. 214  if (foo._last) { foo.src =  webFXTreeConfig.blankIcon; }
  215. 215  else { foo.src =  webFXTreeConfig.iIcon; }
  216. 216  }
  217. 217  return 0;
  218. 218 }
