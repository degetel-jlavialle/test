  354.  354         break;
  355.  355       case 38: // Up arrow.
  356.  356       case 63232: // Safari up arrow.
  357.  357         var previousRow = $(self.rowObject.element).prev('tr').get(0);
  358.  358         while (previousRow && $(previousRow).is(':hidden')) {
  359.  359           previousRow = $(previousRow).prev('tr').get(0);
  360.  360         }
  361.  361         if (previousRow) {
  362.  362           self.safeBlur = false; // Do not allow the onBlur cleanup.
  363.  363           self.rowObject.direction = 'up';
  364.  364           keyChange = true;
  365.  365 
  366.  366           if ($(item).is('.tabledrag-root')) {
  367.  367             // Swap with the previous top-level row.
  368.  368             var groupHeight = 0;
  369.  369             while (previousRow && $('.indentation', previousRow).length) {
  370.  370               previousRow = $(previousRow).prev('tr').get(0);
  371.  371               groupHeight += $(previousRow).is(':hidden') ? 0 : previousRow.offsetHeight;
  372.  372             }
  373.  373             if (previousRow) {
  374.  374               self.rowObject.swap('before', previousRow);
  375.  375               // No need to check for indentation, 0 is the only valid one.
  376.  376               window.scrollBy(0, -groupHeight);
  377.  377             }
  378.  378           }
  379.  379           else if (self.table.tBodies[0].rows[0] != previousRow || $(previousRow).is('.draggable')) {
  380.  380             // Swap with the previous row (unless previous row is the first one
  381.  381             // and undraggable).
  382.  382             self.rowObject.swap('before', previousRow);
  383.  383             self.rowObject.interval = null;
  384.  384             self.rowObject.indent(0);
  385.  385             window.scrollBy(0, -parseInt(item.offsetHeight, 10));
  386.  386           }
  387.  387           handle.get(0).focus(); // Regain focus after the DOM manipulation.
  388.  388         }
  389.  389         break;
  390.  390       case 39: // Right arrow.
  391.  391       case 63235: // Safari right arrow.
  392.  392         keyChange = true;
  393.  393         self.rowObject.indent(1 * self.rtl);
  394.  394         break;
  395.  395       case 40: // Down arrow.
  396.  396       case 63233: // Safari down arrow.
  397.  397         var nextRow = $(self.rowObject.group).filter(':last').next('tr').get(0);
  398.  398         while (nextRow && $(nextRow).is(':hidden')) {
  399.  399           nextRow = $(nextRow).next('tr').get(0);
  400.  400         }
  401.  401         if (nextRow) {
  402.  402           self.safeBlur = false; // Do not allow the onBlur cleanup.
  403.  403           self.rowObject.direction = 'down';
  404.  404           keyChange = true;
  405.  405 
  406.  406           if ($(item).is('.tabledrag-root')) {
  407.  407             // Swap with the next group (necessarily a top-level one).
  408.  408             var groupHeight = 0;
  409.  409             var nextGroup = new self.row(nextRow, 'keyboard', self.indentEnabled, self.maxDepth, false);
  410.  410             if (nextGroup) {
  411.  411               $(nextGroup.group).each(function () {
  412.  412                 groupHeight += $(this).is(':hidden') ? 0 : this.offsetHeight;
  413.  413               });
  414.  414               var nextGroupRow = $(nextGroup.group).filter(':last').get(0);
  415.  415               self.rowObject.swap('after', nextGroupRow);
  416.  416               // No need to check for indentation, 0 is the only valid one.
  417.  417               window.scrollBy(0, parseInt(groupHeight, 10));
  418.  418             }
  419.  419           }
  420.  420           else {
  421.  421             // Swap with the next row.
  422.  422             self.rowObject.swap('after', nextRow);
  423.  423             self.rowObject.interval = null;
  424.  424             self.rowObject.indent(0);
  425.  425             window.scrollBy(0, parseInt(item.offsetHeight, 10));
  426.  426           }
  427.  427           handle.get(0).focus(); // Regain focus after the DOM manipulation.
  428.  428         }
  429.  429         break;
  430.  430     }
  431.  431 
  432.  432     if (self.rowObject && self.rowObject.changed == true) {
  433.  433       $(item).addClass('drag');
  434.  434       if (self.oldRowElement) {
  435.  435         $(self.oldRowElement).removeClass('drag-previous');
  436.  436       }
  437.  437       self.oldRowElement = item;
  438.  438       self.restripeTable();
  439.  439       self.onDrag();
  440.  440     }
  441.  441 
  442.  442     // Returning false if we have an arrow key to prevent scrolling.
  443.  443     if (keyChange) {
  444.  444       return false;
  445.  445     }
  446.  446   });
  447.  447 
  448.  448   // Compatibility addition, return false on keypress to prevent unwanted scrolling.
  449.  449   // IE and Safari will suppress scrolling on keydown, but all other browsers
  450.  450   // need to return false on keypress. http://www.quirksmode.org/js/keys.html
  451.  451   handle.keypress(function (event) {
  452.  452     switch (event.keyCode) {
  453.  453       case 37: // Left arrow.
  454.  454       case 38: // Up arrow.
  455.  455       case 39: // Right arrow.
  456.  456       case 40: // Down arrow.
  457.  457         return false;
  458.  458     }
  459.  459   });
  460.  460 };
  461.  461 
  462.  462 /**
  463.  463  * Mousemove event handler, bound to document.
  464.  464  */
  465.  465 Drupal.tableDrag.prototype.dragRow = function (event, self) {
  466.  466   if (self.dragObject) {
  467.  467     self.currentMouseCoords = self.mouseCoords(event);
  468.  468 
  469.  469     var y = self.currentMouseCoords.y - self.dragObject.initMouseOffset.y;
  470.  470     var x = self.currentMouseCoords.x - self.dragObject.initMouseOffset.x;
  471.  471 
  472.  472     // Check for row swapping and vertical scrolling.
  473.  473     if (y != self.oldY) {
  474.  474       self.rowObject.direction = y > self.oldY ? 'down' : 'up';
  475.  475       self.oldY = y; // Update the old value.
  476.  476 
  477.  477       // Check if the window should be scrolled (and how fast).
  478.  478       var scrollAmount = self.checkScroll(self.currentMouseCoords.y);
  479.  479       // Stop any current scrolling.
  480.  480       clearInterval(self.scrollInterval);
  481.  481       // Continue scrolling if the mouse has moved in the scroll direction.
  482.  482       if (scrollAmount > 0 && self.rowObject.direction == 'down' || scrollAmount < 0 && self.rowObject.direction == 'up') {
  483.  483         self.setScroll(scrollAmount);
  484.  484       }
  485.  485 
  486.  486       // If we have a valid target, perform the swap and restripe the table.
  487.  487       var currentRow = self.findDropTargetRow(x, y);
  488.  488       if (currentRow) {
  489.  489         if (self.rowObject.direction == 'down') {
  490.  490           self.rowObject.swap('after', currentRow, self);
  491.  491         }
  492.  492         else {
  493.  493           self.rowObject.swap('before', currentRow, self);
  494.  494         }
  495.  495         self.restripeTable();
  496.  496       }
  497.  497     }
  498.  498 
  499.  499     // Similar to row swapping, handle indentations.
  500.  500     if (self.indentEnabled) {
  501.  501       var xDiff = self.currentMouseCoords.x - self.dragObject.indentMousePos.x;
  502.  502       // Set the number of indentations the mouse has been moved left or right.
  503.  503       var indentDiff = Math.round(xDiff / self.indentAmount * self.rtl);
  504.  504       // Indent the row with our estimated diff, which may be further
  505.  505       // restricted according to the rows around this row.
  506.  506       var indentChange = self.rowObject.indent(indentDiff);
  507.  507       // Update table and mouse indentations.
  508.  508       self.dragObject.indentMousePos.x += self.indentAmount * indentChange * self.rtl;
  509.  509       self.indentCount = Math.max(self.indentCount, self.rowObject.indents);
  510.  510     }
  511.  511 
  512.  512     return false;
  513.  513   }
  514.  514 };
  515.  515 
  516.  516 /**
  517.  517  * Mouseup event handler, bound to document.
  518.  518  * Blur event handler, bound to drag handle for keyboard support.
  519.  519  */
  520.  520 Drupal.tableDrag.prototype.dropRow = function (event, self) {
  521.  521   // Drop row functionality shared between mouseup and blur events.
  522.  522   if (self.rowObject != null) {
  523.  523     var droppedRow = self.rowObject.element;
  524.  524     // The row is already in the right place so we just release it.
  525.  525     if (self.rowObject.changed == true) {
  526.  526       // Update the fields in the dropped row.
  527.  527       self.updateFields(droppedRow);
  528.  528 
  529.  529       // If a setting exists for affecting the entire group, update all the
  530.  530       // fields in the entire dragged group.
  531.  531       for (var group in self.tableSettings) {
  532.  532         var rowSettings = self.rowSettings(group, droppedRow);
  533.  533         if (rowSettings.relationship == 'group') {
  534.  534           for (var n in self.rowObject.children) {
  535.  535             self.updateField(self.rowObject.children[n], group);
  536.  536           }
  537.  537         }
  538.  538       }
  539.  539 
  540.  540       self.rowObject.markChanged();
  541.  541       if (self.changed == false) {
  542.  542         $(Drupal.theme('tableDragChangedWarning')).insertBefore(self.table).hide().fadeIn('slow');
  543.  543         self.changed = true;
  544.  544       }
  545.  545     }
  546.  546 
  547.  547     if (self.indentEnabled) {
  548.  548       self.rowObject.removeIndentClasses();
  549.  549     }
  550.  550     if (self.oldRowElement) {
  551.  551       $(self.oldRowElement).removeClass('drag-previous');
  552.  552     }
  553.  553     $(droppedRow).removeClass('drag').addClass('drag-previous');
  554.  554     self.oldRowElement = droppedRow;
  555.  555     self.onDrop();
  556.  556     self.rowObject = null;
  557.  557   }
  558.  558 
  559.  559   // Functionality specific only to mouseup event.
  560.  560   if (self.dragObject != null) {
  561.  561     $('.tabledrag-handle', droppedRow).removeClass('tabledrag-handle-hover');
  562.  562 
  563.  563     self.dragObject = null;
  564.  564     $('body').removeClass('drag');
  565.  565     clearInterval(self.scrollInterval);
  566.  566 
  567.  567     // Hack for IE6 that flickers uncontrollably if select lists are moved.
  568.  568     if (navigator.userAgent.indexOf('MSIE 6.') != -1) {
  569.  569       $('select', this.table).css('display', 'block');
  570.  570     }
  571.  571   }
  572.  572 };
  573.  573 
  574.  574 /**
  575.  575  * Get the mouse coordinates from the event (allowing for browser differences).
  576.  576  */
  577.  577 Drupal.tableDrag.prototype.mouseCoords = function (event) {
  578.  578   if (event.pageX || event.pageY) {
  579.  579     return { x: event.pageX, y: event.pageY };
  580.  580   }
  581.  581   return {
  582.  582     x: event.clientX + document.body.scrollLeft - document.body.clientLeft,
  583.  583     y: event.clientY + document.body.scrollTop  - document.body.clientTop
  584.  584   };
  585.  585 };
  586.  586 
  587.  587 /**
  588.  588  * Given a target element and a mouse event, get the mouse offset from that
  589.  589  * element. To do this we need the element's position and the mouse position.
  590.  590  */
  591.  591 Drupal.tableDrag.prototype.getMouseOffset = function (target, event) {
  592.  592   var docPos   = $(target).offset();
  593.  593   var mousePos = this.mouseCoords(event);
  594.  594   return { x: mousePos.x - docPos.left, y: mousePos.y - docPos.top };
  595.  595 };
  596.  596 
  597.  597 /**
  598.  598  * Find the row the mouse is currently over. This row is then taken and swapped
  599.  599  * with the one being dragged.
  600.  600  *
  601.  601  * @param x
  602.  602  *   The x coordinate of the mouse on the page (not the screen).
  603.  603  * @param y
  604.  604  *   The y coordinate of the mouse on the page (not the screen).
  605.  605  */
  606.  606 Drupal.tableDrag.prototype.findDropTargetRow = function (x, y) {
  607.  607   var rows = $(this.table.tBodies[0].rows).not(':hidden');
  608.  608   for (var n = 0; n < rows.length; n++) {
  609.  609     var row = rows[n];
  610.  610     var indentDiff = 0;
  611.  611     var rowY = $(row).offset().top;
  612.  612     // Because Safari does not report offsetHeight on table rows, but does on
  613.  613     // table cells, grab the firstChild of the row and use that instead.
  614.  614     // http://jacob.peargrove.com/blog/2006/technical/table-row-offsettop-bug-in-safari.
  615.  615     if (row.offsetHeight == 0) {
  616.  616       var rowHeight = parseInt(row.firstChild.offsetHeight, 10) / 2;
  617.  617     }
  618.  618     // Other browsers.
  619.  619     else {
  620.  620       var rowHeight = parseInt(row.offsetHeight, 10) / 2;
  621.  621     }
  622.  622 
  623.  623     // Because we always insert before, we need to offset the height a bit.
  624.  624     if ((y > (rowY - rowHeight)) && (y < (rowY + rowHeight))) {
  625.  625       if (this.indentEnabled) {
  626.  626         // Check that this row is not a child of the row being dragged.
  627.  627         for (var n in this.rowObject.group) {
  628.  628           if (this.rowObject.group[n] == row) {
  629.  629             return null;
  630.  630           }
  631.  631         }
  632.  632       }
  633.  633       else {
  634.  634         // Do not allow a row to be swapped with itself.
  635.  635         if (row == this.rowObject.element) {
  636.  636           return null;
  637.  637         }
  638.  638       }
  639.  639 
  640.  640       // Check that swapping with this row is allowed.
  641.  641       if (!this.rowObject.isValidSwap(row)) {
  642.  642         return null;
  643.  643       }
  644.  644 
  645.  645       // We may have found the row the mouse just passed over, but it doesn't
  646.  646       // take into account hidden rows. Skip backwards until we find a draggable
  647.  647       // row.
  648.  648       while ($(row).is(':hidden') && $(row).prev('tr').is(':hidden')) {
  649.  649         row = $(row).prev('tr').get(0);
  650.  650       }
  651.  651       return row;
  652.  652     }
  653.  653   }
  654.  654   return null;
  655.  655 };
  656.  656 
  657.  657 /**
  658.  658  * After the row is dropped, update the table fields according to the settings
  659.  659  * set for this table.
  660.  660  *
  661.  661  * @param changedRow
  662.  662  *   DOM object for the row that was just dropped.
  663.  663  */
  664.  664 Drupal.tableDrag.prototype.updateFields = function (changedRow) {
  665.  665   for (var group in this.tableSettings) {
  666.  666     // Each group may have a different setting for relationship, so we find
  667.  667     // the source rows for each separately.
  668.  668     this.updateField(changedRow, group);
  669.  669   }
  670.  670 };
  671.  671 
  672.  672 /**
  673.  673  * After the row is dropped, update a single table field according to specific
  674.  674  * settings.
  675.  675  *
  676.  676  * @param changedRow
  677.  677  *   DOM object for the row that was just dropped.
  678.  678  * @param group
  679.  679  *   The settings group on which field updates will occur.
  680.  680  */
  681.  681 Drupal.tableDrag.prototype.updateField = function (changedRow, group) {
  682.  682   var rowSettings = this.rowSettings(group, changedRow);
  683.  683 
  684.  684   // Set the row as its own target.
  685.  685   if (rowSettings.relationship == 'self' || rowSettings.relationship == 'group') {
  686.  686     var sourceRow = changedRow;
  687.  687   }
  688.  688   // Siblings are easy, check previous and next rows.
  689.  689   else if (rowSettings.relationship == 'sibling') {
  690.  690     var previousRow = $(changedRow).prev('tr').get(0);
  691.  691     var nextRow = $(changedRow).next('tr').get(0);
  692.  692     var sourceRow = changedRow;
  693.  693     if ($(previousRow).is('.draggable') && $('.' + group, previousRow).length) {
  694.  694       if (this.indentEnabled) {
  695.  695         if ($('.indentations', previousRow).length == $('.indentations', changedRow)) {
  696.  696           sourceRow = previousRow;
  697.  697         }
  698.  698       }
  699.  699       else {
  700.  700         sourceRow = previousRow;
  701.  701       }
  702.  702     }
  703.  703     else if ($(nextRow).is('.draggable') && $('.' + group, nextRow).length) {
  704.  704       if (this.indentEnabled) {
  705.  705         if ($('.indentations', nextRow).length == $('.indentations', changedRow)) {
  706.  706           sourceRow = nextRow;
  707.  707         }
  708.  708       }
  709.  709       else {
  710.  710         sourceRow = nextRow;
  711.  711       }
  712.  712     }
  713.  713   }
  714.  714   // Parents, look up the tree until we find a field not in this group.
  715.  715   // Go up as many parents as indentations in the changed row.
  716.  716   else if (rowSettings.relationship == 'parent') {
  717.  717     var previousRow = $(changedRow).prev('tr');
  718.  718     while (previousRow.length && $('.indentation', previousRow).length >= this.rowObject.indents) {
  719.  719       previousRow = previousRow.prev('tr');
