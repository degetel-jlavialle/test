  329. 329   // handler. In this circumstance, options.extraData is never used. For forms
  330. 330   // with file inputs, the jQuery Form plugin uses the browser's normal form
  331. 331   // submission mechanism, but captures the response in a hidden IFRAME. In this
  332. 332   // circumstance, it calls this handler first, and then appends hidden fields
  333. 333   // to the form to submit the values in options.extraData. There is no simple
  334. 334   // way to know which submission mechanism will be used, so we add to extraData
  335. 335   // regardless, and allow it to be ignored in the former case.

  340. 340     // server can use this information to wrap the JSON response in a TEXTAREA,
  341. 341     // as per http://jquery.malsup.com/form/#file-upload.
  342. 342     options.extraData.ajax_iframe_upload = '1';
  343. 343 
  344. 344     // The triggering element is about to be disabled (see below), but if it
  345. 345     // contains a value (e.g., a checkbox, textfield, select, etc.), ensure that
  346. 346     // value is included in the submission. As per above, submissions that use
  347. 347     // $.ajax() are already serialized prior to the element being disabled, so
  348. 348     // this is only needed for IFRAME submissions.
  349. 349     var v = $.fieldValue(this.element);
  350. 350     if (v !== null) {

  352. 352     }
  353. 353   }

  355. 355   // Disable the element that received the change to prevent user interface
  356. 356   // interaction while the Ajax request is in progress. ajax.ajaxing prevents
  357. 357   // the element from triggering a new request, but does not prevent the user
  358. 358   // from changing its value.
  359. 359   $(this.element).addClass('progress-disabled').attr('disabled', true);

  365. 365       progressBar.setProgress(-1, this.progress.message);
  366. 366     }
  367. 367     if (this.progress.url) {
  368. 368       progressBar.startMonitoring(this.progress.url, this.progress.interval || 1500);
  369. 369     }

  473. 473 Drupal.ajax.prototype.commands = {
  474. 474   /**
  475. 475    * Command to insert new content into the DOM.
  476. 476    */
  477. 477   insert: function (ajax, response, status) {
  478. 478     // Get information from the response. If it is not there, default to
  479. 479     // our presets.
  480. 480     var wrapper = response.selector ? $(response.selector) : $(ajax.wrapper);
  481. 481     var method = response.method || ajax.method;
  482. 482     var effect = ajax.getEffect(response);
  483. 483 
  484. 484     // We don't know what response.data contains: it might be a string of text
  485. 485     // without HTML, so don't rely on jQuery correctly iterpreting
  486. 486     // $(response.data) as new HTML rather than a CSS selector. Also, if
  487. 487     // response.data contains top-level text nodes, they get lost with either
  488. 488     // $(response.data) or $('<div></div>').replaceWith(response.data).
  489. 489     var new_content_wrapped = $('<div></div>').html(response.data);
  490. 490     var new_content = new_content_wrapped.contents();
  491. 491 
  492. 492     // For legacy reasons, the effects processing code assumes that new_content
  493. 493     // consists of a single top-level element. Also, it has not been
  494. 494     // sufficiently tested whether attachBehaviors() can be successfully called
  495. 495     // with a context object that includes top-level text nodes. However, to
  496. 496     // give developers full control of the HTML appearing in the page, and to
  497. 497     // enable Ajax content to be inserted in places where DIV elements are not
  498. 498     // allowed (e.g., within TABLE, TR, and SPAN parents), we check if the new
  499. 499     // content satisfies the requirement of a single top-level element, and
  500. 500     // only use the container DIV created above when it doesn't. For more
  501. 501     // information, please see http://drupal.org/node/736066.
  502. 502     if (new_content.length != 1 || new_content.get(0).nodeType != 1) {
  503. 503       new_content = new_content_wrapped;
  504. 504     }
  505. 505 
  506. 506     // If removing content from the wrapper, detach behaviors first.
  507. 507     switch (method) {
  508. 508       case 'html':
  509. 509       case 'replaceWith':
  510. 510       case 'replaceAll':
  511. 511       case 'empty':
  512. 512       case 'remove':
  513. 513         var settings = response.settings || ajax.settings || Drupal.settings;
  514. 514         Drupal.detachBehaviors(wrapper, settings);
  515. 515     }
  516. 516 
  517. 517     // Add the new content to the page.
  518. 518     wrapper[method](new_content);
  519. 519 
  520. 520     // Immediately hide the new content if we're using any effects.
  521. 521     if (effect.showEffect != 'show') {
  522. 522       new_content.hide();
  523. 523     }
  524. 524 
  525. 525     // Determine which effect to use and what content will receive the
  526. 526     // effect, then show the new content.
  527. 527     if ($('.ajax-new-content', new_content).length > 0) {
  528. 528       $('.ajax-new-content', new_content).hide();
  529. 529       new_content.show();
  530. 530       $('.ajax-new-content', new_content)[effect.showEffect](effect.showSpeed);
  531. 531     }
  532. 532     else if (effect.showEffect != 'show') {
  533. 533       new_content[effect.showEffect](effect.showSpeed);
  534. 534     }
  535. 535 
  536. 536     // Attach all JavaScript behaviors to the new content, if it was successfully
  537. 537     // added to the page, this if statement allows #ajax['wrapper'] to be
  538. 538     // optional.
  539. 539     if (new_content.parents('html').length > 0) {
  540. 540       // Apply any settings from the returned JSON if available.
  541. 541       var settings = response.settings || ajax.settings || Drupal.settings;
  542. 542       Drupal.attachBehaviors(new_content, settings);
  543. 543     }
