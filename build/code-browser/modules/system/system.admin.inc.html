  657.  657 /**
  658.  658  * Helper function for the system_theme_settings form.
  659.  659  *
  660.  660  * Attempts to validate normal system paths, paths relative to the public files
  661.  661  * directory, or stream wrapper URIs. If the given path is any of the above,
  662.  662  * returns a valid path or URI that the theme system can display.
  663.  663  *
  664.  664  * @param $path
  665.  665  *   A path relative to the Drupal root or to the public files directory, or
  666.  666  *   a stream wrapper URI.
  667.  667  * @return mixed
  668.  668  *   A valid path that can be displayed through the theme system, or FALSE if
  669.  669  *   the path could not be validated.
  670.  670  */
  671.  671 function _system_theme_settings_validate_path($path) {
  672.  672   // Absolute local file paths are invalid.
  673.  673   if (drupal_realpath($path) == $path) {
  674.  674     return FALSE;
  675.  675   }
  676.  676   // A path relative to the Drupal root or a fully qualified URI is valid.
  677.  677   if (is_file($path)) {
  678.  678     return $path;
  679.  679   }
  680.  680   // Prepend 'public://' for relative file paths within public filesystem.
  681.  681   if (file_uri_scheme($path) === FALSE) {
  682.  682     $path = 'public://' . $path;
  683.  683   }
  684.  684   if (is_file($path)) {
  685.  685     return $path;
  686.  686   }
  687.  687   return FALSE;
  688.  688 }
