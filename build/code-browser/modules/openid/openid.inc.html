  139. 139 function _openid_xrds_parse($raw_xml) {
  140. 140   $services = array();
  141. 141 
  142. 142   // For PHP version >= 5.2.11, we can use this function to protect against
  143. 143   // malicious doctype declarations and other unexpected entity loading.
  144. 144   // However, we will not rely on it, and reject any XML with a DOCTYPE.
  145. 145   $disable_entity_loader = function_exists('libxml_disable_entity_loader');
  146. 146   if ($disable_entity_loader) {
  147. 147     $load_entities = libxml_disable_entity_loader(TRUE);
  148. 148   }
  149. 149 
  150. 150   // Load the XML into a DOM document.
  151. 151   $dom = new DOMDocument();
  152. 152   @$dom->loadXML($raw_xml);
  153. 153 
  154. 154   // Since DOCTYPE declarations from an untrusted source could be malicious, we
  155. 155   // stop parsing here and treat the XML as invalid since XRDS documents do not
  156. 156   // require, and are not expected to have, a DOCTYPE.
  157. 157   if (isset($dom->doctype)) {
  158. 158     return array();
  159. 159   }
  160. 160 
  161. 161   // Also stop parsing if there is an unreasonably large number of tags.
  162. 162   if ($dom->getElementsByTagName('*')->length > variable_get('openid_xrds_maximum_tag_count', 30000)) {
  163. 163     return array();
  164. 164   }
  165. 165 
  166. 166   // Parse the DOM document for the information we need.
  167. 167   if ($xml = simplexml_import_dom($dom)) {
  168. 168     foreach ($xml->children(OPENID_NS_XRD)->XRD as $xrd) {
  169. 169       foreach ($xrd->children(OPENID_NS_XRD)->Service as $service_element) {
  170. 170         $service = array(
  171. 171           'priority' => $service_element->attributes()->priority ? (int)$service_element->attributes()->priority : PHP_INT_MAX,
  172. 172           'types' => array(),
  173. 173           'uri' => (string)$service_element->children(OPENID_NS_XRD)->URI,
  174. 174           'service' => $service_element,
  175. 175           'xrd' => $xrd,
  176. 176         );
  177. 177         foreach ($service_element->Type as $type) {
  178. 178           $service['types'][] = (string)$type;
  179. 179         }
  180. 180         if ($service_element->children(OPENID_NS_XRD)->LocalID) {
  181. 181           $service['identity'] = (string)$service_element->children(OPENID_NS_XRD)->LocalID;
  182. 182         }
  183. 183         elseif ($service_element->children(OPENID_NS_OPENID)->Delegate) {
  184. 184           $service['identity'] = (string)$service_element->children(OPENID_NS_OPENID)->Delegate;
  185. 185         }
  186. 186         else {
  187. 187           $service['identity'] = FALSE;
  188. 188         }
  189. 189         $services[] = $service;
  190. 190       }
  191. 191     }
  192. 192   }
  193. 193 
  194. 194   // Return the LIBXML options to the previous state before returning.
  195. 195   if ($disable_entity_loader) {

  200. 200 }

  202. 202 /**
  203. 203  * Select a service element.
  204. 204  *
  205. 205  * The procedure is described in OpenID Authentication 2.0, section 7.3.2.
  206. 206  *
  207. 207  * A new entry is added to the returned array with the key 'version' and the
  208. 208  * value 1 or 2 specifying the protocol version used by the service.
  209. 209  *
  210. 210  * @param $services
  211. 211  *   An array of service arrays as returned by openid_discovery().
  212. 212  * @return
  213. 213  *   The selected service array, or NULL if no valid services were found.
  214. 214  */
  215. 215 function _openid_select_service(array $services) {
  216. 216   // Extensible Resource Identifier (XRI) Resolution Version 2.0, section 4.3.3:
  217. 217   // Find the service with the highest priority (lowest integer value). If there
  218. 218   // is a tie, select a random one, not just the first in the XML document.
  219. 219   shuffle($services);
  220. 220   $selected_service = NULL;
  221. 221   $selected_type_priority = FALSE;
  222. 222 
  223. 223   // Search for an OP Identifier Element.
  224. 224   foreach ($services as $service) {
  225. 225     if (!empty($service['uri'])) {
  226. 226       $type_priority = FALSE;
  227. 227       if (in_array('http://specs.openid.net/auth/2.0/server', $service['types'])) {
  228. 228         $service['version'] = 2;
  229. 229         $type_priority = 1;
  230. 230       }
  231. 231       elseif (in_array('http://specs.openid.net/auth/2.0/signon', $service['types'])) {
  232. 232         $service['version'] = 2;
  233. 233         $type_priority = 2;
  234. 234       }
  235. 235       elseif (in_array(OPENID_NS_1_0, $service['types']) || in_array(OPENID_NS_1_1, $service['types'])) {
  236. 236         $service['version'] = 1;
  237. 237         $type_priority = 3;
  238. 238       }
  239. 239 
  240. 240       if ($type_priority
  241. 241           && (!$selected_service
  242. 242               || $type_priority < $selected_type_priority
  243. 243               || ($type_priority == $selected_type_priority && $service['priority'] < $selected_service['priority']))) {
  244. 244         $selected_service = $service;
  245. 245         $selected_type_priority = $type_priority;
  246. 246       }
  247. 247     }
  248. 248   }
  249. 249 
  250. 250   if ($selected_service) {
  251. 251     // Unset SimpleXMLElement instances that cannot be saved in $_SESSION.
  252. 252     unset($selected_service['xrd']);
  253. 253     unset($selected_service['service']);
  254. 254   }
  255. 255 
  256. 256   return $selected_service;
  257. 257 }

  608. 608 /**
  609. 609  * Extract all the parameters belonging to an extension in a response message.
  610. 610  *
  611. 611  * OpenID 2.0 defines a simple extension mechanism, based on a namespace prefix.
  612. 612  *
  613. 613  * Each request or response can define a prefix using:
  614. 614  * @code
  615. 615  *   openid.ns.[prefix] = [extension_namespace]
  616. 616  *   openid.[prefix].[key1] = [value1]
  617. 617  *   openid.[prefix].[key2] = [value2]
  618. 618  *   ...
  619. 619  * @endcode
  620. 620  *
  621. 621  * This function extracts all the keys belonging to an extension namespace in a
  622. 622  * response, optionally using a fallback prefix if none is provided in the response.
  623. 623  *
  624. 624  * Note that you cannot assume that a given extension namespace will use the same
  625. 625  * prefix on the response and the request: each party may use a different prefix
  626. 626  * to refer to the same namespace.
  627. 627  *
  628. 628  * @param $response
  629. 629  *   The response array.
  630. 630  * @param $extension_namespace
  631. 631  *   The namespace of the extension.
  632. 632  * @param $fallback_prefix
  633. 633  *   An optional prefix that will be used in case no prefix is found for the
  634. 634  *   target extension namespace.
  635. 635  * @param $only_signed
  636. 636  *   Return only keys that are included in the message signature in openid.sig.
  637. 637  *   Unsigned fields may have been modified or added by other parties than the
  638. 638  *   OpenID Provider.
  639. 639  *
  640. 640  * @return
  641. 641  *   An associative array containing all the parameters in the response message
  642. 642  *   that belong to the extension. The keys are stripped from their namespace
  643. 643  *   prefix.
  644. 644  *
  645. 645  * @see http://openid.net/specs/openid-authentication-2_0.html#extensions
  646. 646  */
  647. 647 function openid_extract_namespace($response, $extension_namespace, $fallback_prefix = NULL, $only_signed = FALSE) {
  648. 648   $signed_keys = explode(',', $response['openid.signed']);
  649. 649

  651. 651   $prefix = $fallback_prefix;

  653. 653     if ($value == $extension_namespace && preg_match('/^openid\.ns\.([^.]+)$/', $key, $matches)) {
  654. 654       $prefix = $matches[1];

  656. 656         // The namespace was defined but was not signed as required. In this
  657. 657         // case we do not fall back to $fallback_prefix.
  658. 658         $prefix = NULL;

  663. 663

  667. 667     return $output;
  668. 668   }
  669. 669   foreach ($response as $key => $value) {

  671. 671       $local_key = $matches[1];

  673. 673         $output[$local_key] = $value;
  674. 674       }
  675. 675     }
  676. 676   }
  677. 677 
  678. 678   return $output;
  679. 679 }
