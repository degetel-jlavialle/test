  256. 256 function node_type_form_validate($form, &$form_state) {

  260. 260 
  261. 261   // Work out what the type was before the user submitted this form
  262. 262   $old_type = $form_state['values']['old_type'];
  263. 263 
  264. 264   $types = node_type_get_names();
  265. 265 
  266. 266   if (!$form_state['values']['locked']) {
  267. 267     // 'theme' conflicts with theme_node_form().

  269. 269     if (in_array($type->type, array('0', 'theme'))) {

  271. 271     }
  272. 272   }
  273. 273 
  274. 274   $names = array_flip($types);
  275. 275 
  276. 276   if (isset($names[$type->name]) && $names[$type->name] != $old_type) {
  277. 277     form_set_error('name', t('The human-readable name %name is already taken.', array('%name' => $type->name)));
  278. 278   }

  295. 295

  298. 298   $type->title_label = $form_state['values']['title_label'];

  300. 300   // module alters the title field.
  301. 301   $type->has_title = ($type->title_label != '');
  302. 302 
  303. 303   $type->base = !empty($form_state['values']['base']) ? $form_state['values']['base'] : 'node_content';
  304. 304   $type->custom = $form_state['values']['custom'];
  305. 305   $type->modified = TRUE;
  306. 306   $type->locked = $form_state['values']['locked'];
  307. 307   if (isset($form['#node_type']->module)) {
  308. 308     $type->module = $form['#node_type']->module;
  309. 309   }
  310. 310 
  311. 311   if ($op == t('Delete content type')) {
  312. 312     $form_state['redirect'] = 'admin/structure/types/manage/' . str_replace('_', '-', $type->old_type) . '/delete';
  313. 313     return;
  314. 314   }
  315. 315 
  316. 316   $variables = $form_state['values'];
  317. 317 
  318. 318   // Remove everything that's been saved already - whatever's left is assumed
  319. 319   // to be a persistent variable.
  320. 320   foreach ($variables as $key => $value) {
  321. 321     if (isset($type->$key)) {
  322. 322       unset($variables[$key]);
  323. 323     }
  324. 324   }
  325. 325 
  326. 326   unset($variables['form_token'], $variables['op'], $variables['submit'], $variables['delete'], $variables['reset'], $variables['form_id'], $variables['form_build_id']);
  327. 327 
  328. 328   // Save or reset persistent variable values.
  329. 329   foreach ($variables as $key => $value) {
  330. 330     $variable_new = $key . '_' . $type->type;
  331. 331     $variable_old = $key . '_' . $type->old_type;
  332. 332 
  333. 333     if (is_array($value)) {
  334. 334       $value = array_keys(array_filter($value));
  335. 335     }
  336. 336     variable_set($variable_new, $value);
  337. 337 
  338. 338     if ($variable_new != $variable_old) {
  339. 339       variable_del($variable_old);
  340. 340     }
  341. 341   }
  342. 342 
  343. 343   // Saving the content type after saving the variables allows modules to act
  344. 344   // on those variables via hook_node_type_insert().
  345. 345   $status = node_type_save($type);
  346. 346 
  347. 347   node_types_rebuild();
  348. 348   menu_rebuild();
  349. 349   $t_args = array('%name' => $type->name);
  350. 350 
  351. 351   if ($status == SAVED_UPDATED) {
  352. 352     drupal_set_message(t('The content type %name has been updated.', $t_args));
  353. 353   }
  354. 354   elseif ($status == SAVED_NEW) {
  355. 355     node_add_body_field($type);
  356. 356     drupal_set_message(t('The content type %name has been added.', $t_args));
  357. 357     watchdog('node', 'Added content type %name.', $t_args, WATCHDOG_NOTICE, l(t('view'), 'admin/structure/types'));
  358. 358   }
  359. 359 
  360. 360   $form_state['redirect'] = 'admin/structure/types';
  361. 361   return;
  362. 362 }
