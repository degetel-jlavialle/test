  66.  66   validateExtension: function (event) {
  67.  67     // Remove any previous errors.
  68.  68     $('.file-upload-js-error').remove();
  69.  69 
  70.  70     // Add client side validation for the input[type=file].
  71.  71     var extensionPattern = event.data.extensions.replace(/,\s*/g, '|');
  72.  72     if (extensionPattern.length > 1 && this.value.length > 0) {
  73.  73       var acceptableMatch = new RegExp('\\.(' + extensionPattern + ')$', 'gi');
  74.  74       if (!acceptableMatch.test(this.value)) {
  75.  75         var error = Drupal.t("The selected file %filename cannot be uploaded. Only files with the following extensions are allowed: %extensions.", {
  76.  76           // According to the specifications of HTML5, a file upload control
  77.  77           // should not reveal the real local path to the file that a user
  78.  78           // has selected. Some web browsers implement this restriction by
  79.  79           // replacing the local path with "C:\fakepath\", which can cause
  80.  80           // confusion by leaving the user thinking perhaps Drupal could not
  81.  81           // find the file because it messed up the file path. To avoid this
  82.  82           // confusion, therefore, we strip out the bogus fakepath string.
  83.  83           '%filename': this.value.replace('C:\\fakepath\\', ''),
  84.  84           '%extensions': extensionPattern.replace(/\|/g, ', ')
  85.  85         });
  86.  86         $(this).closest('div.form-managed-file').prepend('<div class="messages error file-upload-js-error" aria-live="polite">' + error + '</div>');
  87.  87         this.value = '';
  88.  88         return false;
  89.  89       }
  90.  90     }

  106. 106       $enabledFields = $(this).closest('div.form-managed-file').find('input.form-file');
  107. 107     }
  108. 108 
  109. 109     // Temporarily disable upload fields other than the one we're currently
  110. 110     // working with. Filter out fields that are already disabled so that they
  111. 111     // do not get enabled when we re-enable these fields at the end of behavior
  112. 112     // processing. Re-enable in a setTimeout set to a relatively short amount
  113. 113     // of time (1 second). All the other mousedown handlers (like Drupal's Ajax

  118. 118     $fieldsToTemporarilyDisable.attr('disabled', 'disabled');
  119. 119     setTimeout(function (){
  120. 120       $fieldsToTemporarilyDisable.attr('disabled', false);
  121. 121     }, 1000);
  122. 122   },
