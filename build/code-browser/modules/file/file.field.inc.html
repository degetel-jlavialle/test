  133.  133 /**

  135.  135  *
  136.  136  * This doubles as a convenience clean-up function and a validation routine.
  137.  137  * Commas are allowed by the end-user, but ultimately the value will be stored

  140.  140 function _file_generic_settings_extensions($element, &$form_state) {
  141.  141   if (!empty($element['#value'])) {
  142.  142     $extensions = preg_replace('/([, ]+\.?)/', ' ', trim(strtolower($element['#value'])));

  146.  146       form_error($element, t('The list of allowed extensions is not valid, be sure to exclude leading dots and to separate extensions with a comma or space.'));

  148.  148     else {
  149.  149       form_set_value($element, $extensions, $form_state);
  150.  150     }

  169.  169 function file_field_load($entity_type, $entities, $field, $instances, $langcode, &$items, $age) {
  170.  170 
  171.  171   $fids = array();
  172.  172   foreach ($entities as $id => $entity) {
  173.  173     // Load the files from the files table.
  174.  174     foreach ($items[$id] as $delta => $item) {
  175.  175       if (!empty($item['fid'])) {
  176.  176         $fids[] = $item['fid'];
  177.  177       }
  178.  178     }
  179.  179   }
  180.  180   $files = file_load_multiple($fids);
  181.  181 
  182.  182   foreach ($entities as $id => $entity) {
  183.  183     foreach ($items[$id] as $delta => $item) {
  184.  184       // If the file does not exist, mark the entire item as empty.
  185.  185       if (empty($item['fid']) || !isset($files[$item['fid']])) {
  186.  186         $items[$id][$delta] = NULL;
  187.  187       }
  188.  188       else {
  189.  189         $items[$id][$delta] = array_merge((array) $files[$item['fid']], $item);
  190.  190       }
  191.  191     }
  192.  192   }
  193.  193 }

  214.  214 function file_field_presave($entity_type, $entity, $field, $instance, $langcode, &$items) {
  215.  215   // Make sure that each file which will be saved with this object has a
  216.  216   // permanent status, so that it will not be removed when temporary files are
  217.  217   // cleaned up.
  218.  218   foreach ($items as $delta => $item) {
  219.  219     if (empty($item['fid'])) {
  220.  220       unset($items[$delta]);
  221.  221       continue;
  222.  222     }
  223.  223     $file = file_load($item['fid']);
  224.  224     if (empty($file)) {
  225.  225       unset($items[$delta]);
  226.  226       continue;
  227.  227     }
  228.  228     if (!$file->status) {
  229.  229       $file->status = FILE_STATUS_PERMANENT;
  230.  230       file_save($file);
  231.  231     }
  232.  232   }
  233.  233 }

  248.  248 /**
  249.  249  * Implements hook_field_update().
  250.  250  *
  251.  251  * Checks for files that have been removed from the object.
  252.  252  */
  253.  253 function file_field_update($entity_type, $entity, $field, $instance, $langcode, &$items) {
  254.  254   list($id, $vid, $bundle) = entity_extract_ids($entity_type, $entity);
  255.  255 
  256.  256   // On new revisions, all files are considered to be a new usage and no
  257.  257   // deletion of previous file usages are necessary.
  258.  258   if (!empty($entity->revision)) {
  259.  259     foreach ($items as $item) {
  260.  260       $file = (object) $item;
  261.  261       file_usage_add($file, 'file', $entity_type, $id);
  262.  262     }
  263.  263     return;
  264.  264   }
  265.  265 
  266.  266   // Build a display of the current FIDs.
  267.  267   $current_fids = array();
  268.  268   foreach ($items as $item) {
  269.  269     $current_fids[] = $item['fid'];
  270.  270   }
  271.  271 
  272.  272   // Compare the original field values with the ones that are being saved. Use
  273.  273   // $entity->original to check this when possible, but if it isn't available,
  274.  274   // create a bare-bones entity and load its previous values instead.
  275.  275   if (isset($entity->original)) {
  276.  276     $original = $entity->original;
  277.  277   }
  278.  278   else {
  279.  279     $original = entity_create_stub_entity($entity_type, array($id, $vid, $bundle));
  280.  280     field_attach_load($entity_type, array($id => $original), FIELD_LOAD_CURRENT, array('field_id' => $field['id']));
  281.  281   }
  282.  282   $original_fids = array();
  283.  283   if (!empty($original->{$field['field_name']}[$langcode])) {
  284.  284     foreach ($original->{$field['field_name']}[$langcode] as $original_item) {
  285.  285       $original_fids[] = $original_item['fid'];
  286.  286       if (isset($original_item['fid']) && !in_array($original_item['fid'], $current_fids)) {
  287.  287         // Decrement the file usage count by 1 and delete the file if possible.
  288.  288         file_field_delete_file($original_item, $field, $entity_type, $id);
  289.  289       }
  290.  290     }
  291.  291   }
  292.  292 
  293.  293   // Add new usage entries for newly added files.
  294.  294   foreach ($items as $item) {
  295.  295     if (!in_array($item['fid'], $original_fids)) {
  296.  296       $file = (object) $item;
  297.  297       file_usage_add($file, 'file', $entity_type, $id);
  298.  298     }
  299.  299   }
  300.  300 }

  317.  317 function file_field_delete_revision($entity_type, $entity, $field, $instance, $langcode, &$items) {
  318.  318   list($id, $vid, $bundle) = entity_extract_ids($entity_type, $entity);
  319.  319   foreach ($items as $delta => $item) {

  327.  327 /**
  328.  328  * Decrements the usage count for a file and attempts to delete it.
  329.  329  *
  330.  330  * This function only has an effect if the file being deleted is used only by
  331.  331  * File module.
  332.  332  *
  333.  333  * @param $item
  334.  334  *   The field item that contains a file array.
  335.  335  * @param $field
  336.  336  *   The field structure for the operation.
  337.  337  * @param $entity_type
  338.  338  *   The type of $entity.
  339.  339  * @param $id
  340.  340  *   The entity ID which contains the file being deleted.
  341.  341  * @param $count
  342.  342  *   (optional) The number of references to decrement from the object
  343.  343  *   containing the file. Defaults to 1.
  344.  344  *
  345.  345  * @return
  346.  346  *   Boolean TRUE if the file was deleted, or an array of remaining references
  347.  347  *   if the file is still in use by other modules. Boolean FALSE if an error
  348.  348  *   was encountered.
  349.  349  */
  350.  350 function file_field_delete_file($item, $field, $entity_type, $id, $count = 1) {
  351.  351   // To prevent the file field from deleting files it doesn't know about, check
  352.  352   // the file reference count. Temporary files can be deleted because they
  353.  353   // are not yet associated with any content at all.
  354.  354   $file = (object) $item;
  355.  355   $file_usage = file_usage_list($file);
  356.  356   if ($file->status == 0 || !empty($file_usage['file'])) {
  357.  357     file_usage_delete($file, 'file', $entity_type, $id, $count);
