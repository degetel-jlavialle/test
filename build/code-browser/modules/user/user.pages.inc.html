  93.  93 function user_pass_reset($form, &$form_state, $uid, $timestamp, $hashed_pass, $action = NULL) {
  94.  94   global $user;
  95.  95 
  96.  96   // When processing the one-time login link, we have to make sure that a user
  97.  97   // isn't already logged in.
  98.  98   if ($user->uid) {
  99.  99     // The existing user is already logged in.
  100. 100     if ($user->uid == $uid) {
  101. 101       drupal_set_message(t('You are logged in as %user. <a href="!user_edit">Change your password.</a>', array('%user' => $user->name, '!user_edit' => url("user/$user->uid/edit"))));
  102. 102     }
  103. 103     // A different user is already logged in on the computer.
  104. 104     else {
  105. 105       $reset_link_account = user_load($uid);
  106. 106       if (!empty($reset_link_account)) {
  107. 107         drupal_set_message(t('Another user (%other_user) is already logged into the site on this computer, but you tried to use a one-time link for user %resetting_user. Please <a href="!logout">logout</a> and try using the link again.',
  108. 108           array('%other_user' => $user->name, '%resetting_user' => $reset_link_account->name, '!logout' => url('user/logout'))));
  109. 109       } else {
  110. 110         // Invalid one-time link specifies an unknown user.
  111. 111         drupal_set_message(t('The one-time login link you clicked is invalid.'));
  112. 112       }
  113. 113     }
  114. 114     drupal_goto();
  115. 115   }
  116. 116   else {
  117. 117     // Time out, in seconds, until login URL expires. Defaults to 24 hours =
  118. 118     // 86400 seconds.
  119. 119     $timeout = variable_get('user_password_reset_timeout', 86400);
  120. 120     $current = REQUEST_TIME;
  121. 121     // Some redundant checks for extra security ?
  122. 122     $users = user_load_multiple(array($uid), array('status' => '1'));
  123. 123     if ($timestamp <= $current && $account = reset($users)) {
  124. 124       // No time out for first time login.
  125. 125       if ($account->login && $current - $timestamp > $timeout) {
  126. 126         drupal_set_message(t('You have tried to use a one-time login link that has expired. Please request a new one using the form below.'));
  127. 127         drupal_goto('user/password');
  128. 128       }
  129. 129       elseif ($account->uid && $timestamp >= $account->login && $timestamp <= $current && $hashed_pass == user_pass_rehash($account->pass, $timestamp, $account->login)) {
  130. 130         // First stage is a confirmation form, then login
  131. 131         if ($action == 'login') {
  132. 132           // Set the new user.
  133. 133           $user = $account;
  134. 134           // user_login_finalize() also updates the login timestamp of the
  135. 135           // user, which invalidates further use of the one-time login link.
  136. 136           user_login_finalize();
  137. 137           watchdog('user', 'User %name used one-time login link at time %timestamp.', array('%name' => $account->name, '%timestamp' => $timestamp));
  138. 138           drupal_set_message(t('You have just used your one-time login link. It is no longer necessary to use this link to log in. Please change your password.'));
  139. 139           // Let the user's password be changed without the current password check.
  140. 140           $token = drupal_random_key();
  141. 141           $_SESSION['pass_reset_' . $user->uid] = $token;
  142. 142           drupal_goto('user/' . $user->uid . '/edit', array('query' => array('pass-reset-token' => $token)));
  143. 143         }
  144. 144         else {
  145. 145           $form['message'] = array('#markup' => t('<p>This is a one-time login for %user_name and will expire on %expiration_date.</p><p>Click on this button to log in to the site and change your password.</p>', array('%user_name' => $account->name, '%expiration_date' => format_date($timestamp + $timeout))));
  146. 146           $form['help'] = array('#markup' => '<p>' . t('This login can be used only once.') . '</p>');
  147. 147           $form['actions'] = array('#type' => 'actions');
  148. 148           $form['actions']['submit'] = array('#type' => 'submit', '#value' => t('Log in'));
  149. 149           $form['#action'] = url("user/reset/$uid/$timestamp/$hashed_pass/login");
  150. 150           return $form;
  151. 151         }
  152. 152       }
  153. 153       else {
  154. 154         drupal_set_message(t('You have tried to use a one-time login link that has either been used or is no longer valid. Please request a new one using the form below.'));
  155. 155         drupal_goto('user/password');
  156. 156       }
  157. 157     }
  158. 158     else {
  159. 159       // Deny access, no more clues.
  160. 160       // Everything will be in the watchdog's URL for the administrator to check.
  161. 161       drupal_access_denied();
  162. 162       drupal_exit();
  163. 163     }
  164. 164   }
  165. 165 }

  306. 306 function user_profile_form_submit($form, &$form_state) {

  308. 308   $category = $form['#user_category'];
  309. 309   // Remove unneeded values.
  310. 310   form_state_values_clean($form_state);
  311. 311 
  312. 312   // Before updating the account entity, keep an unchanged copy for use with
  313. 313   // user_save() later. This is necessary for modules implementing the user
  314. 314   // hooks to be able to react on changes by comparing the values of $account
  315. 315   // and $edit.
  316. 316   $account_unchanged = clone $account;
  317. 317 
  318. 318   entity_form_submit_build_entity('user', $account, $form, $form_state);
  319. 319 
  320. 320   // Populate $edit with the properties of $account, which have been edited on
  321. 321   // this form by taking over all values, which appear in the form values too.
  322. 322   $edit = array_intersect_key((array) $account, $form_state['values']);
  323. 323 
  324. 324   user_save($account_unchanged, $edit, $category);
  325. 325   $form_state['values']['uid'] = $account->uid;
  326. 326 
  327. 327   if ($category == 'account' && !empty($edit['pass'])) {
  328. 328     // Remove the password reset tag since a new password was saved.
  329. 329     unset($_SESSION['pass_reset_'. $account->uid]);
  330. 330   }
  331. 331   // Clear the page cache because pages can contain usernames and/or profile information:
  332. 332   cache_clear_all();
  333. 333 
  334. 334   drupal_set_message(t('The changes have been saved.'));
  335. 335 }
