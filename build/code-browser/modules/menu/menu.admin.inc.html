  152. 152 /**
  153. 153  * Submit handler for the menu overview form.
  154. 154  *
  155. 155  * This function takes great care in saving parent items first, then items
  156. 156  * underneath them. Saving items in the incorrect order can break the menu tree.
  157. 157  *
  158. 158  * @see menu_overview_form()
  159. 159  */
  160. 160 function menu_overview_form_submit($form, &$form_state) {
  161. 161   // When dealing with saving menu items, the order in which these items are
  162. 162   // saved is critical. If a changed child item is saved before its parent,
  163. 163   // the child item could be saved with an invalid path past its immediate
  164. 164   // parent. To prevent this, save items in the form in the same order they
  165. 165   // are sent by $_POST, ensuring parents are saved first, then their children.
  166. 166   // See http://drupal.org/node/181126#comment-632270
  167. 167   $order = array_flip(array_keys($form_state['input'])); // Get the $_POST order.
  168. 168   $form = array_merge($order, $form); // Update our original form with the new order.
  169. 169 
  170. 170   $updated_items = array();
  171. 171   $fields = array('weight', 'plid');
  172. 172   foreach (element_children($form) as $mlid) {
  173. 173     if (isset($form[$mlid]['#item'])) {
  174. 174       $element = $form[$mlid];
  175. 175       // Update any fields that have changed in this menu item.
  176. 176       foreach ($fields as $field) {
  177. 177         if ($element[$field]['#value'] != $element[$field]['#default_value']) {
  178. 178           $element['#item'][$field] = $element[$field]['#value'];
  179. 179           $updated_items[$mlid] = $element['#item'];
  180. 180         }
  181. 181       }
  182. 182       // Hidden is a special case, the value needs to be reversed.
  183. 183       if ($element['hidden']['#value'] != $element['hidden']['#default_value']) {
  184. 184         // Convert to integer rather than boolean due to PDO cast to string.
  185. 185         $element['#item']['hidden'] = $element['hidden']['#value'] ? 0 : 1;
  186. 186         $updated_items[$mlid] = $element['#item'];
  187. 187       }
  188. 188     }
  189. 189   }
  190. 190 
  191. 191   // Save all our changed items to the database.
  192. 192   foreach ($updated_items as $item) {
  193. 193     $item['customized'] = 1;
  194. 194     menu_link_save($item);
  195. 195   }
  196. 196   drupal_set_message(t('Your configuration has been saved.'));
  197. 197 }
