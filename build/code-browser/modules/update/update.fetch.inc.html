  31.  31 /**
  32.  32  * Batch callback: Processes a step in batch for fetching available update data.
  33.  33  *
  34.  34  * @param $context
  35.  35  *   Reference to an array used for Batch API storage.
  36.  36  */
  37.  37 function update_fetch_data_batch(&$context) {
  38.  38   $queue = DrupalQueue::get('update_fetch_tasks');
  39.  39   if (empty($context['sandbox']['max'])) {
  40.  40     $context['finished'] = 0;
  41.  41     $context['sandbox']['max'] = $queue->numberOfItems();
  42.  42     $context['sandbox']['progress'] = 0;
  43.  43     $context['message'] = t('Checking available update data ...');
  44.  44     $context['results']['updated'] = 0;
  45.  45     $context['results']['failures'] = 0;
  46.  46     $context['results']['processed'] = 0;
  47.  47   }
  48.  48 
  49.  49   // Grab another item from the fetch queue.
  50.  50   for ($i = 0; $i < 5; $i++) {
  51.  51     if ($item = $queue->claimItem()) {
  52.  52       if (_update_process_fetch_task($item->data)) {
  53.  53         $context['results']['updated']++;
  54.  54         $context['message'] = t('Checked available update data for %title.', array('%title' => $item->data['info']['name']));
  55.  55       }
  56.  56       else {
  57.  57         $context['message'] = t('Failed to check available update data for %title.', array('%title' => $item->data['info']['name']));
  58.  58         $context['results']['failures']++;
  59.  59       }
  60.  60       $context['sandbox']['progress']++;
  61.  61       $context['results']['processed']++;
  62.  62       $context['finished'] = $context['sandbox']['progress'] / $context['sandbox']['max'];
  63.  63       $queue->deleteItem($item);
  64.  64     }
  65.  65     else {
  66.  66       // If the queue is currently empty, we're done. It's possible that
  67.  67       // another thread might have added new fetch tasks while we were
  68.  68       // processing this batch. In that case, the usual 'finished' math could
  69.  69       // get confused, since we'd end up processing more tasks that we thought
  70.  70       // we had when we started and initialized 'max' with numberOfItems(). By
  71.  71       // forcing 'finished' to be exactly 1 here, we ensure that batch
  72.  72       // processing is terminated.
  73.  73       $context['finished'] = 1;
  74.  74       return;
  75.  75     }
  76.  76   }
  77.  77 }

  117. 117 /**
  118. 118  * Processes a task to fetch available update data for a single project.
  119. 119  *
  120. 120  * Once the release history XML data is downloaded, it is parsed and saved into
  121. 121  * the {cache_update} table in an entry just for that project.
  122. 122  *
  123. 123  * @param $project
  124. 124  *   Associative array of information about the project to fetch data for.
  125. 125  *
  126. 126  * @return
  127. 127  *   TRUE if we fetched parsable XML, otherwise FALSE.
  128. 128  */
  129. 129 function _update_process_fetch_task($project) {
  130. 130   global $base_url;
  131. 131   $fail = &drupal_static(__FUNCTION__, array());
  132. 132   // This can be in the middle of a long-running batch, so REQUEST_TIME won't
  133. 133   // necessarily be valid.
  134. 134   $now = time();
  135. 135   if (empty($fail)) {
  136. 136     // If we have valid data about release history XML servers that we have
  137. 137     // failed to fetch from on previous attempts, load that from the cache.
  138. 138     if (($cache = _update_cache_get('fetch_failures')) && ($cache->expire > $now)) {
  139. 139       $fail = $cache->data;
  140. 140     }
  141. 141   }
  142. 142 
  143. 143   $max_fetch_attempts = variable_get('update_max_fetch_attempts', UPDATE_MAX_FETCH_ATTEMPTS);
  144. 144 
  145. 145   $success = FALSE;
  146. 146   $available = array();
  147. 147   $site_key = drupal_hmac_base64($base_url, drupal_get_private_key());
  148. 148   $url = _update_build_fetch_url($project, $site_key);
  149. 149   $fetch_url_base = _update_get_fetch_url_base($project);
  150. 150   $project_name = $project['name'];
  151. 151 
  152. 152   if (empty($fail[$fetch_url_base]) || $fail[$fetch_url_base] < $max_fetch_attempts) {
  153. 153     $xml = drupal_http_request($url);
  154. 154     if (!isset($xml->error) && isset($xml->data)) {
  155. 155       $data = $xml->data;
  156. 156     }
  157. 157   }
  158. 158 
  159. 159   if (!empty($data)) {
  160. 160     $available = update_parse_xml($data);
  161. 161     // @todo: Purge release data we don't need (http://drupal.org/node/238950).
  162. 162     if (!empty($available)) {
  163. 163       // Only if we fetched and parsed something sane do we return success.
  164. 164       $success = TRUE;
  165. 165     }
  166. 166   }
  167. 167   else {
  168. 168     $available['project_status'] = 'not-fetched';
  169. 169     if (empty($fail[$fetch_url_base])) {
  170. 170       $fail[$fetch_url_base] = 1;
  171. 171     }
  172. 172     else {
  173. 173       $fail[$fetch_url_base]++;
  174. 174     }
  175. 175   }
  176. 176 
  177. 177   $frequency = variable_get('update_check_frequency', 1);
  178. 178   $cid = 'available_releases::' . $project_name;
  179. 179   _update_cache_set($cid, $available, $now + (60 * 60 * 24 * $frequency));
  180. 180 
  181. 181   // Stash the $fail data back in the DB for the next 5 minutes.
  182. 182   _update_cache_set('fetch_failures', $fail, $now + (60 * 5));
  183. 183 
  184. 184   // Whether this worked or not, we did just (try to) check for updates.
  185. 185   variable_set('update_last_check', $now);
  186. 186 
  187. 187   // Now that we processed the fetch task for this project, clear out the
  188. 188   // record in {cache_update} for this task so we're willing to fetch again.
  189. 189   _update_cache_clear('fetch_task::' . $project_name);
  190. 190 
  191. 191   return $success;
  192. 192 }

  228. 228  * @param $project
  229. 229  *   Associative array of information about a project as created by
  230. 230  *   update_get_projects(), including keys such as 'name' (short name), and the

  232. 232  *
  233. 233  * @see update_get_projects()
  234. 234  * @see update_get_available()
  235. 235  * @see update_refresh()
  236. 236  * @see update_fetch_data()
  237. 237  * @see _update_process_fetch_task()
  238. 238  */
  239. 239 function _update_create_fetch_task($project) {
  240. 240   $fetch_tasks = &drupal_static(__FUNCTION__, array());
  241. 241   if (empty($fetch_tasks)) {
  242. 242     $fetch_tasks = _update_get_cache_multiple('fetch_task');
  243. 243   }
  244. 244   $cid = 'fetch_task::' . $project['name'];
  245. 245   if (empty($fetch_tasks[$cid])) {
  246. 246     $queue = DrupalQueue::get('update_fetch_tasks');
  247. 247     $queue->createItem($project);
  248. 248     // Due to race conditions, it is possible that another process already
  249. 249     // inserted a row into the {cache_update} table and the following query will
  250. 250     // throw an exception.
  251. 251     // @todo: Remove the need for the manual check by relying on a queue that
  252. 252     // enforces unique items.
  253. 253     try {
  254. 254       db_insert('cache_update')
  255. 255         ->fields(array(
  256. 256           'cid' => $cid,
  257. 257           'created' => REQUEST_TIME,
  258. 258         ))
  259. 259         ->execute();
  260. 260     }
  261. 261     catch (Exception $e) {
  262. 262       // The exception can be ignored safely.
  263. 263     }
  264. 264     $fetch_tasks[$cid] = REQUEST_TIME;
  265. 265   }
  266. 266 }

  271. 271  * This figures out the right URL to use, based on the project's .info file and

  273. 273  * configured to report usage stats.
  274. 274  *
  275. 275  * @param $project
  276. 276  *   The array of project information from update_get_projects().
  277. 277  * @param $site_key
  278. 278  *   (optional) The anonymous site key hash. Defaults to an empty string.

  282. 282  *

  287. 287 function _update_build_fetch_url($project, $site_key = '') {
  288. 288   $name = $project['name'];
  289. 289   $url = _update_get_fetch_url_base($project);
  290. 290   $url .= '/' . $name . '/' . DRUPAL_CORE_COMPATIBILITY;
  291. 291 
  292. 292   // Only append usage information if we have a site key and the project is
  293. 293   // enabled. We do not want to record usage statistics for disabled projects.
  294. 294   if (!empty($site_key) && (strpos($project['project_type'], 'disabled') === FALSE)) {
  295. 295     // Append the site key.
  296. 296     $url .= (strpos($url, '?') !== FALSE) ? '&' : '?';
  297. 297     $url .= 'site_key=';
  298. 298     $url .= rawurlencode($site_key);
  299. 299 
  300. 300     // Append the version.
  301. 301     if (!empty($project['info']['version'])) {
  302. 302       $url .= '&version=';
  303. 303       $url .= rawurlencode($project['info']['version']);
  304. 304     }
  305. 305 
  306. 306     // Append the list of modules or themes enabled.
  307. 307     $list = array_keys($project['includes']);
  308. 308     $url .= '&list=';
  309. 309     $url .= rawurlencode(implode(',', $list));
  310. 310   }
  311. 311   return $url;
  312. 312 }

  364. 364         // Track when the last mail was successfully sent to avoid sending
  365. 365         // too many e-mails.

  367. 367           variable_set('update_last_email_notification', REQUEST_TIME);
  368. 368         }
  369. 369       }
  370. 370     }
  371. 371   }

  374. 374 /**
  375. 375  * Parses the XML of the Drupal release history info files.
  376. 376  *
  377. 377  * @param $raw_xml
  378. 378  *   A raw XML string of available release data for a given project.
  379. 379  *
  380. 380  * @return
  381. 381  *   Array of parsed data about releases for a given project, or NULL if there
  382. 382  *   was an error parsing the string.
  383. 383  */
  384. 384 function update_parse_xml($raw_xml) {
  385. 385   try {
  386. 386     $xml = new SimpleXMLElement($raw_xml);
  387. 387   }
  388. 388   catch (Exception $e) {
