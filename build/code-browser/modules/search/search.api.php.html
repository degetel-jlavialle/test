  167. 167 function hook_search_execute($keys = NULL, $conditions = NULL) {
  168. 168   // Build matching conditions
  169. 169   $query = db_select('search_index', 'i', array('target' => 'slave'))->extend('SearchQuery')->extend('PagerDefault');
  170. 170   $query->join('node', 'n', 'n.nid = i.sid');
  171. 171   $query
  172. 172     ->condition('n.status', 1)
  173. 173     ->addTag('node_access')
  174. 174     ->searchExpression($keys, 'node');
  175. 175 
  176. 176   // Insert special keywords.
  177. 177   $query->setOption('type', 'n.type');
  178. 178   $query->setOption('language', 'n.language');
  179. 179   if ($query->setOption('term', 'ti.tid')) {
  180. 180     $query->join('taxonomy_index', 'ti', 'n.nid = ti.nid');
  181. 181   }
  182. 182   // Only continue if the first pass query matches.
  183. 183   if (!$query->executeFirstPass()) {
  184. 184     return array();
  185. 185   }
  186. 186 
  187. 187   // Add the ranking expressions.
  188. 188   _node_rankings($query);
  189. 189 
  190. 190   // Load results.
  191. 191   $find = $query
  192. 192     ->limit(10)
  193. 193     ->execute();
  194. 194   $results = array();
  195. 195   foreach ($find as $item) {
  196. 196     // Build the node body.
  197. 197     $node = node_load($item->sid);
  198. 198     node_build_content($node, 'search_result');
  199. 199     $node->body = drupal_render($node->content);
  200. 200 
  201. 201     // Fetch comments for snippet.
  202. 202     $node->rendered .= ' ' . module_invoke('comment', 'node_update_index', $node);
  203. 203     // Fetch terms for snippet.
  204. 204     $node->rendered .= ' ' . module_invoke('taxonomy', 'node_update_index', $node);
  205. 205 
  206. 206     $extra = module_invoke_all('node_search_result', $node);
  207. 207 
  208. 208     $results[] = array(
  209. 209       'link' => url('node/' . $item->sid, array('absolute' => TRUE)),
  210. 210       'type' => check_plain(node_type_get_name($node)),
  211. 211       'title' => $node->title,
  212. 212       'user' => theme('username', array('account' => $node)),
  213. 213       'date' => $node->changed,
  214. 214       'node' => $node,
  215. 215       'extra' => $extra,
  216. 216       'score' => $item->calculated_score,
  217. 217       'snippet' => search_excerpt($keys, $node->body),
  218. 218     );
  219. 219   }
  220. 220   return $results;
  221. 221 }

  288. 288 /**
  289. 289  * Update the search index for this module.
  290. 290  *
  291. 291  * This hook is called every cron run if search.module is enabled, your
  292. 292  * module has implemented hook_search_info(), and your module has been set as
  293. 293  * an active search module on the Search settings page
  294. 294  * (admin/config/search/settings). It allows your module to add items to the
  295. 295  * built-in search index using search_index(), or to add them to your module's
  296. 296  * own indexing mechanism.
  297. 297  *
  298. 298  * When implementing this hook, your module should index content items that
  299. 299  * were modified or added since the last run. PHP has a time limit
  300. 300  * for cron, though, so it is advisable to limit how many items you index
  301. 301  * per run using variable_get('search_cron_limit') (see example below). Also,
  302. 302  * since the cron run could time out and abort in the middle of your run, you
  303. 303  * should update your module's internal bookkeeping on when items have last
  304. 304  * been indexed as you go rather than waiting to the end of indexing.
  305. 305  *
  306. 306  * @ingroup search
  307. 307  */
  308. 308 function hook_update_index() {
  309. 309   $limit = (int)variable_get('search_cron_limit', 100);
  310. 310 
  311. 311   $result = db_query_range("SELECT n.nid FROM {node} n LEFT JOIN {search_dataset} d ON d.type = 'node' AND d.sid = n.nid WHERE d.sid IS NULL OR d.reindex <> 0 ORDER BY d.reindex ASC, n.nid ASC", 0, $limit);
  312. 312 
  313. 313   foreach ($result as $node) {
  314. 314     $node = node_load($node->nid);
  315. 315 
  316. 316     // Save the changed time of the most recent indexed node, for the search
  317. 317     // results half-life calculation.
  318. 318     variable_set('node_cron_last', $node->changed);
  319. 319 
  320. 320     // Render the node.
  321. 321     node_build_content($node, 'search_index');
  322. 322     $node->rendered = drupal_render($node->content);
  323. 323 
  324. 324     $text = '<h1>' . check_plain($node->title) . '</h1>' . $node->rendered;
  325. 325 
  326. 326     // Fetch extra data normally not visible
  327. 327     $extra = module_invoke_all('node_update_index', $node);
  328. 328     foreach ($extra as $t) {
  329. 329       $text .= $t;
  330. 330     }
  331. 331 
  332. 332     // Update index
  333. 333     search_index($node->nid, 'node', $text);
  334. 334   }
  335. 335 }
