  181. 181   /**

  183. 183    *
  184. 184    * We build two queries that match the dataset bodies.
  185. 185    */
  186. 186   protected function parseSearchExpression() {
  187. 187     // Matchs words optionally prefixed by a dash. A word in this case is
  188. 188     // something between two spaces, optionally quoted.
  189. 189     preg_match_all('/ (-?)("[^"]+"|[^" ]+)/i', ' ' .  $this->searchExpression , $keywords, PREG_SET_ORDER);
  190. 190 
  191. 191     if (count($keywords) ==  0) {
  192. 192       return;
  193. 193     }
  194. 194 
  195. 195     // Classify tokens.
  196. 196     $or = FALSE;
  197. 197     $warning = '';
  198. 198     $limit_combinations = variable_get('search_and_or_limit', 7);
  199. 199     // The first search expression does not count as AND.
  200. 200     $and_count = -1;
  201. 201     $or_count = 0;
  202. 202     foreach ($keywords as $match) {
  203. 203       if ($or_count && $and_count + $or_count >= $limit_combinations) {
  204. 204         // Ignore all further search expressions to prevent Denial-of-Service
  205. 205         // attacks using a high number of AND/OR combinations.
  206. 206         $this->expressionsIgnored = TRUE;
  207. 207         break;
  208. 208       }
  209. 209       $phrase = FALSE;
  210. 210       // Strip off phrase quotes.
  211. 211       if ($match[2]{0} == '"') {
  212. 212         $match[2] = substr($match[2], 1, -1);
  213. 213         $phrase = TRUE;
  214. 214         $this->simple = FALSE;
  215. 215       }
  216. 216       // Simplify keyword according to indexing rules and external
  217. 217       // preprocessors. Use same process as during search indexing, so it
  218. 218       // will match search index.
  219. 219       $words = search_simplify($match[2]);
  220. 220       // Re-explode in case simplification added more words, except when
  221. 221       // matching a phrase.
  222. 222       $words = $phrase ? array($words) : preg_split('/ /', $words, -1, PREG_SPLIT_NO_EMPTY);
  223. 223       // Negative matches.
  224. 224       if ($match[1] == '-') {
  225. 225         $this->keys['negative'] = array_merge($this->keys['negative'], $words);
  226. 226       }
  227. 227       // OR operator: instead of a single keyword, we store an array of all
  228. 228       // OR'd keywords.
  229. 229       elseif ($match[2] == 'OR' && count($this->keys['positive'])) {
  230. 230         $last = array_pop($this->keys['positive']);
  231. 231         // Starting a new OR?
  232. 232         if (!is_array($last)) {
  233. 233           $last = array($last);
  234. 234         }
  235. 235         $this->keys['positive'][] = $last;
  236. 236         $or = TRUE;
  237. 237         $or_count++;
  238. 238         continue;
  239. 239       }
  240. 240       // AND operator: implied, so just ignore it.
  241. 241       elseif ($match[2] == 'AND' || $match[2] == 'and') {
  242. 242         $warning = $match[2];
  243. 243         continue;
  244. 244       }
  245. 245 
  246. 246       // Plain keyword.
  247. 247       else {
  248. 248         if ($match[2] == 'or') {
  249. 249           $warning = $match[2];
  250. 250         }
  251. 251         if ($or) {
  252. 252           // Add to last element (which is an array).
  253. 253           $this->keys['positive'][count($this->keys['positive']) - 1] = array_merge($this->keys['positive'][count($this->keys['positive']) - 1], $words);
  254. 254         }
  255. 255         else {
  256. 256           $this->keys['positive'] = array_merge($this->keys['positive'], $words);
  257. 257           $and_count++;
  258. 258         }
  259. 259       }
  260. 260       $or = FALSE;
  261. 261     }
  262. 262 
  263. 263     // Convert keywords into SQL statements.
  264. 264     $this->conditions = db_and();
  265. 265     $simple_and = FALSE;
  266. 266     $simple_or = FALSE;
  267. 267     // Positive matches.
  268. 268     foreach ($this->keys['positive'] as $key) {
  269. 269       // Group of ORed terms.
  270. 270       if (is_array($key) && count($key)) {
  271. 271         $simple_or = TRUE;
  272. 272         $any = FALSE;
  273. 273         $queryor = db_or();
  274. 274         foreach ($key as $or) {
  275. 275           list($num_new_scores) = $this->parseWord($or);
  276. 276           $any |= $num_new_scores;
  277. 277           $queryor->condition('d.data', "% $or %", 'LIKE');
  278. 278         }
  279. 279         if (count($queryor)) {
  280. 280           $this->conditions->condition($queryor);
  281. 281           // A group of OR keywords only needs to match once.
  282. 282           $this->matches += ($any > 0);
  283. 283         }
  284. 284       }
  285. 285       // Single ANDed term.
  286. 286       else {
  287. 287         $simple_and = TRUE;
  288. 288         list($num_new_scores, $num_valid_words) = $this->parseWord($key);
  289. 289         $this->conditions->condition('d.data', "% $key %", 'LIKE');
  290. 290         if (!$num_valid_words) {
  291. 291           $this->simple = FALSE;
  292. 292         }
  293. 293         // Each AND keyword needs to match at least once.
  294. 294         $this->matches += $num_new_scores;
  295. 295       }
  296. 296     }
  297. 297     if ($simple_and && $simple_or) {
  298. 298       $this->simple = FALSE;
  299. 299     }
  300. 300     // Negative matches.
  301. 301     foreach ($this->keys['negative'] as $key) {
  302. 302       $this->conditions->condition('d.data', "% $key %", 'NOT LIKE');
  303. 303       $this->simple = FALSE;
  304. 304     }
  305. 305 
  306. 306     if ($warning == 'or') {
  307. 307       drupal_set_message(t('Search for either of the two terms with uppercase <strong>OR</strong>. For example, <strong>cats OR dogs</strong>.'));
  308. 308     }
  309. 309   }

  314. 314   protected function parseWord($word) {
  315. 315     $num_new_scores = 0;

  319. 319     foreach ($split as $s) {

  322. 322         if (!isset($this->words[$s])) {
  323. 323           $this->words[$s] = $s;
  324. 324           $num_new_scores++;
  325. 325         }
  326. 326         $num_valid_words++;
  327. 327       }
  328. 328     }
  329. 329     // Return matching snippet and number of added words.
  330. 330     return array($num_new_scores, $num_valid_words);
  331. 331   }

  333. 333   /**
  334. 334    * Executes the first pass query.
  335. 335    *
  336. 336    * This can either be done explicitly, so that additional scores and
  337. 337    * conditions can be applied to the second pass query, or implicitly by
  338. 338    * addScore() or execute().
  339. 339    *
  340. 340    * @return
  341. 341    *   TRUE if search items exist, FALSE if not.
  342. 342    */
  343. 343   public function executeFirstPass() {
  344. 344     $this->parseSearchExpression();
  345. 345 
  346. 346     if (count($this->words) == 0) {
  347. 347       form_set_error('keys', format_plural(variable_get('minimum_word_size', 3), 'You must include at least one positive keyword with 1 character or more.', 'You must include at least one positive keyword with @count characters or more.'));
  348. 348       return FALSE;
  349. 349     }
  350. 350     if ($this->expressionsIgnored) {
  351. 351       drupal_set_message(t('Your search used too many AND/OR expressions. Only the first @count terms were included in this search.', array('@count' => variable_get('search_and_or_limit', 7))), 'warning');
  352. 352     }
  353. 353     $this->executedFirstPass = TRUE;
  354. 354 
  355. 355     if (!empty($this->words)) {
  356. 356       $or = db_or();
  357. 357       foreach ($this->words as $word) {
  358. 358         $or->condition('i.word', $word);
  359. 359       }
  360. 360       $this->condition($or);
  361. 361     }
  362. 362     // Build query for keyword normalization.
  363. 363     $this->join('search_total', 't', 'i.word = t.word');
  364. 364     $this
  365. 365       ->condition('i.type', $this->type)
  366. 366       ->groupBy('i.type')
  367. 367       ->groupBy('i.sid')
  368. 368       ->having('COUNT(*) >= :matches', array(':matches' => $this->matches));
  369. 369 
  370. 370     // Clone the query object to do the firstPass query;
  371. 371     $first = clone $this->query;
  372. 372 
  373. 373     // For complex search queries, add the LIKE conditions to the first pass query.
  374. 374     if (!$this->simple) {
  375. 375       $first->join('search_dataset', 'd', 'i.sid = d.sid AND i.type = d.type');
  376. 376       $first->condition($this->conditions);
  377. 377     }
  378. 378 
  379. 379     // Calculate maximum keyword relevance, to normalize it.
  380. 380     $first->addExpression('SUM(i.score * t.count)', 'calculated_score');
  381. 381     $this->normalize = $first
  382. 382       ->range(0, 1)
  383. 383       ->orderBy('calculated_score', 'DESC')
  384. 384       ->execute()
  385. 385       ->fetchField();
  386. 386 
  387. 387     if ($this->normalize) {
  388. 388       return TRUE;
  389. 389     }
  390. 390     return FALSE;
  391. 391   }

  393. 393   /**
  394. 394    * Adds a custom score expression to the search query.
  395. 395    *
  396. 396    * Score expressions are used to order search results. If no calls to
  397. 397    * addScore() have taken place, a default keyword relevance score will be
  398. 398    * used. However, if at least one call to addScore() has taken place, the
  399. 399    * keyword relevance score is not automatically added.
  400. 400    *
  401. 401    * Also note that if you call orderBy() directly on the query, search scores
  402. 402    * will not automatically be used to order search results. Your orderBy()
  403. 403    * expression can reference 'calculated_score', which will be the total
  404. 404    * calculated score value.
  405. 405    *
  406. 406    * @param $score
  407. 407    *   The score expression, which should evaluate to a number between 0 and 1.
  408. 408    *   The string 'i.relevance' in a score expression will be replaced by a
  409. 409    *   measure of keyword relevance between 0 and 1.
  410. 410    * @param $arguments
  411. 411    *   Query arguments needed to provide values to the score expression.
  412. 412    * @param $multiply
  413. 413    *   If set, the score is multiplied with this value. However, all scores
  414. 414    *   with multipliers are then divided by the total of all multipliers, so
  415. 415    *   that overall, the normalization is maintained.
  416. 416    *
  417. 417    * @return object
  418. 418    *   The updated query object.
  419. 419    */
  420. 420   public function addScore($score, $arguments = array(), $multiply = FALSE) {
  421. 421     if ($multiply) {
  422. 422       $i = count($this->multiply);
  423. 423       // Modify the score expression so it is multiplied by the multiplier,
