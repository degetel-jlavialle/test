  23.   23 /**
  24.   24  * Creates a field.
  25.   25  *
  26.   26  * This function does not bind the field to any bundle; use
  27.   27  * field_create_instance() for that.
  28.   28  *
  29.   29  * @param $field
  30.   30  *   A field definition array. The field_name and type properties are required.
  31.   31  *   Other properties, if omitted, will be given the following default values:
  32.   32  *   - cardinality: 1
  33.   33  *   - locked: FALSE
  34.   34  *   - indexes: the field-type indexes, specified by the field type's
  35.   35  *     hook_field_schema(). The indexes specified in $field are added
  36.   36  *     to those default indexes. It is possible to override the
  37.   37  *     definition of a field-type index by providing an index with the
  38.   38  *     same name, or to remove it by redefining it as an empty array
  39.   39  *     of columns. Overriding field-type indexes should be done
  40.   40  *     carefully, for it might seriously affect the site's performance.
  41.   41  *   - settings: each omitted setting is given the default value defined in
  42.   42  *     hook_field_info().
  43.   43  *   - storage:
  44.   44  *     - type: the storage backend specified in the 'field_storage_default'
  45.   45  *       system variable.
  46.   46  *     - settings: each omitted setting is given the default value specified in
  47.   47  *       hook_field_storage_info().
  48.   48  *
  49.   49  * @return
  50.   50  *   The $field array with the id property filled in.
  51.   51  *
  52.   52  * @throws FieldException
  53.   53  *
  54.   54  * See: @link field Field API data structures @endlink.
  55.   55  */
  56.   56 function field_create_field($field) {
  57.   57   // Field name is required.
  58.   58   if (empty($field['field_name'])) {
  59.   59     throw new FieldException('Attempt to create an unnamed field.');
  60.   60   }
  61.   61   // Field type is required.
  62.   62   if (empty($field['type'])) {
  63.   63     throw new FieldException('Attempt to create a field with no type.');
  64.   64   }
  65.   65   // Field name cannot contain invalid characters.
  66.   66   if (!preg_match('/^[_a-z]+[_a-z0-9]*$/', $field['field_name'])) {
  67.   67     throw new FieldException('Attempt to create a field with invalid characters. Only lowercase alphanumeric characters and underscores are allowed, and only lowercase letters and underscore are allowed as the first character');
  68.   68   }
  69.   69 
  70.   70   // Field name cannot be longer than 32 characters. We use drupal_strlen()
  71.   71   // because the DB layer assumes that column widths are given in characters,
  72.   72   // not bytes.
  73.   73   if (drupal_strlen($field['field_name']) > 32) {
  74.   74     throw new FieldException(t('Attempt to create a field with a name longer than 32 characters: %name',
  75.   75       array('%name' => $field['field_name'])));
  76.   76   }
  77.   77 
  78.   78   // Ensure the field name is unique over active and disabled fields.
  79.   79   // We do not care about deleted fields.
  80.   80   $prior_field = field_read_field($field['field_name'], array('include_inactive' => TRUE));
  81.   81   if (!empty($prior_field)) {
  82.   82     $message = $prior_field['active']?
  83.   83       t('Attempt to create field name %name which already exists and is active.', array('%name' => $field['field_name'])):
  84.   84       t('Attempt to create field name %name which already exists, although it is inactive.', array('%name' => $field['field_name']));
  85.   85     throw new FieldException($message);
  86.   86   }
  87.   87 
  88.   88   // Disallow reserved field names. This can't prevent all field name
  89.   89   // collisions with existing entity properties, but some is better
  90.   90   // than none.
  91.   91   foreach (entity_get_info() as $type => $info) {
  92.   92     if (in_array($field['field_name'], $info['entity keys'])) {
  93.   93       throw new FieldException(t('Attempt to create field name %name which is reserved by entity type %type.', array('%name' => $field['field_name'], '%type' => $type)));
  94.   94     }
  95.   95   }
  96.   96 
  97.   97   $field += array(
  98.   98     'entity_types' => array(),
  99.   99     'cardinality' => 1,
  100.  100     'translatable' => FALSE,
  101.  101     'locked' => FALSE,
  102.  102     'settings' => array(),
  103.  103     'storage' => array(),
  104.  104     'deleted' => 0,
  105.  105   );
  106.  106 
  107.  107   // Check that the field type is known.
  108.  108   $field_type = field_info_field_types($field['type']);
  109.  109   if (!$field_type) {
  110.  110     throw new FieldException(t('Attempt to create a field of unknown type %type.', array('%type' => $field['type'])));
  111.  111   }
  112.  112   // Create all per-field-type properties (needed here as long as we have
  113.  113   // settings that impact column definitions).
  114.  114   $field['settings'] += field_info_field_settings($field['type']);
  115.  115   $field['module'] = $field_type['module'];
  116.  116   $field['active'] = 1;
  117.  117 
  118.  118   // Provide default storage.
  119.  119   $field['storage'] += array(
  120.  120     'type' => variable_get('field_storage_default', 'field_sql_storage'),
  121.  121     'settings' => array(),
  122.  122   );
  123.  123   // Check that the storage type is known.
  124.  124   $storage_type = field_info_storage_types($field['storage']['type']);
  125.  125   if (!$storage_type) {
  126.  126     throw new FieldException(t('Attempt to create a field with unknown storage type %type.', array('%type' => $field['storage']['type'])));
  127.  127   }
  128.  128   // Provide default storage settings.
  129.  129   $field['storage']['settings'] += field_info_storage_settings($field['storage']['type']);
  130.  130   $field['storage']['module'] = $storage_type['module'];
  131.  131   $field['storage']['active'] = 1;
  132.  132   // Collect storage information.
  133.  133   module_load_install($field['module']);
  134.  134   $schema = (array) module_invoke($field['module'], 'field_schema', $field);
  135.  135   $schema += array('columns' => array(), 'indexes' => array(), 'foreign keys' => array());
  136.  136   // 'columns' are hardcoded in the field type.
  137.  137   $field['columns'] = $schema['columns'];
  138.  138   // 'foreign keys' are hardcoded in the field type.
  139.  139   $field['foreign keys'] = $schema['foreign keys'];
  140.  140   // 'indexes' can be both hardcoded in the field type, and specified in the
  141.  141   // incoming $field definition.
  142.  142   $field += array(
  143.  143     'indexes' => array(),
  144.  144   );
  145.  145   $field['indexes'] += $schema['indexes'];
  146.  146 
  147.  147   // The serialized 'data' column contains everything from $field that does not
  148.  148   // have its own column and is not automatically populated when the field is
  149.  149   // read.
  150.  150   $data = $field;
  151.  151   unset($data['columns'], $data['field_name'], $data['type'], $data['active'], $data['module'], $data['storage_type'], $data['storage_active'], $data['storage_module'], $data['locked'], $data['cardinality'], $data['deleted']);
  152.  152   // Additionally, do not save the 'bundles' property populated by
  153.  153   // field_info_field().
  154.  154   unset($data['bundles']);
  155.  155 
  156.  156   $record = array(
  157.  157     'field_name' => $field['field_name'],
  158.  158     'type' => $field['type'],
  159.  159     'module' => $field['module'],
  160.  160     'active' => $field['active'],
  161.  161     'storage_type' => $field['storage']['type'],
  162.  162     'storage_module' => $field['storage']['module'],
  163.  163     'storage_active' => $field['storage']['active'],
  164.  164     'locked' => $field['locked'],
  165.  165     'data' => $data,
  166.  166     'cardinality' => $field['cardinality'],
  167.  167     'translatable' => $field['translatable'],
  168.  168     'deleted' => $field['deleted'],
  169.  169   );
  170.  170 
  171.  171   // Store the field and get the id back.
  172.  172   drupal_write_record('field_config', $record);
  173.  173   $field['id'] = $record['id'];
  174.  174 
  175.  175   // Invoke hook_field_storage_create_field after the field is
  176.  176   // complete (e.g. it has its id).
  177.  177   try {
  178.  178     // Invoke hook_field_storage_create_field after
  179.  179     // drupal_write_record() sets the field id.
  180.  180     module_invoke($storage_type['module'], 'field_storage_create_field', $field);
  181.  181   }
  182.  182   catch (Exception $e) {
  183.  183     // If storage creation failed, remove the field_config record before
  184.  184     // rethrowing the exception.
  185.  185     db_delete('field_config')
  186.  186       ->condition('id', $field['id'])
  187.  187       ->execute();
  188.  188     throw $e;
  189.  189   }
  190.  190 
  191.  191   // Clear caches
  192.  192   field_cache_clear(TRUE);
  193.  193 
  194.  194   // Invoke external hooks after the cache is cleared for API consistency.
  195.  195   module_invoke_all('field_create_field', $field);
  196.  196 
  197.  197   return $field;
  198.  198 }

  200.  200 /**
  201.  201  * Updates a field.
  202.  202  *
  203.  203  * Any module may forbid any update for any reason. For example, the
  204.  204  * field's storage module might forbid an update if it would change
  205.  205  * the storage schema while data for the field exists. A field type
  206.  206  * module might forbid an update if it would change existing data's
  207.  207  * semantics, or if there are external dependencies on field settings
  208.  208  * that cannot be updated.
  209.  209  *
  210.  210  * @param $field
  211.  211  *   A field structure. $field['field_name'] must provided; it
  212.  212  *   identifies the field that will be updated to match this
  213.  213  *   structure. Any other properties of the field that are not
  214.  214  *   specified in $field will be left unchanged, so it is not
  215.  215  *   necessary to pass in a fully populated $field structure.
  216.  216  * @return
  217.  217  *   Throws a FieldException if the update cannot be performed.
  218.  218  * @see field_create_field()
  219.  219  */
  220.  220 function field_update_field($field) {
  221.  221   // Check that the specified field exists.
  222.  222   $prior_field = field_read_field($field['field_name']);
  223.  223   if (empty($prior_field)) {
  224.  224     throw new FieldException('Attempt to update a non-existent field.');
  225.  225   }
  226.  226 
  227.  227   // Use the prior field values for anything not specifically set by the new
  228.  228   // field to be sure that all values are set.
  229.  229   $field += $prior_field;
  230.  230   $field['settings'] += $prior_field['settings'];
  231.  231 
  232.  232   // Some updates are always disallowed.
  233.  233   if ($field['type'] != $prior_field['type']) {
  234.  234     throw new FieldException("Cannot change an existing field's type.");
  235.  235   }
  236.  236   if ($field['entity_types'] != $prior_field['entity_types']) {
  237.  237     throw new FieldException("Cannot change an existing field's entity_types property.");
  238.  238   }
  239.  239   if ($field['storage']['type'] != $prior_field['storage']['type']) {
  240.  240     throw new FieldException("Cannot change an existing field's storage type.");
  241.  241   }
  242.  242 
  243.  243   // Collect the new storage information, since what is in
  244.  244   // $prior_field may no longer be right.
  245.  245   module_load_install($field['module']);
  246.  246   $schema = (array) module_invoke($field['module'], 'field_schema', $field);
  247.  247   $schema += array('columns' => array(), 'indexes' => array(), 'foreign keys' => array());
  248.  248   // 'columns' are hardcoded in the field type.
  249.  249   $field['columns'] = $schema['columns'];
  250.  250   // 'foreign keys' are hardcoded in the field type.
  251.  251   $field['foreign keys'] = $schema['foreign keys'];
  252.  252   // 'indexes' can be both hardcoded in the field type, and specified in the
  253.  253   // incoming $field definition.
  254.  254   $field += array(
  255.  255     'indexes' => array(),
  256.  256   );
  257.  257   $field['indexes'] += $schema['indexes'];
  258.  258 
  259.  259   $has_data = field_has_data($field);
  260.  260 
  261.  261   // See if any module forbids the update by throwing an exception.
  262.  262   foreach (module_implements('field_update_forbid') as $module) {
  263.  263     $function = $module . '_field_update_forbid';
  264.  264     $function($field, $prior_field, $has_data);
  265.  265   }
  266.  266 
  267.  267   // Tell the storage engine to update the field. Do this before
  268.  268   // saving the new definition since it still might fail.
  269.  269   $storage_type = field_info_storage_types($field['storage']['type']);
  270.  270   module_invoke($storage_type['module'], 'field_storage_update_field', $field, $prior_field, $has_data);
  271.  271 
  272.  272   // Save the new field definition. @todo: refactor with
  273.  273   // field_create_field.
  274.  274 
  275.  275   // The serialized 'data' column contains everything from $field that does not
  276.  276   // have its own column and is not automatically populated when the field is
  277.  277   // read.
  278.  278   $data = $field;
  279.  279   unset($data['columns'], $data['field_name'], $data['type'], $data['locked'], $data['module'], $data['cardinality'], $data['active'], $data['deleted']);
  280.  280   // Additionally, do not save the 'bundles' property populated by
  281.  281   // field_info_field().
  282.  282   unset($data['bundles']);
  283.  283 
  284.  284   $field['data'] = $data;
  285.  285 
  286.  286   // Store the field and create the id.
  287.  287   $primary_key = array('id');
  288.  288   drupal_write_record('field_config', $field, $primary_key);
  289.  289 
  290.  290   // Clear caches
  291.  291   field_cache_clear(TRUE);
  292.  292 
  293.  293   // Invoke external hooks after the cache is cleared for API consistency.
  294.  294   module_invoke_all('field_update_field', $field, $prior_field, $has_data);
  295.  295 }

  297.  297 /**
  298.  298  * Reads a single field record directly from the database.
  299.  299  *
  300.  300  * Generally, you should use the field_info_field() instead.
  301.  301  *
  302.  302  * This function will not return deleted fields. Use
  303.  303  * field_read_fields() instead for this purpose.
  304.  304  *
  305.  305  * @param $field_name
  306.  306  *   The field name to read.
  307.  307  * @param array $include_additional
  308.  308  *   The default behavior of this function is to not return a field that
  309.  309  *   is inactive. Setting
  310.  310  *   $include_additional['include_inactive'] to TRUE will override this
  311.  311  *   behavior.
  312.  312  * @return
  313.  313  *   A field definition array, or FALSE.
  314.  314  */
  315.  315 function field_read_field($field_name, $include_additional = array()) {
  316.  316   $fields = field_read_fields(array('field_name' => $field_name), $include_additional);
  317.  317   return $fields ? current($fields) : FALSE;
  318.  318 }

  320.  320 /**
  321.  321  * Reads in fields that match an array of conditions.
  322.  322  *
  323.  323  * @param array $params
  324.  324  *   An array of conditions to match against. Keys are columns from the
  325.  325  *   'field_config' table, values are conditions to match. Additionally,
  326.  326  *   conditions on the 'entity_type' and 'bundle' columns from the
  327.  327  *   'field_config_instance' table are supported (select fields having an
  328.  328  *   instance on a given bundle).
  329.  329  * @param array $include_additional
  330.  330  *   The default behavior of this function is to not return fields that
  331.  331  *   are inactive or have been deleted. Setting
  332.  332  *   $include_additional['include_inactive'] or
  333.  333  *   $include_additional['include_deleted'] to TRUE will override this
  334.  334  *   behavior.
  335.  335  * @return
  336.  336  *   An array of fields matching $params. If
  337.  337  *   $include_additional['include_deleted'] is TRUE, the array is keyed
  338.  338  *   by field id, otherwise it is keyed by field name.
  339.  339  */
  340.  340 function field_read_fields($params = array(), $include_additional = array()) {
  341.  341   $query = db_select('field_config', 'fc', array('fetch' => PDO::FETCH_ASSOC));
  342.  342   $query->fields('fc');
  343.  343 
  344.  344   // Turn the conditions into a query.
  345.  345   foreach ($params as $key => $value) {
  346.  346     // Allow filtering on the 'entity_type' and 'bundle' columns of the
  347.  347     // field_config_instance table.
  348.  348     if ($key == 'entity_type' || $key == 'bundle') {
  349.  349       if (empty($fci_join)) {
  350.  350         $fci_join = $query->join('field_config_instance', 'fci', 'fc.id = fci.field_id');
  351.  351       }
  352.  352       $key = 'fci.' . $key;
  353.  353     }
  354.  354     else {
  355.  355       $key = 'fc.' . $key;
  356.  356     }
  357.  357 
  358.  358     $query->condition($key, $value);
  359.  359   }
  360.  360 
  361.  361   if (!isset($include_additional['include_inactive']) || !$include_additional['include_inactive']) {
  362.  362     $query
  363.  363       ->condition('fc.active', 1)
  364.  364       ->condition('fc.storage_active', 1);
  365.  365   }
  366.  366   $include_deleted = (isset($include_additional['include_deleted']) && $include_additional['include_deleted']);
  367.  367   if (!$include_deleted) {
  368.  368     $query->condition('fc.deleted', 0);
  369.  369   }
  370.  370 
  371.  371   $fields = array();
  372.  372   $results = $query->execute();
  373.  373   foreach ($results as $record) {
  374.  374     $field = unserialize($record['data']);
  375.  375     $field['id'] = $record['id'];
  376.  376     $field['field_name'] = $record['field_name'];
  377.  377     $field['type'] = $record['type'];
  378.  378     $field['module'] = $record['module'];
  379.  379     $field['active'] = $record['active'];
  380.  380     $field['storage']['type'] = $record['storage_type'];
  381.  381     $field['storage']['module'] = $record['storage_module'];
  382.  382     $field['storage']['active'] = $record['storage_active'];
  383.  383     $field['locked'] = $record['locked'];
  384.  384     $field['cardinality'] = $record['cardinality'];
  385.  385     $field['translatable'] = $record['translatable'];
  386.  386     $field['deleted'] = $record['deleted'];
  387.  387 
  388.  388     module_invoke_all('field_read_field', $field);
  389.  389 
  390.  390     // Populate storage information.
  391.  391     module_load_install($field['module']);
  392.  392     $schema = (array) module_invoke($field['module'], 'field_schema', $field);
  393.  393     $schema += array('columns' => array(), 'indexes' => array());
  394.  394     $field['columns'] = $schema['columns'];
  395.  395 
  396.  396     $field_name = $field['field_name'];
  397.  397     if ($include_deleted) {
  398.  398       $field_name = $field['id'];
  399.  399     }
  400.  400     $fields[$field_name] = $field;
  401.  401   }
  402.  402   return $fields;
  403.  403 }

  405.  405 /**
  406.  406  * Marks a field and its instances and data for deletion.
  407.  407  *
  408.  408  * @param $field_name
  409.  409  *   The field name to delete.
  410.  410  */
  411.  411 function field_delete_field($field_name) {
  412.  412   // Delete all non-deleted instances.
  413.  413   $field = field_info_field($field_name);
  414.  414   if (isset($field['bundles'])) {
  415.  415     foreach ($field['bundles'] as $entity_type => $bundles) {
  416.  416       foreach ($bundles as $bundle) {
  417.  417         $instance = field_info_instance($entity_type, $field_name, $bundle);
  418.  418         field_delete_instance($instance, FALSE);
  419.  419       }
  420.  420     }
  421.  421   }
  422.  422 
  423.  423   // Mark field data for deletion.
  424.  424   module_invoke($field['storage']['module'], 'field_storage_delete_field', $field);
  425.  425 
  426.  426   // Mark the field for deletion.
  427.  427   db_update('field_config')
  428.  428     ->fields(array('deleted' => 1))
  429.  429     ->condition('field_name', $field_name)
  430.  430     ->execute();
  431.  431 
  432.  432   // Clear the cache.
  433.  433   field_cache_clear(TRUE);
  434.  434 
  435.  435   module_invoke_all('field_delete_field', $field);
  436.  436 }

  438.  438 /**
  439.  439  * Creates an instance of a field, binding it to a bundle.
  440.  440  *
  441.  441  * @param $instance
  442.  442  *   A field instance definition array. The field_name, entity_type and
  443.  443  *   bundle properties are required. Other properties, if omitted,
  444.  444  *   will be given the following default values:
  445.  445  *   - label: the field name
  446.  446  *   - description: empty string
  447.  447  *   - required: FALSE
  448.  448  *   - default_value_function: empty string
  449.  449  *   - settings: each omitted setting is given the default value specified in
  450.  450  *     hook_field_info().
  451.  451  *   - widget:
  452.  452  *     - type: the default widget specified in hook_field_info().
  453.  453  *     - settings: each omitted setting is given the default value specified in
  454.  454  *       hook_field_widget_info().
  455.  455  *   - display:
  456.  456  *     Settings for the 'default' view mode will be added if not present, and
  457.  457  *     each view mode in the definition will be completed with the following
  458.  458  *     default values:
  459.  459  *     - label: 'above'
  460.  460  *     - type: the default formatter specified in hook_field_info().
  461.  461  *     - settings: each omitted setting is given the default value specified in
  462.  462  *       hook_field_formatter_info().
  463.  463  *     View modes not present in the definition are left empty, and the field
  464.  464  *     will not be displayed in this mode.
  465.  465  *
  466.  466  * @return
  467.  467  *   The $instance array with the id property filled in.
  468.  468  *
  469.  469  * @throws FieldException
  470.  470  *
  471.  471  * See: @link field Field API data structures @endlink.
  472.  472  */
  473.  473 function field_create_instance($instance) {
  474.  474   $field = field_read_field($instance['field_name']);
  475.  475   if (empty($field)) {
  476.  476     throw new FieldException(t("Attempt to create an instance of a field @field_name that doesn't exist or is currently inactive.", array('@field_name' => $instance['field_name'])));
  477.  477   }
  478.  478   // Check that the required properties exists.
  479.  479   if (empty($instance['entity_type'])) {
  480.  480     throw new FieldException(t('Attempt to create an instance of field @field_name without an entity type.', array('@field_name' => $instance['field_name'])));
  481.  481   }
  482.  482   if (empty($instance['bundle'])) {
  483.  483     throw new FieldException(t('Attempt to create an instance of field @field_name without a bundle.', array('@field_name' => $instance['field_name'])));
  484.  484   }
  485.  485   // Check that the field can be attached to this entity type.
  486.  486   if (!empty($field['entity_types']) && !in_array($instance['entity_type'], $field['entity_types'])) {
  487.  487     throw new FieldException(t('Attempt to create an instance of field @field_name on forbidden entity type @entity_type.', array('@field_name' => $instance['field_name'], '@entity_type' => $instance['entity_type'])));
  488.  488   }
  489.  489 
  490.  490   // Set the field id.
  491.  491   $instance['field_id'] = $field['id'];
  492.  492 
  493.  493   // Note that we do *not* prevent creating a field on non-existing bundles,
  494.  494   // because that would break the 'Body as field' upgrade for contrib
  495.  495   // node types.
  496.  496 
  497.  497   // TODO: Check that the widget type is known and can handle the field type ?
  498.  498   // TODO: Check that the formatters are known and can handle the field type ?
  499.  499   // TODO: Check that the display view modes are known for the entity type ?
  500.  500   // Those checks should probably happen in _field_write_instance() ?
  501.  501   // Problem : this would mean that a UI module cannot update an instance with a disabled formatter.
  502.  502 
  503.  503   // Ensure the field instance is unique within the bundle.
  504.  504   // We only check for instances of active fields, since adding an instance of
  505.  505   // a disabled field is not supported.
  506.  506   $prior_instance = field_read_instance($instance['entity_type'], $instance['field_name'], $instance['bundle']);
  507.  507   if (!empty($prior_instance)) {
  508.  508     $message = t('Attempt to create an instance of field @field_name on bundle @bundle that already has an instance of that field.', array('@field_name' => $instance['field_name'], '@bundle' => $instance['bundle']));
  509.  509     throw new FieldException($message);
  510.  510   }
  511.  511 
  512.  512   _field_write_instance($instance);
  513.  513 
  514.  514   // Clear caches
  515.  515   field_cache_clear();
  516.  516 
  517.  517   // Invoke external hooks after the cache is cleared for API consistency.
  518.  518   module_invoke_all('field_create_instance', $instance);
  519.  519 
  520.  520   return $instance;
  521.  521 }
  522.  522 
  523.  523 /**
  524.  524  * Updates an instance of a field.
  525.  525  *
  526.  526  * @param $instance
  527.  527  *   An associative array representing an instance structure. The following
  528.  528  *   required array elements specify which field instance is being updated:
  529.  529  *   - entity_type: The type of the entity the field is attached to.
  530.  530  *   - bundle: The bundle this field belongs to.
  531.  531  *   - field_name: The name of an existing field.
  532.  532  *   The other array elements represent properties of the instance, and all
  533.  533  *   properties must be specified or their default values will be used (except
  534.  534  *   internal-use properties, which are assigned automatically). To avoid
  535.  535  *   losing the previously stored properties of the instance when making a
  536.  536  *   change, first load the instance with field_info_instance(), then override
  537.  537  *   the values you want to override, and finally save using this function.
  538.  538  *   Example:
  539.  539  *   @code
  540.  540  *   // Fetch an instance info array.
  541.  541  *   $instance_info = field_info_instance($entity_type, $field_name, $bundle_name);
  542.  542  *   // Change a single property in the instance definition.
  543.  543  *   $instance_info['required'] = TRUE;
  544.  544  *   // Write the changed definition back.
  545.  545  *   field_update_instance($instance_info);
  546.  546  *   @endcode
  547.  547  *
  548.  548  * @throws FieldException
  549.  549  *
  550.  550  * @see field_info_instance()
  551.  551  * @see field_create_instance()
  552.  552  */
  553.  553 function field_update_instance($instance) {
  554.  554   // Check that the specified field exists.
  555.  555   $field = field_read_field($instance['field_name']);
  556.  556   if (empty($field)) {
  557.  557     throw new FieldException(t('Attempt to update an instance of a nonexistent field @field.', array('@field' => $instance['field_name'])));
  558.  558   }
  559.  559 
  560.  560   // Check that the field instance exists (even if it is inactive, since we
  561.  561   // want to be able to replace inactive widgets with new ones).
  562.  562   $prior_instance = field_read_instance($instance['entity_type'], $instance['field_name'], $instance['bundle'], array('include_inactive' => TRUE));
  563.  563   if (empty($prior_instance)) {
  564.  564     throw new FieldException(t("Attempt to update an instance of field @field on bundle @bundle that doesn't exist.", array('@field' => $instance['field_name'], '@bundle' => $instance['bundle'])));
  565.  565   }
  566.  566 
  567.  567   $instance['id'] = $prior_instance['id'];
  568.  568   $instance['field_id'] = $prior_instance['field_id'];
  569.  569 
  570.  570   _field_write_instance($instance, TRUE);
  571.  571 
  572.  572   // Clear caches.
  573.  573   field_cache_clear();
  574.  574 
  575.  575   module_invoke_all('field_update_instance', $instance, $prior_instance);
  576.  576 }
  577.  577 
  578.  578 /**
  579.  579  * Stores an instance record in the field configuration database.
  580.  580  *
  581.  581  * @param $instance
  582.  582  *   An instance structure.
  583.  583  * @param $update
  584.  584  *   Whether this is a new or existing instance.
  585.  585  */
  586.  586 function _field_write_instance($instance, $update = FALSE) {
  587.  587   $field = field_read_field($instance['field_name']);
  588.  588   $field_type = field_info_field_types($field['type']);
  589.  589 
  590.  590   // Set defaults.
  591.  591   $instance += array(
  592.  592     'settings' => array(),
  593.  593     'display' => array(),
  594.  594     'widget' => array(),
  595.  595     'required' => FALSE,
  596.  596     'label' => $instance['field_name'],
  597.  597     'description' => '',
  598.  598     'deleted' => 0,
  599.  599   );
  600.  600 
  601.  601   // Set default instance settings.
  602.  602   $instance['settings'] += field_info_instance_settings($field['type']);
  603.  603 
  604.  604   // Set default widget and settings.
  605.  605   $instance['widget'] += array(
  606.  606     // TODO: what if no 'default_widget' specified ?
  607.  607     'type' => $field_type['default_widget'],
  608.  608     'settings' => array(),
  609.  609   );
  610.  610   // If no weight specified, make sure the field sinks at the bottom.
  611.  611   if (!isset($instance['widget']['weight'])) {
  612.  612     $max_weight = field_info_max_weight($instance['entity_type'], $instance['bundle'], 'form');
  613.  613     $instance['widget']['weight'] = isset($max_weight) ? $max_weight + 1 : 0;
  614.  614   }
  615.  615   // Check widget module.
  616.  616   $widget_type = field_info_widget_types($instance['widget']['type']);
  617.  617   $instance['widget']['module'] = $widget_type['module'];
  618.  618   $instance['widget']['settings'] += field_info_widget_settings($instance['widget']['type']);
  619.  619 
  620.  620   // Make sure there are at least display settings for the 'default' view mode,
  621.  621   // and fill in defaults for each view mode specified in the definition.
  622.  622   $instance['display'] += array(
  623.  623     'default' => array(),
  624.  624   );
  625.  625   foreach ($instance['display'] as $view_mode => $display) {
  626.  626     $display += array(
  627.  627       'label' => 'above',
  628.  628       'type' => isset($field_type['default_formatter']) ? $field_type['default_formatter'] : 'hidden',
  629.  629       'settings' => array(),
  630.  630     );
  631.  631     if ($display['type'] != 'hidden') {
  632.  632       $formatter_type = field_info_formatter_types($display['type']);
  633.  633       $display['module'] = $formatter_type['module'];
  634.  634       $display['settings'] += field_info_formatter_settings($display['type']);
  635.  635     }
  636.  636     // If no weight specified, make sure the field sinks at the bottom.
  637.  637     if (!isset($display['weight'])) {
  638.  638       $max_weight = field_info_max_weight($instance['entity_type'], $instance['bundle'], $view_mode);
  639.  639       $display['weight'] = isset($max_weight) ? $max_weight + 1 : 0;
  640.  640     }
  641.  641     $instance['display'][$view_mode] = $display;
  642.  642   }
  643.  643 
  644.  644   // The serialized 'data' column contains everything from $instance that does
  645.  645   // not have its own column and is not automatically populated when the
  646.  646   // instance is read.
  647.  647   $data = $instance;
  648.  648   unset($data['id'], $data['field_id'], $data['field_name'], $data['entity_type'], $data['bundle'], $data['deleted']);
  649.  649 
  650.  650   $record = array(
  651.  651     'field_id' => $instance['field_id'],
  652.  652     'field_name' => $instance['field_name'],
  653.  653     'entity_type' => $instance['entity_type'],
  654.  654     'bundle' => $instance['bundle'],
  655.  655     'data' => $data,
  656.  656     'deleted' => $instance['deleted'],
  657.  657   );
  658.  658   // We need to tell drupal_update_record() the primary keys to trigger an
  659.  659   // update.
  660.  660   if ($update) {
  661.  661     $record['id'] = $instance['id'];
  662.  662     $primary_key = array('id');
  663.  663   }
  664.  664   else {
  665.  665     $primary_key = array();
  666.  666   }
  667.  667   drupal_write_record('field_config_instance', $record, $primary_key);
  668.  668 }
  669.  669 
  670.  670 /**
  671.  671  * Reads a single instance record from the database.
  672.  672  *
  673.  673  * Generally, you should use field_info_instance() instead, as it
  674.  674  * provides caching and allows other modules the opportunity to
  675.  675  * append additional formatters, widgets, and other information.
  676.  676  *
  677.  677  * @param $entity_type
  678.  678  *   The type of entity to which the field is bound.
  679.  679  * @param $field_name
  680.  680  *   The field name to read.
  681.  681  * @param $bundle
  682.  682  *   The bundle to which the field is bound.
  683.  683  * @param array $include_additional
  684.  684  *   The default behavior of this function is to not return an instance that
  685.  685  *   has been deleted, or whose field is inactive. Setting
  686.  686  *   $include_additional['include_inactive'] or
  687.  687  *   $include_additional['include_deleted'] to TRUE will override this
  688.  688  *   behavior.
  689.  689  * @return

  797.  797 /**
