  42.  42 function field_default_form($entity_type, $entity, $field, $instance, $langcode, $items, &$form, &$form_state, $get_delta = NULL) {
  43.  43   // This could be called with no entity, as when a UI module creates a

  48.  48

  50.  50

  58.  58   }
  59.  59 
  60.  60   // Let modules alter the widget properties.
  61.  61   $context = array(
  62.  62     'entity_type' => $entity_type,
  63.  63     'entity' => $entity,
  64.  64     'field' => $field,

  66.  66   );

  68.  68 
  69.  69   // Collect widget elements.
  70.  70   $elements = array();
  71.  71 
  72.  72   // Store field information in $form_state.
  73.  73   if (!field_form_get_state($parents, $field_name, $langcode, $form_state)) {
  74.  74     $field_state = array(
  75.  75       'field' => $field,

  78.  78       'array_parents' => array(),

  83.  83

  85.  85   // displaying an individual element, process the multiple value form.

  87.  87     // Store the entity in the form.

  92.  92   // displaying an individual element, just get a single form element and make
  93.  93   // it the $delta value.
  94.  94   else {
  95.  95     $delta = isset($get_delta) ? $get_delta : 0;
  96.  96     $function = $instance['widget']['module'] . '_field_widget_form';
  97.  97     if (function_exists($function)) {
  98.  98       $element = array(
  99.  99         '#entity' => $entity,
  100. 100         '#entity_type' => $instance['entity_type'],

  103. 103         '#language' => $langcode,
  104. 104         '#field_parents' => $parents,
  105. 105         '#columns' => array_keys($field['columns']),
  106. 106         '#title' => check_plain($instance['label']),
  107. 107         '#description' => field_filter_xss($instance['description']),
  108. 108         // Only the first widget should be required.
  109. 109         '#required' => $delta == 0 && $instance['required'],
  110. 110         '#delta' => $delta,
  111. 111       );
  112. 112       if ($element = $function($form, $form_state, $field, $instance, $langcode, $items, $delta, $element)) {
  113. 113         // Allow modules to alter the field widget form element.
  114. 114         $context = array(
  115. 115           'form' => $form,
  116. 116           'field' => $field,
  117. 117           'instance' => $instance,
  118. 118           'langcode' => $langcode,
  119. 119           'items' => $items,
  120. 120           'delta' => $delta,

  129. 129         if (field_behaviors_widget('multiple values', $instance) == FIELD_BEHAVIOR_DEFAULT) {

  131. 131         }
  132. 132         else {

  138. 138 
  139. 139   // Also aid in theming of field widgets by rendering a classified container.
  140. 140   $addition[$field_name] = array(
  141. 141     '#type' => 'container',
  142. 142     '#attributes' => array(

  144. 144         'field-type-' . drupal_html_class($field['type']),

  148. 148     ),

  151. 151 
  152. 152   // Populate the 'array_parents' information in $form_state['field'] after
  153. 153   // the form is built, so that we catch changes in the form structure performed

  155. 155   $elements['#after_build'][] = 'field_form_element_after_build';
  156. 156   $elements['#field_name'] = $field_name;
  157. 157   $elements['#language'] = $langcode;
  158. 158   $elements['#field_parents'] = $parents;
  159. 159 
  160. 160   $addition[$field_name] += array(
  161. 161     '#tree' => TRUE,
  162. 162     // The '#language' key can be used to access the field's form element
  163. 163     // when $langcode is unknown.
  164. 164     '#language' => $langcode,
  165. 165     $langcode => $elements,
  166. 166     '#access' => field_access('edit', $field, $entity_type, $entity),
  167. 167   );
  168. 168 
  169. 169   return $addition;
  170. 170 }

  172. 172 /**
  173. 173  * Special handling to create form elements for multiple values.
  174. 174  *
  175. 175  * Handles generic features for multiple fields:
  176. 176  * - number of widgets
  177. 177  * - AHAH-'add more' button
  178. 178  * - drag-n-drop value reordering
  179. 179  */
  180. 180 function field_multiple_value_form($field, $instance, $langcode, $items, &$form, &$form_state) {
  181. 181   $field_name = $field['field_name'];
  182. 182   $parents = $form['#parents'];
  183. 183 
  184. 184   // Determine the number of widgets to display.
  185. 185   switch ($field['cardinality']) {
  186. 186     case FIELD_CARDINALITY_UNLIMITED:
  187. 187       $field_state = field_form_get_state($parents, $field_name, $langcode, $form_state);
  188. 188       $max = $field_state['items_count'];
  189. 189       break;
  190. 190 
  191. 191     default:
  192. 192       $max = $field['cardinality'] - 1;
  193. 193       break;
  194. 194   }
  195. 195 
  196. 196   $title = check_plain($instance['label']);
  197. 197   $description = field_filter_xss($instance['description']);
  198. 198 
  199. 199   $id_prefix = implode('-', array_merge($parents, array($field_name)));
  200. 200   $wrapper_id = drupal_html_id($id_prefix . '-add-more-wrapper');
  201. 201 
  202. 202   $field_elements = array();
  203. 203 
  204. 204   $function = $instance['widget']['module'] . '_field_widget_form';
  205. 205   if (function_exists($function)) {
  206. 206     for ($delta = 0; $delta <= $max; $delta++) {
  207. 207       $multiple = $field['cardinality'] > 1 || $field['cardinality'] == FIELD_CARDINALITY_UNLIMITED;
  208. 208       $element = array(
  209. 209         '#entity_type' => $instance['entity_type'],
  210. 210         '#entity' => $form['#entity'],
  211. 211         '#bundle' => $instance['bundle'],
  212. 212         '#field_name' => $field_name,
  213. 213         '#language' => $langcode,
  214. 214         '#field_parents' => $parents,
  215. 215         '#columns' => array_keys($field['columns']),
  216. 216         // For multiple fields, title and description are handled by the wrapping table.
  217. 217         '#title' => $multiple ? '' : $title,
  218. 218         '#description' => $multiple ? '' : $description,
  219. 219         // Only the first widget should be required.
  220. 220         '#required' => $delta == 0 && $instance['required'],
  221. 221         '#delta' => $delta,
  222. 222         '#weight' => $delta,
  223. 223       );
  224. 224       if ($element = $function($form, $form_state, $field, $instance, $langcode, $items, $delta, $element)) {
  225. 225         // Input field for the delta (drag-n-drop reordering).
  226. 226         if ($multiple) {
  227. 227           // We name the element '_weight' to avoid clashing with elements
  228. 228           // defined by widget.
  229. 229           $element['_weight'] = array(
  230. 230             '#type' => 'weight',
  231. 231             '#title' => t('Weight for row @number', array('@number' => $delta + 1)),
  232. 232             '#title_display' => 'invisible',
  233. 233              // Note: this 'delta' is the FAPI 'weight' element's property.
  234. 234             '#delta' => $max,
  235. 235             '#default_value' => isset($items[$delta]['_weight']) ? $items[$delta]['_weight'] : $delta,
  236. 236             '#weight' => 100,
  237. 237           );
  238. 238         }
  239. 239 
  240. 240         // Allow modules to alter the field widget form element.
  241. 241         $context = array(
  242. 242           'form' => $form,
  243. 243           'field' => $field,
  244. 244           'instance' => $instance,
  245. 245           'langcode' => $langcode,
  246. 246           'items' => $items,
  247. 247           'delta' => $delta,
  248. 248         );
  249. 249         drupal_alter(array('field_widget_form', 'field_widget_' . $instance['widget']['type'] . '_form'), $element, $form_state, $context);
  250. 250 
  251. 251         $field_elements[$delta] = $element;
  252. 252       }
  253. 253     }
  254. 254 
  255. 255     if ($field_elements) {
  256. 256       $field_elements += array(
  257. 257         '#theme' => 'field_multiple_value_form',
  258. 258         '#field_name' => $field['field_name'],
  259. 259         '#cardinality' => $field['cardinality'],
  260. 260         '#title' => $title,
  261. 261         '#required' => $instance['required'],
  262. 262         '#description' => $description,
  263. 263         '#prefix' => '<div id="' . $wrapper_id . '">',
  264. 264         '#suffix' => '</div>',
  265. 265         '#max_delta' => $max,
  266. 266       );
  267. 267       // Add 'add more' button, if not working with a programmed form.
  268. 268       if ($field['cardinality'] == FIELD_CARDINALITY_UNLIMITED && empty($form_state['programmed'])) {
  269. 269         $field_elements['add_more'] = array(
  270. 270           '#type' => 'submit',
  271. 271           '#name' => strtr($id_prefix, '-', '_') . '_add_more',
  272. 272           '#value' => t('Add another item'),
  273. 273           '#attributes' => array('class' => array('field-add-more-submit')),
  274. 274           '#limit_validation_errors' => array(array_merge($parents, array($field_name, $langcode))),
  275. 275           '#submit' => array('field_add_more_submit'),
  276. 276           '#ajax' => array(
  277. 277             'callback' => 'field_add_more_js',
  278. 278             'wrapper' => $wrapper_id,
  279. 279             'effect' => 'fade',
  280. 280           ),
  281. 281         );
  282. 282       }
  283. 283     }
  284. 284   }
  285. 285 
  286. 286   return $field_elements;
  287. 287 }

  289. 289 /**
  290. 290  * Returns HTML for an individual form element.
  291. 291  *
  292. 292  * Combine multiple values into a table with drag-n-drop reordering.
  293. 293  * TODO : convert to a template.

  301. 301 function theme_field_multiple_value_form($variables) {
  302. 302   $element = $variables['element'];
  303. 303   $output = '';
  304. 304 
  305. 305   if ($element['#cardinality'] > 1 || $element['#cardinality'] == FIELD_CARDINALITY_UNLIMITED) {
  306. 306     $table_id = drupal_html_id($element['#field_name'] . '_values');
  307. 307     $order_class = $element['#field_name'] . '-delta-order';
  308. 308     $required = !empty($element['#required']) ? theme('form_required_marker', $variables) : '';
  309. 309 
  310. 310     $header = array(
  311. 311       array(
  312. 312         'data' => '<label>' . t('!title !required', array('!title' => $element['#title'], '!required' => $required)) . "</label>",
  313. 313         'colspan' => 2,
  314. 314         'class' => array('field-label'),
  315. 315       ),
  316. 316       t('Order'),
  317. 317     );
  318. 318     $rows = array();
  319. 319 
  320. 320     // Sort items according to '_weight' (needed when the form comes back after

  322. 322     $items = array();

  324. 324       if ($key === 'add_more') {
  325. 325         $add_more_button = &$element[$key];
  326. 326       }
  327. 327       else {
  328. 328         $items[] = &$element[$key];
  329. 329       }
  330. 330     }
  331. 331     usort($items, '_field_sort_items_value_helper');
  332. 332 
  333. 333     // Add the items as table rows.
  334. 334     foreach ($items as $key => $item) {
  335. 335       $item['_weight']['#attributes']['class'] = array($order_class);
  336. 336       $delta_element = drupal_render($item['_weight']);
  337. 337       $cells = array(
  338. 338         array('data' => '', 'class' => array('field-multiple-drag')),
  339. 339         drupal_render($item),
  340. 340         array('data' => $delta_element, 'class' => array('delta-order')),
  341. 341       );

  343. 343         'data' => $cells,
  344. 344         'class' => array('draggable'),
  345. 345       );
  346. 346     }
  347. 347 
  348. 348     $output = '<div class="form-item">';
  349. 349     $output .= theme('table', array('header' => $header, 'rows' => $rows, 'attributes' => array('id' => $table_id, 'class' => array('field-multiple-table'))));
  350. 350     $output .= $element['#description'] ? '<div class="description">' . $element['#description'] . '</div>' : '';
  351. 351     $output .= '<div class="clearfix">' . drupal_render($add_more_button) . '</div>';
  352. 352     $output .= '</div>';
  353. 353 
  354. 354     drupal_add_tabledrag($table_id, 'order', 'sibling', $order_class);
  355. 355   }
  356. 356   else {
  357. 357     foreach (element_children($element) as $key) {
  358. 358       $output .= drupal_render($element[$key]);
  359. 359     }
  360. 360   }
  361. 361 
  362. 362   return $output;
  363. 363 }
