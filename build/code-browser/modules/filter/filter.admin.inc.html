  19.  19 
  20.  20   $form['#tree'] = TRUE;
  21.  21   foreach ($formats as $id => $format) {
  22.  22     // Check whether this is the fallback text format. This format is available
  23.  23     // to all roles and cannot be disabled via the admin interface.

  28.  28     }
  29.  29     else {
  30.  30       $form['formats'][$id]['name'] = array('#markup' => check_plain($format->name));
  31.  31       $roles = array_map('check_plain', filter_get_roles_by_format($format));
  32.  32       $roles_markup = $roles ? implode(', ', $roles) : t('No roles may use this format');

  38.  38       '#type' => 'weight',

  41.  41       '#default_value' => $format->weight,
  42.  42     );
  43.  43   }
