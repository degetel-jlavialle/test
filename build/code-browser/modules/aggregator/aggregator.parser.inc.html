  290. 290 /**
  291. 291  * Parses the W3C date/time format, a subset of ISO 8601.
  292. 292  *
  293. 293  * PHP date parsing functions do not handle this format. See
  294. 294  * http://www.w3.org/TR/NOTE-datetime for more information. Originally from
  295. 295  * MagpieRSS (http://magpierss.sourceforge.net/).
  296. 296  *
  297. 297  * @param $date_str

  305. 305     list($year, $month, $day, $hours, $minutes, $seconds) = array($match[1], $match[2], $match[3], $match[4], $match[5], $match[6]);
  306. 306     // Calculate the epoch for current date assuming GMT.
  307. 307     $epoch = gmmktime($hours, $minutes, $seconds, $month, $day, $year);

  309. 309       list($tz_mod, $tz_hour, $tz_min) = array($match[8], $match[9], $match[10]);

  311. 311       if (!$tz_hour) {
  312. 312         $tz_hour = 0;
  313. 313       }

  315. 315         $tz_min = 0;
  316. 316       }

  319. 319       if ($tz_mod == '+') {

  321. 321       }
  322. 322       $epoch += $offset_secs;
  323. 323     }

  325. 325   }
  326. 326   else {
  327. 327     return FALSE;
  328. 328   }
